62      @Test
63      public void testElseWithCurly() {
64          final DetailAST ast = new DetailAST();
65          ast.setType(TokenTypes.ASSIGN);
66          ast.setText("ASSIGN");
67          assertFalse("Invalid elseIf check result 'ASSIGN' is not 'else if'",
68                  CheckUtils.isElseIf(ast));
69  
70          final DetailAST parentAst = new DetailAST();
71          parentAst.setType(TokenTypes.LCURLY);
72          parentAst.setText("LCURLY");
73  
74          final DetailAST ifAst = new DetailAST();
75          ifAst.setType(TokenTypes.LITERAL_IF);
76          ifAst.setText("IF");
77          parentAst.addChild(ifAst);
78  
79          assertFalse("Invalid elseIf check result: 'IF' is not 'else if'",
80                  CheckUtils.isElseIf(ifAst));
81  
82          final DetailAST parentAst2 = new DetailAST();
83          parentAst2.setType(TokenTypes.SLIST);
84          parentAst2.setText("SLIST");
85  
86          parentAst2.addChild(ifAst);
87  
88          assertFalse("Invalid elseIf check result: 'SLIST' is not 'else if'",
89                  CheckUtils.isElseIf(ifAst));
90  
91          final DetailAST elseAst = new DetailAST();
92          elseAst.setType(TokenTypes.LITERAL_ELSE);
93  
94          elseAst.setFirstChild(ifAst);
95          assertTrue("Invalid elseIf check result", CheckUtils.isElseIf(ifAst));
96      }

98      @Test
99      public void testEquals() {
100         final DetailAST litStatic = new DetailAST();
101         litStatic.setType(TokenTypes.LITERAL_STATIC);
102 
103         final DetailAST modifiers = new DetailAST();
104         modifiers.setType(TokenTypes.MODIFIERS);
105         modifiers.addChild(litStatic);
106 
107         final DetailAST metDef = new DetailAST();
108         metDef.setType(TokenTypes.METHOD_DEF);
109         metDef.addChild(modifiers);
110 
111         assertFalse("Invalid result: ast is not equals method",
112                 CheckUtils.isEqualsMethod(metDef));
113 
114         metDef.removeChildren();
115 
116         final DetailAST metName = new DetailAST();
117         metName.setType(TokenTypes.IDENT);
118         metName.setText("equals");
119         metDef.addChild(metName);
120 
121         final DetailAST modifiers2 = new DetailAST();
122         modifiers2.setType(TokenTypes.MODIFIERS);
123         metDef.addChild(modifiers2);
124 
125         final DetailAST parameter1 = new DetailAST();
126         final DetailAST parameter2 = new DetailAST();
127 
128         final DetailAST parameters = new DetailAST();
129         parameters.setType(TokenTypes.PARAMETERS);
130 
131         parameters.addChild(parameter2);
132 
133         parameters.addChild(parameter1);
134         metDef.addChild(parameters);
135 
136         assertFalse("Invalid result: ast is not equals method",
137                 CheckUtils.isEqualsMethod(metDef));
138     }

218     @Test
219     public void testIsElseIf() throws Exception {
220         final DetailAST targetMethodNode = getNodeFromFile(TokenTypes.METHOD_DEF).getNextSibling();
221         final DetailAST firstElseNode = getNode(targetMethodNode, TokenTypes.LITERAL_ELSE);
222         final DetailAST ifElseWithCurlyBraces = firstElseNode.getFirstChild().getFirstChild();
223         final DetailAST ifElse = getNode(firstElseNode.getParent().getNextSibling(),
224                 TokenTypes.LITERAL_ELSE).getFirstChild();
225         final DetailAST ifWithoutElse =
226                 firstElseNode.getParent().getNextSibling().getNextSibling();
227 
228         assertTrue("Invalid result: AST provided is not else if with curly",
229                 CheckUtils.isElseIf(ifElseWithCurlyBraces));
230         assertTrue("Invalid result: AST provided is not else if with curly",
231                 CheckUtils.isElseIf(ifElse));
232         assertFalse("Invalid result: AST provided is else if with curly",
233                 CheckUtils.isElseIf(ifWithoutElse));
234     }

327     public void testParseDouble() throws Exception {
328         assertEquals("Invalid parse result", 1.0,
329                 CheckUtils.parseDouble("1", TokenTypes.NUM_INT), 0);
330         assertEquals("Invalid parse result", -0.05,
331                 CheckUtils.parseDouble("-0.05f", TokenTypes.NUM_FLOAT), 0);

334         assertEquals("Invalid parse result", 0.0,

336         assertEquals("Invalid parse result", 2915.0,

339                 CheckUtils.parseDouble("21474836470", TokenTypes.NUM_LONG), 0);

353     private DetailAST getNodeFromFile(int type) throws Exception {
354         return getNode(parseFile(new File(getPath("InputCheckUtilsTest.java"))), type);
355     }
356 
357     private static DetailAST getNode(DetailAST root, int type) {
358         final Optional<DetailAST> node = findTokenInAstByPredicate(root,
359             ast -> ast.getType() == type);
360 
361         if (!node.isPresent()) {
362             fail("Cannot find node of specified type: " + type);
363         }
364 
365         return node.get();
366     }
