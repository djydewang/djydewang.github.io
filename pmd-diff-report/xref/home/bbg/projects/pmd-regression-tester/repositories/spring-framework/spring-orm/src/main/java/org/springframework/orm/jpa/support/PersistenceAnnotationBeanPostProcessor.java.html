70/**

71 * BeanPostProcessor that processes {@link javax.persistence.PersistenceUnit}

72 * and {@link javax.persistence.PersistenceContext} annotations, for injection of

73 * the corresponding JPA resources {@link javax.persistence.EntityManagerFactory}

74 * and {@link javax.persistence.EntityManager}. Any such annotated fields or methods

75 * in any Spring-managed object will automatically be injected.

76 *

77 * <p>This post-processor will inject sub-interfaces of {@code EntityManagerFactory}

78 * and {@code EntityManager} if the annotated fields or methods are declared as such.

79 * The actual type will be verified early, with the exception of a shared ("transactional")

80 * {@code EntityManager} reference, where type mismatches might be detected as late

81 * as on the first actual invocation.

82 *

83 * <p>Note: In the present implementation, PersistenceAnnotationBeanPostProcessor

84 * only supports {@code @PersistenceUnit} and {@code @PersistenceContext}

85 * with the "unitName" attribute, or no attribute at all (for the default unit).

86 * If those annotations are present with the "name" attribute at the class level,

87 * they will simply be ignored, since those only serve as deployment hint

88 * (as per the Java EE specification).

89 *

90 * <p>This post-processor can either obtain EntityManagerFactory beans defined

91 * in the Spring application context (the default), or obtain EntityManagerFactory

92 * references from JNDI ("persistence unit references"). In the bean case,

93 * the persistence unit name will be matched against the actual deployed unit,

94 * with the bean name used as fallback unit name if no deployed name found.

95 * Typically, Spring's {@link org.springframework.orm.jpa.LocalContainerEntityManagerFactoryBean}

96 * will be used for setting up such EntityManagerFactory beans. Alternatively,

97 * such beans may also be obtained from JNDI, e.g. using the {@code jee:jndi-lookup}

98 * XML configuration element (with the bean name matching the requested unit name).

99 * In both cases, the post-processor definition will look as simple as this:

100 *

101 * <pre class="code">

102 * &lt;bean class="org.springframework.orm.jpa.support.PersistenceAnnotationBeanPostProcessor"/&gt;</pre>

103 *

104 * In the JNDI case, specify the corresponding JNDI names in this post-processor's

105 * {@link #setPersistenceUnits "persistenceUnits" map}, typically with matching

106 * {@code persistence-unit-ref} entries in the Java EE deployment descriptor.

107 * By default, those names are considered as resource references (according to the

108 * Java EE resource-ref convention), located underneath the "java:comp/env/" namespace.

109 * For example:

110 *

111 * <pre class="code">

112 * &lt;bean class="org.springframework.orm.jpa.support.PersistenceAnnotationBeanPostProcessor"&gt;

113 *   &lt;property name="persistenceUnits"&gt;

114 *     &lt;map/gt;

115 *       &lt;entry key="unit1" value="persistence/unit1"/&gt;

116 *       &lt;entry key="unit2" value="persistence/unit2"/&gt;

117 *     &lt;/map/gt;

118 *   &lt;/property&gt;

119 * &lt;/bean&gt;</pre>

120 *

121 * In this case, the specified persistence units will always be resolved in JNDI

122 * rather than as Spring-defined beans. The entire persistence unit deployment,

123 * including the weaving of persistent classes, is then up to the Java EE server.

124 * Persistence contexts (i.e. EntityManager references) will be built based on

125 * those server-provided EntityManagerFactory references, using Spring's own

126 * transaction synchronization facilities for transactional EntityManager handling

127 * (typically with Spring's {@code @Transactional} annotation for demarcation

128 * and {@link org.springframework.transaction.jta.JtaTransactionManager} as backend).

129 *

130 * <p>If you prefer the Java EE server's own EntityManager handling, specify entries

131 * in this post-processor's {@link #setPersistenceContexts "persistenceContexts" map}

132 * (or {@link #setExtendedPersistenceContexts "extendedPersistenceContexts" map},

133 * typically with matching {@code persistence-context-ref} entries in the

134 * Java EE deployment descriptor. For example:

135 *

136 * <pre class="code">

137 * &lt;bean class="org.springframework.orm.jpa.support.PersistenceAnnotationBeanPostProcessor"&gt;

138 *   &lt;property name="persistenceContexts"&gt;

139 *     &lt;map/gt;

140 *       &lt;entry key="unit1" value="persistence/context1"/&gt;

141 *       &lt;entry key="unit2" value="persistence/context2"/&gt;

142 *     &lt;/map/gt;

143 *   &lt;/property&gt;

144 * &lt;/bean&gt;</pre>

145 *

146 * If the application only obtains EntityManager references in the first place,

147 * this is all you need to specify. If you need EntityManagerFactory references

148 * as well, specify entries for both "persistenceUnits" and "persistenceContexts",

149 * pointing to matching JNDI locations.

150 *

151 * <p><b>NOTE: In general, do not inject EXTENDED EntityManagers into STATELESS beans,

152 * i.e. do not use {@code @PersistenceContext} with type {@code EXTENDED} in

153 * Spring beans defined with scope 'singleton' (Spring's default scope).</b>

154 * Extended EntityManagers are <i>not</i> thread-safe, hence they must not be used

155 * in concurrently accessed beans (which Spring-managed singletons usually are).

156 *

157 * <p>Note: A default PersistenceAnnotationBeanPostProcessor will be registered

158 * by the "context:annotation-config" and "context:component-scan" XML tags.

159 * Remove or turn off the default annotation configuration there if you intend

160 * to specify a custom PersistenceAnnotationBeanPostProcessor bean definition.

161 *

162 * @author Rod Johnson

163 * @author Juergen Hoeller

164 * @since 2.0

165 * @see javax.persistence.PersistenceUnit

166 * @see javax.persistence.PersistenceContext

167 */

382	private InjectionMetadata findPersistenceMetadata(String beanName, final Class<?> clazz, @Nullable PropertyValues pvs) {

383		// Fall back to class name as cache key, for backwards compatibility with custom callers.

384		String cacheKey = (StringUtils.hasLength(beanName) ? beanName : clazz.getName());

385		// Quick check on the concurrent map first, with minimal locking.

386		InjectionMetadata metadata = this.injectionMetadataCache.get(cacheKey);

387		if (InjectionMetadata.needsRefresh(metadata, clazz)) {

388			synchronized (this.injectionMetadataCache) {

389				metadata = this.injectionMetadataCache.get(cacheKey);

390				if (InjectionMetadata.needsRefresh(metadata, clazz)) {

391					if (metadata != null) {

392						metadata.clear(pvs);

393					}

394					metadata = buildPersistenceMetadata(clazz);

395					this.injectionMetadataCache.put(cacheKey, metadata);

396				}

397			}

398		}

399		return metadata;

400	}

401

402	private InjectionMetadata buildPersistenceMetadata(final Class<?> clazz) {

403		LinkedList<InjectionMetadata.InjectedElement> elements = new LinkedList<>();

404		Class<?> targetClass = clazz;

405

406		do {

407			final LinkedList<InjectionMetadata.InjectedElement> currElements =

408					new LinkedList<>();

409

410			ReflectionUtils.doWithLocalFields(targetClass, field -> {

411				if (field.isAnnotationPresent(PersistenceContext.class) ||

412						field.isAnnotationPresent(PersistenceUnit.class)) {

413					if (Modifier.isStatic(field.getModifiers())) {

414						throw new IllegalStateException("Persistence annotations are not supported on static fields");

415					}

416					currElements.add(new PersistenceElement(field, field, null));

417				}

418			});

419

420			ReflectionUtils.doWithLocalMethods(targetClass, method -> {

421				Method bridgedMethod = BridgeMethodResolver.findBridgedMethod(method);

422				if (!BridgeMethodResolver.isVisibilityBridgeMethodPair(method, bridgedMethod)) {

423					return;

424				}

425				if ((bridgedMethod.isAnnotationPresent(PersistenceContext.class) ||

426						bridgedMethod.isAnnotationPresent(PersistenceUnit.class)) &&

427						method.equals(ClassUtils.getMostSpecificMethod(method, clazz))) {

428					if (Modifier.isStatic(method.getModifiers())) {

429						throw new IllegalStateException("Persistence annotations are not supported on static methods");

430					}

431					if (method.getParameterCount() != 1) {

432						throw new IllegalStateException("Persistence annotation requires a single-arg method: " + method);

433					}

434					PropertyDescriptor pd = BeanUtils.findPropertyForMethod(bridgedMethod, clazz);

435					currElements.add(new PersistenceElement(method, bridgedMethod, pd));

436				}

437			});

438

439			elements.addAll(0, currElements);

440			targetClass = targetClass.getSuperclass();

441		}

442		while (targetClass != null && targetClass != Object.class);

443

444		return new InjectionMetadata(clazz, elements);

445	}

447	/**

448	 * Return a specified persistence unit for the given unit name,

449	 * as defined through the "persistenceUnits" map.

450	 * @param unitName the name of the persistence unit

451	 * @return the corresponding EntityManagerFactory,

452	 * or {@code null} if none found

453	 * @see #setPersistenceUnits

454	 */

455	@Nullable

456	protected EntityManagerFactory getPersistenceUnit(@Nullable String unitName) {

457		if (this.persistenceUnits != null) {

458			String unitNameForLookup = (unitName != null ? unitName : "");

459			if ("".equals(unitNameForLookup)) {

460				unitNameForLookup = this.defaultPersistenceUnitName;

461			}

462			String jndiName = this.persistenceUnits.get(unitNameForLookup);

463			if (jndiName == null && "".equals(unitNameForLookup) && this.persistenceUnits.size() == 1) {

464				jndiName = this.persistenceUnits.values().iterator().next();

465			}

466			if (jndiName != null) {

467				try {

468					return lookup(jndiName, EntityManagerFactory.class);

469				}

470				catch (Exception ex) {

471					throw new IllegalStateException("Could not obtain EntityManagerFactory [" + jndiName + "] from JNDI", ex);

472				}

473			}

474		}

475		return null;

476	}

478	/**

479	 * Return a specified persistence context for the given unit name, as defined

480	 * through the "persistenceContexts" (or "extendedPersistenceContexts") map.

481	 * @param unitName the name of the persistence unit

482	 * @param extended whether to obtain an extended persistence context

483	 * @return the corresponding EntityManager, or {@code null} if none found

484	 * @see #setPersistenceContexts

485	 * @see #setExtendedPersistenceContexts

486	 */

487	@Nullable

488	protected EntityManager getPersistenceContext(@Nullable String unitName, boolean extended) {

489		Map<String, String> contexts = (extended ? this.extendedPersistenceContexts : this.persistenceContexts);

490		if (contexts != null) {

491			String unitNameForLookup = (unitName != null ? unitName : "");

492			if ("".equals(unitNameForLookup)) {

493				unitNameForLookup = this.defaultPersistenceUnitName;

494			}

495			String jndiName = contexts.get(unitNameForLookup);

496			if (jndiName == null && "".equals(unitNameForLookup) && contexts.size() == 1) {

497				jndiName = contexts.values().iterator().next();

498			}

499			if (jndiName != null) {

500				try {

501					return lookup(jndiName, EntityManager.class);

502				}

503				catch (Exception ex) {

504					throw new IllegalStateException("Could not obtain EntityManager [" + jndiName + "] from JNDI", ex);

505				}

506			}

507		}

508		return null;

509	}

511	/**

512	 * Find an EntityManagerFactory with the given name in the current Spring

513	 * application context, falling back to a single default EntityManagerFactory

514	 * (if any) in case of no unit name specified.

515	 * @param unitName the name of the persistence unit (may be {@code null} or empty)

516	 * @param requestingBeanName the name of the requesting bean

517	 * @return the EntityManagerFactory

518	 * @throws NoSuchBeanDefinitionException if there is no such EntityManagerFactory in the context

519	 */

520	protected EntityManagerFactory findEntityManagerFactory(@Nullable String unitName, @Nullable String requestingBeanName)

521			throws NoSuchBeanDefinitionException {

522

523		String unitNameForLookup = (unitName != null ? unitName : "");

524		if ("".equals(unitNameForLookup)) {

525			unitNameForLookup = this.defaultPersistenceUnitName;

526		}

527		if (!"".equals(unitNameForLookup)) {

528			return findNamedEntityManagerFactory(unitNameForLookup, requestingBeanName);

529		}

530		else {

531			return findDefaultEntityManagerFactory(requestingBeanName);

532		}

533	}

535	/**

536	 * Find an EntityManagerFactory with the given name in the current

537	 * Spring application context.

538	 * @param unitName the name of the persistence unit (never empty)

539	 * @param requestingBeanName the name of the requesting bean

540	 * @return the EntityManagerFactory

541	 * @throws NoSuchBeanDefinitionException if there is no such EntityManagerFactory in the context

542	 */

543	protected EntityManagerFactory findNamedEntityManagerFactory(String unitName, @Nullable String requestingBeanName)

544			throws NoSuchBeanDefinitionException {

545

546		Assert.state(this.beanFactory != null, "ListableBeanFactory required for EntityManagerFactory bean lookup");

547

548		EntityManagerFactory emf = EntityManagerFactoryUtils.findEntityManagerFactory(this.beanFactory, unitName);

549		if (requestingBeanName != null && this.beanFactory instanceof ConfigurableBeanFactory) {

550			((ConfigurableBeanFactory) this.beanFactory).registerDependentBean(unitName, requestingBeanName);

551		}

552		return emf;

553	}

554

555	/**

556	 * Find a single default EntityManagerFactory in the Spring application context.

557	 * @return the default EntityManagerFactory

558	 * @throws NoSuchBeanDefinitionException if there is no single EntityManagerFactory in the context

559	 */

560	protected EntityManagerFactory findDefaultEntityManagerFactory(@Nullable String requestingBeanName)

561			throws NoSuchBeanDefinitionException {

562

563		Assert.state(this.beanFactory != null, "ListableBeanFactory required for EntityManagerFactory bean lookup");

564

565		if (this.beanFactory instanceof ConfigurableListableBeanFactory) {

566			// Fancy variant with dependency registration

567			ConfigurableListableBeanFactory clbf = (ConfigurableListableBeanFactory) this.beanFactory;

568			NamedBeanHolder<EntityManagerFactory> emfHolder = clbf.resolveNamedBean(EntityManagerFactory.class);

569			if (requestingBeanName != null) {

570				clbf.registerDependentBean(emfHolder.getBeanName(), requestingBeanName);

571			}

572			return emfHolder.getBeanInstance();

573		}

574		else {

575			// Plain variant: just find a default bean

576			return this.beanFactory.getBean(EntityManagerFactory.class);

577		}

578	}

580	/**

581	 * Perform a JNDI lookup for the given resource by name.

582	 * <p>Called for EntityManagerFactory and EntityManager lookup

583	 * when JNDI names are mapped for specific persistence units.

584	 * @param jndiName the JNDI name to look up

585	 * @param requiredType the required type of the object

586	 * @return the obtained object

587	 * @throws Exception if the JNDI lookup failed

588	 */

589	protected <T> T lookup(String jndiName, Class<T> requiredType) throws Exception {

590		return new LocatorDelegate().lookup(jndiName, requiredType);

591	}

594	/**

595	 * Separate inner class to isolate the JNDI API dependency

596	 * (for compatibility with Google App Engine's API white list).

597	 */

598	private class LocatorDelegate {

599

600		public <T> T lookup(String jndiName, Class<T> requiredType) throws Exception {

601			JndiLocatorDelegate locator = new JndiLocatorDelegate();

602			if (jndiEnvironment instanceof JndiTemplate) {

603				locator.setJndiTemplate((JndiTemplate) jndiEnvironment);

604			}

605			else if (jndiEnvironment instanceof Properties) {

606				locator.setJndiEnvironment((Properties) jndiEnvironment);

607			}

608			else if (jndiEnvironment != null) {

609				throw new IllegalStateException("Illegal 'jndiEnvironment' type: " + jndiEnvironment.getClass());

610			}

611			locator.setResourceRef(resourceRef);

612			return locator.lookup(jndiName, requiredType);

613		}

614	}

617	/**

618	 * Class representing injection information about an annotated field

619	 * or setter method.

620	 */

621	private class PersistenceElement extends InjectionMetadata.InjectedElement {

622

623		private final String unitName;

624

625		@Nullable

626		private PersistenceContextType type;

627

628		private boolean synchronizedWithTransaction = false;

629

630		@Nullable

631		private Properties properties;

632

633		public PersistenceElement(Member member, AnnotatedElement ae, @Nullable PropertyDescriptor pd) {

634			super(member, pd);

635			PersistenceContext pc = ae.getAnnotation(PersistenceContext.class);

636			PersistenceUnit pu = ae.getAnnotation(PersistenceUnit.class);

637			Class<?> resourceType = EntityManager.class;

638			if (pc != null) {

639				if (pu != null) {

640					throw new IllegalStateException("Member may only be annotated with either " +

641							"@PersistenceContext or @PersistenceUnit, not both: " + member);

642				}

643				Properties properties = null;

644				PersistenceProperty[] pps = pc.properties();

645				if (!ObjectUtils.isEmpty(pps)) {

646					properties = new Properties();

647					for (PersistenceProperty pp : pps) {

648						properties.setProperty(pp.name(), pp.value());

649					}

650				}

651				this.unitName = pc.unitName();

652				this.type = pc.type();

653				this.synchronizedWithTransaction = SynchronizationType.SYNCHRONIZED.equals(pc.synchronization());

654				this.properties = properties;

655			}

656			else {

657				resourceType = EntityManagerFactory.class;

658				this.unitName = pu.unitName();

659			}

660			checkResourceType(resourceType);

661		}

662

663		/**

664		 * Resolve the object against the application context.

665		 */

666		@Override

667		protected Object getResourceToInject(Object target, @Nullable String requestingBeanName) {

668			// Resolves to EntityManagerFactory or EntityManager.

669			if (this.type != null) {

670				return (this.type == PersistenceContextType.EXTENDED ?

671						resolveExtendedEntityManager(target, requestingBeanName) :

672						resolveEntityManager(requestingBeanName));

673			}

674			else {

675				// OK, so we need an EntityManagerFactory...

676				return resolveEntityManagerFactory(requestingBeanName);

677			}

678		}

679

680		private EntityManagerFactory resolveEntityManagerFactory(@Nullable String requestingBeanName) {

681			// Obtain EntityManagerFactory from JNDI?

682			EntityManagerFactory emf = getPersistenceUnit(this.unitName);

683			if (emf == null) {

684				// Need to search for EntityManagerFactory beans.

685				emf = findEntityManagerFactory(this.unitName, requestingBeanName);

686			}

687			return emf;

688		}

689

690		private EntityManager resolveEntityManager(@Nullable String requestingBeanName) {

691			// Obtain EntityManager reference from JNDI?

692			EntityManager em = getPersistenceContext(this.unitName, false);

693			if (em == null) {

694				// No pre-built EntityManager found -> build one based on factory.

695				// Obtain EntityManagerFactory from JNDI?

696				EntityManagerFactory emf = getPersistenceUnit(this.unitName);

697				if (emf == null) {

698					// Need to search for EntityManagerFactory beans.

699					emf = findEntityManagerFactory(this.unitName, requestingBeanName);

700				}

701				// Inject a shared transactional EntityManager proxy.

702				if (emf instanceof EntityManagerFactoryInfo &&

703						((EntityManagerFactoryInfo) emf).getEntityManagerInterface() != null) {

704					// Create EntityManager based on the info's vendor-specific type

705					// (which might be more specific than the field's type).

706					em = SharedEntityManagerCreator.createSharedEntityManager(

707							emf, this.properties, this.synchronizedWithTransaction);

708				}

709				else {

710					// Create EntityManager based on the field's type.

711					em = SharedEntityManagerCreator.createSharedEntityManager(

712							emf, this.properties, this.synchronizedWithTransaction, getResourceType());

713				}

714			}

715			return em;

716		}

717

718		private EntityManager resolveExtendedEntityManager(Object target, @Nullable String requestingBeanName) {

719			// Obtain EntityManager reference from JNDI?

720			EntityManager em = getPersistenceContext(this.unitName, true);

721			if (em == null) {

722				// No pre-built EntityManager found -> build one based on factory.

723				// Obtain EntityManagerFactory from JNDI?

724				EntityManagerFactory emf = getPersistenceUnit(this.unitName);

725				if (emf == null) {

726					// Need to search for EntityManagerFactory beans.

727					emf = findEntityManagerFactory(this.unitName, requestingBeanName);

728				}

729				// Inject a container-managed extended EntityManager.

730				em = ExtendedEntityManagerCreator.createContainerManagedEntityManager(

731						emf, this.properties, this.synchronizedWithTransaction);

732			}

733			if (em instanceof EntityManagerProxy && beanFactory != null && requestingBeanName != null &&

734					beanFactory.containsBean(requestingBeanName) && !beanFactory.isPrototype(requestingBeanName)) {

735				extendedEntityManagersToClose.put(target, ((EntityManagerProxy) em).getTargetEntityManager());

736			}

737			return em;

738		}

739	}
