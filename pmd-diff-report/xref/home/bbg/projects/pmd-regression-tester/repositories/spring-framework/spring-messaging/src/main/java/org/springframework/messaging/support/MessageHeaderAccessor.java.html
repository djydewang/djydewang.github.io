40/**

41 * A base for classes providing strongly typed getters and setters as well as

42 * behavior around specific categories of headers (e.g. STOMP headers).

43 * Supports creating new headers, modifying existing headers (when still mutable),

44 * or copying and modifying existing headers.

45 *

46 * <p>The method {@link #getMessageHeaders()} provides access to the underlying,

47 * fully-prepared {@link MessageHeaders} that can then be used as-is (i.e.

48 * without copying) to create a single message as follows:

49 *

50 * <pre class="code">

51 * MessageHeaderAccessor accessor = new MessageHeaderAccessor();

52 * accessor.setHeader("foo", "bar");

53 * Message message = MessageBuilder.createMessage("payload", accessor.getMessageHeaders());

54 * </pre>

55 *

56 * <p>After the above, by default the {@code MessageHeaderAccessor} becomes

57 * immutable. However it is possible to leave it mutable for further initialization

58 * in the same thread, for example:

59 *

60 * <pre class="code">

61 * MessageHeaderAccessor accessor = new MessageHeaderAccessor();

62 * accessor.setHeader("foo", "bar");

63 * accessor.setLeaveMutable(true);

64 * Message message = MessageBuilder.createMessage("payload", accessor.getMessageHeaders());

65 *

66 * // later on in the same thread...

67 *

68 * MessageHeaderAccessor accessor = MessageHeaderAccessor.getAccessor(message);

69 * accessor.setHeader("bar", "baz");

70 * accessor.setImmutable();

71 * </pre>

72 *

73 * <p>The method {@link #toMap()} returns a copy of the underlying headers. It can

74 * be used to prepare multiple messages from the same {@code MessageHeaderAccessor}

75 * instance:

76 * <pre class="code">

77 * MessageHeaderAccessor accessor = new MessageHeaderAccessor();

78 * MessageBuilder builder = MessageBuilder.withPayload("payload").setHeaders(accessor);

79 *

80 * accessor.setHeader("foo", "bar1");

81 * Message message1 = builder.build();

82 *

83 * accessor.setHeader("foo", "bar2");

84 * Message message2 = builder.build();

85 *

86 * accessor.setHeader("foo", "bar3");

87 * Message  message3 = builder.build();

88 * </pre>

89 *

90 * <p>However note that with the above style, the header accessor is shared and

91 * cannot be re-obtained later on. Alternatively it is also possible to create

92 * one {@code MessageHeaderAccessor} per message:

93 *

94 * <pre class="code">

95 * MessageHeaderAccessor accessor1 = new MessageHeaderAccessor();

96 * accessor.set("foo", "bar1");

97 * Message message1 = MessageBuilder.createMessage("payload", accessor1.getMessageHeaders());

98 *

99 * MessageHeaderAccessor accessor2 = new MessageHeaderAccessor();

100 * accessor.set("foo", "bar2");

101 * Message message2 = MessageBuilder.createMessage("payload", accessor2.getMessageHeaders());

102 *

103 * MessageHeaderAccessor accessor3 = new MessageHeaderAccessor();

104 * accessor.set("foo", "bar3");

105 * Message message3 = MessageBuilder.createMessage("payload", accessor3.getMessageHeaders());

106 * </pre>

107 *

108 * <p>Note that the above examples aim to demonstrate the general idea of using

109 * header accessors. The most likely usage however is through subclasses.

110 *

111 * @author Rossen Stoyanchev

112 * @author Juergen Hoeller

113 * @since 4.0

114 */

165	/**

166	 * By default when {@link #getMessageHeaders()} is called, {@code "this"}

167	 * {@code MessageHeaderAccessor} instance can no longer be used to modify the

168	 * underlying message headers and the returned {@code MessageHeaders} is immutable.

169	 * <p>However when this is set to {@code true}, the returned (underlying)

170	 * {@code MessageHeaders} instance remains mutable. To make further modifications

171	 * continue to use the same accessor instance or re-obtain it via:<br>

172	 * {@link MessageHeaderAccessor#getAccessor(Message, Class)

173	 * MessageHeaderAccessor.getAccessor(Message, Class)}

174	 * <p>When modifications are complete use {@link #setImmutable()} to prevent

175	 * further changes. The intended use case for this mechanism is initialization

176	 * of a Message within a single thread.

177	 * <p>By default this is set to {@code false}.

178	 * @since 4.1

179	 */

180	public void setLeaveMutable(boolean leaveMutable) {

181		Assert.state(this.headers.isMutable(), "Already immutable");

182		this.leaveMutable = leaveMutable;

183	}

184

185	/**

186	 * By default when {@link #getMessageHeaders()} is called, {@code "this"}

187	 * {@code MessageHeaderAccessor} instance can no longer be used to modify the

188	 * underlying message headers. However if {@link #setLeaveMutable(boolean)}

189	 * is used, this method is necessary to indicate explicitly when the

190	 * {@code MessageHeaders} instance should no longer be modified.

191	 * @since 4.1

192	 */

193	public void setImmutable() {

194		this.headers.setImmutable();

195	}

245	/**

246	 * Return the underlying {@code MessageHeaders} instance.

247	 * <p>Unless {@link #setLeaveMutable(boolean)} was set to {@code true}, after

248	 * this call, the headers are immutable and this accessor can no longer

249	 * modify them.

250	 * <p>This method always returns the same {@code MessageHeaders} instance if

251	 * invoked multiples times. To obtain a copy of the underlying headers, use

252	 * {@link #toMessageHeaders()} or {@link #toMap()} instead.

253	 * @since 4.1

255	public MessageHeaders getMessageHeaders() {

256		if (!this.leaveMutable) {

257			setImmutable();

258		}

259		return this.headers;

260	}

284	/**

285	 * Retrieve the value for the header with the given name.

286	 * @param headerName the name of the header

287	 * @return the associated value, or {@code null} if none found

288	 */

289	@Nullable

290	public Object getHeader(String headerName) {

291		return this.headers.get(headerName);

292	}

293

294	/**

295	 * Set the value for the given header name.

296	 * <p>If the provided value is {@code null}, the header will be removed.

297	 */

298	public void setHeader(String name, @Nullable Object value) {

299		if (isReadOnly(name)) {

300			throw new IllegalArgumentException("'" + name + "' header is read-only");

301		}

302		verifyType(name, value);

303		if (value != null) {

304			// Modify header if necessary

305			if (!ObjectUtils.nullSafeEquals(value, getHeader(name))) {

306				this.modified = true;

307				this.headers.getRawHeaders().put(name, value);

308			}

309		}

310		else {

311			// Remove header if available

312			if (this.headers.containsKey(name)) {

313				this.modified = true;

314				this.headers.getRawHeaders().remove(name);

315			}

316		}

317	}

318

319	protected void verifyType(@Nullable String headerName, @Nullable Object headerValue) {

320		if (headerName != null && headerValue != null) {

321			if (MessageHeaders.ERROR_CHANNEL.equals(headerName) ||

322					MessageHeaders.REPLY_CHANNEL.endsWith(headerName)) {

323				if (!(headerValue instanceof MessageChannel || headerValue instanceof String)) {

324					throw new IllegalArgumentException(

325							"'" + headerName + "' header value must be a MessageChannel or String");

326				}

327			}

328		}

329	}

330

331	/**

332	 * Set the value for the given header name only if the header name is not

333	 * already associated with a value.

334	 */

335	public void setHeaderIfAbsent(String name, Object value) {

336		if (getHeader(name) == null) {

337			setHeader(name, value);

338		}

339	}

340

341	/**

342	 * Remove the value for the given header name.

343	 */

344	public void removeHeader(String headerName) {

345		if (StringUtils.hasLength(headerName) && !isReadOnly(headerName)) {

346			setHeader(headerName, null);

347		}

348	}

349

350	/**

351	 * Removes all headers provided via array of 'headerPatterns'.

352	 * <p>As the name suggests, array may contain simple matching patterns for header

353	 * names. Supported pattern styles are: "xxx*", "*xxx", "*xxx*" and "xxx*yyy".

354	 */

355	public void removeHeaders(String... headerPatterns) {

356		List<String> headersToRemove = new ArrayList<>();

357		for (String pattern : headerPatterns) {

358			if (StringUtils.hasLength(pattern)){

359				if (pattern.contains("*")){

360					headersToRemove.addAll(getMatchingHeaderNames(pattern, this.headers));

361				}

362				else {

363					headersToRemove.add(pattern);

364				}

365			}

366		}

367		for (String headerToRemove : headersToRemove) {

368			removeHeader(headerToRemove);

369		}

370	}

371

372	private List<String> getMatchingHeaderNames(String pattern, @Nullable Map<String, Object> headers) {

373		List<String> matchingHeaderNames = new ArrayList<>();

374		if (headers != null) {

375			for (String key : headers.keySet()) {

376				if (PatternMatchUtils.simpleMatch(pattern, key)) {

377					matchingHeaderNames.add(key);

378				}

379			}

380		}

381		return matchingHeaderNames;

382	}

383

384	/**

385	 * Copy the name-value pairs from the provided Map.

386	 * <p>This operation will overwrite any existing values. Use

387	 * {@link #copyHeadersIfAbsent(Map)} to avoid overwriting values.

388	 */

389	public void copyHeaders(@Nullable Map<String, ?> headersToCopy) {

390		if (headersToCopy != null) {

391			headersToCopy.forEach((key, value) -> {

392				if (!isReadOnly(key)) {

393					setHeader(key, value);

394				}

395			});

396		}

397	}

398

399	/**

400	 * Copy the name-value pairs from the provided Map.

401	 * <p>This operation will <em>not</em> overwrite any existing values.

402	 */

403	public void copyHeadersIfAbsent(@Nullable Map<String, ?> headersToCopy) {

404		if (headersToCopy != null) {

405			headersToCopy.forEach((key, value) -> {

406				if (!isReadOnly(key)) {

407					setHeaderIfAbsent(key, value);

408				}

409			});

410		}

411	}

413	protected boolean isReadOnly(String headerName) {

414		return (MessageHeaders.ID.equals(headerName) || MessageHeaders.TIMESTAMP.equals(headerName));

415	}

422		Object value = getHeader(MessageHeaders.ID);

423		if (value == null) {

424			return null;

425		}

426		return (value instanceof UUID ? (UUID) value : UUID.fromString(value.toString()));

431		Object value = getHeader(MessageHeaders.TIMESTAMP);

432		if (value == null) {

433			return null;

434		}

435		return (value instanceof Long ? (Long) value : Long.parseLong(value.toString()));

486	/**

487	 * Return a concise message for logging purposes.

488	 * @param payload the payload that corresponds to the headers.

489	 * @return the message

490	 */

491	public String getShortLogMessage(Object payload) {

492		return "headers=" + this.headers.toString() + getShortPayloadLogMessage(payload);

493	}

494

495	/**

496	 * Return a more detailed message for logging purposes.

497	 * @param payload the payload that corresponds to the headers.

498	 * @return the message

499	 */

500	public String getDetailedLogMessage(@Nullable Object payload) {

501		return "headers=" + this.headers.toString() + getDetailedPayloadLogMessage(payload);

502	}

503

504	protected String getShortPayloadLogMessage(Object payload) {

505		if (payload instanceof String) {

506			String payloadText = (String) payload;

507			return (payloadText.length() < 80) ?

508				" payload=" + payloadText :

509				" payload=" + payloadText.substring(0, 80) + "...(truncated)";

510		}

511		else if (payload instanceof byte[]) {

512			byte[] bytes = (byte[]) payload;

513			if (isReadableContentType()) {

514				return (bytes.length < 80) ?

515						" payload=" + new String(bytes, getCharset()) :

516						" payload=" + new String(Arrays.copyOf(bytes, 80), getCharset()) + "...(truncated)";

517			}

518			else {

519				return " payload=byte[" + bytes.length + "]";

520			}

521		}

522		else {

523			String payloadText = payload.toString();

524			return (payloadText.length() < 80) ?

525					" payload=" + payloadText :

526					" payload=" + ObjectUtils.identityToString(payload);

527		}

528	}

529

530	protected String getDetailedPayloadLogMessage(@Nullable Object payload) {

531		if (payload instanceof String) {

532			return " payload=" + payload;

533		}

534		else if (payload instanceof byte[]) {

535			byte[] bytes = (byte[]) payload;

536			if (isReadableContentType()) {

537				return " payload=" + new String(bytes, getCharset());

538			}

539			else {

540				return " payload=byte[" + bytes.length + "]";

541			}

542		}

543		else {

544			return " payload=" + payload;

545		}

546	}

547

548	protected boolean isReadableContentType() {

549		MimeType contentType = getContentType();

550		for (MimeType mimeType : READABLE_MIME_TYPES) {

551			if (mimeType.includes(contentType)) {

552				return true;

553			}

554		}

555		return false;

556	}

566	/**

567	 * Return the original {@code MessageHeaderAccessor} used to create the headers

568	 * of the given {@code Message}, or {@code null} if that's not available or if

569	 * its type does not match the required type.

570	 * <p>This is for cases where the existence of an accessor is strongly expected

571	 * (followed up with an assertion) or where an accessor will be created otherwise.

572	 * @param message the message to get an accessor for

573	 * @param requiredType the required accessor type (or {@code null} for any)

574	 * @return an accessor instance of the specified type, or {@code null} if none

575	 * @since 4.1

576	 */

577	@Nullable

578	public static <T extends MessageHeaderAccessor> T getAccessor(Message<?> message, @Nullable Class<T> requiredType) {

579		return getAccessor(message.getHeaders(), requiredType);

580	}

581

582	/**

583	 * A variation of {@link #getAccessor(org.springframework.messaging.Message, Class)}

584	 * with a {@code MessageHeaders} instance instead of a {@code Message}.

585	 * <p>This is for cases when a full message may not have been created yet.

586	 * @param messageHeaders the message headers to get an accessor for

587	 * @param requiredType the required accessor type (or {@code null} for any)

588	 * @return an accessor instance of the specified type, or {@code null} if none

589	 * @since 4.1

590	 */

591	@SuppressWarnings("unchecked")

592	@Nullable

593	public static <T extends MessageHeaderAccessor> T getAccessor(

594			MessageHeaders messageHeaders, @Nullable Class<T> requiredType) {

595

596		if (messageHeaders instanceof MutableMessageHeaders) {

597			MutableMessageHeaders mutableHeaders = (MutableMessageHeaders) messageHeaders;

598			MessageHeaderAccessor headerAccessor = mutableHeaders.getAccessor();

599			if (requiredType == null || requiredType.isInstance(headerAccessor))  {

600				return (T) headerAccessor;

601			}

602		}

603		return null;

604	}

605

606	/**

607	 * Return a mutable {@code MessageHeaderAccessor} for the given message attempting

608	 * to match the type of accessor used to create the message headers, or otherwise

609	 * wrapping the message with a {@code MessageHeaderAccessor} instance.

610	 * <p>This is for cases where a header needs to be updated in generic code

611	 * while preserving the accessor type for downstream processing.

612	 * @return an accessor of the required type (never {@code null})

613	 * @since 4.1

614	 */

615	public static MessageHeaderAccessor getMutableAccessor(Message<?> message) {

616		if (message.getHeaders() instanceof MutableMessageHeaders) {

617			MutableMessageHeaders mutableHeaders = (MutableMessageHeaders) message.getHeaders();

618			MessageHeaderAccessor accessor = mutableHeaders.getAccessor();

619			return (accessor.isMutable() ? accessor : accessor.createAccessor(message));

620		}

621		return new MessageHeaderAccessor(message);

622	}

625	@SuppressWarnings("serial")

626	private class MutableMessageHeaders extends MessageHeaders {

627

628		private boolean mutable = true;

629

630		public MutableMessageHeaders(@Nullable Map<String, Object> headers) {

631			super(headers, MessageHeaders.ID_VALUE_NONE, -1L);

632		}

633

634		@Override

635		public Map<String, Object> getRawHeaders() {

636			Assert.state(this.mutable, "Already immutable");

637			return super.getRawHeaders();

638		}

639

640		public void setImmutable() {

641			if (!this.mutable) {

642				return;

643			}

644

645			if (getId() == null) {

646				IdGenerator idGenerator = (MessageHeaderAccessor.this.idGenerator != null ?

647						MessageHeaderAccessor.this.idGenerator : MessageHeaders.getIdGenerator());

648				UUID id = idGenerator.generateId();

649				if (id != MessageHeaders.ID_VALUE_NONE) {

650					getRawHeaders().put(ID, id);

651				}

652			}

653

654			if (getTimestamp() == null) {

655				if (MessageHeaderAccessor.this.enableTimestamp) {

656					getRawHeaders().put(TIMESTAMP, System.currentTimeMillis());

657				}

658			}

659

660			this.mutable = false;

661		}

662

663		public boolean isMutable() {

664			return this.mutable;

665		}

666

667		public MessageHeaderAccessor getAccessor() {

668			return MessageHeaderAccessor.this;

669		}

670

671		protected Object writeReplace() {

672			// Serialize as regular MessageHeaders (without MessageHeaderAccessor reference)

673			return new MessageHeaders(this);

674		}

675	}
