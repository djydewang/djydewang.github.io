41/**

72	public AbstractView() {

76	public AbstractView(ReactiveAdapterRegistry registry) {

77		this.mediaTypes.add(ViewResolverSupport.DEFAULT_CONTENT_TYPE);

82	/**

83	 * Set the supported media types for this view.

84	 * Default is "text/html;charset=UTF-8".

85	 */

86	public void setSupportedMediaTypes(@Nullable List<MediaType> supportedMediaTypes) {

87		Assert.notEmpty(supportedMediaTypes, "MediaType List must not be empty");

88		this.mediaTypes.clear();

89		if (supportedMediaTypes != null) {

90			this.mediaTypes.addAll(supportedMediaTypes);

91		}

92	}

108		Assert.notNull(defaultCharset, "'defaultCharset' must not be null");

146	/**

147	 * Obtain the ApplicationContext for actual use.

148	 * @return the ApplicationContext (never {@code null})

149	 * @throws IllegalStateException in case of no ApplicationContext set

150	 */

151	protected final ApplicationContext obtainApplicationContext() {

152		ApplicationContext applicationContext = getApplicationContext();

153		Assert.state(applicationContext != null, "No ApplicationContext");

154		return applicationContext;

155	}

158	/**

159	 * Prepare the model to render.

160	 * @param model Map with name Strings as keys and corresponding model

161	 * objects as values (Map can also be {@code null} in case of empty model)

162	 * @param contentType the content type selected to render with which should

163	 * match one of the {@link #getSupportedMediaTypes() supported media types}.

164	 * @param exchange the current exchange

165	 * @return {@code Mono} to represent when and if rendering succeeds

166	 */

167	@Override

168	public Mono<Void> render(@Nullable Map<String, ?> model, @Nullable MediaType contentType,

169			ServerWebExchange exchange) {

170

171		if (logger.isTraceEnabled()) {

172			logger.trace("Rendering view with model " + model);

173		}

174

175		if (contentType != null) {

176			exchange.getResponse().getHeaders().setContentType(contentType);

177		}

178

179		return getModelAttributes(model, exchange).flatMap(mergedModel -> {

180			// Expose RequestContext?

181			if (this.requestContextAttribute != null) {

182				mergedModel.put(this.requestContextAttribute, createRequestContext(exchange, mergedModel));

183			}

184			return renderInternal(mergedModel, contentType, exchange);

185		});

186	}

187

188	/**

189	 * Prepare the model to use for rendering.

190	 * <p>The default implementation creates a combined output Map that includes

191	 * model as well as static attributes with the former taking precedence.

192	 */

193	protected Mono<Map<String, Object>> getModelAttributes(@Nullable Map<String, ?> model,

194			ServerWebExchange exchange) {

195

196		int size = (model != null ? model.size() : 0);

197

198		Map<String, Object> attributes = new LinkedHashMap<>(size);

199		if (model != null) {

200			attributes.putAll(model);

201		}

202

203		return resolveAsyncAttributes(attributes).then(Mono.just(attributes));

204	}

205

206	/**

207	 * By default, resolve async attributes supported by the

208	 * {@link ReactiveAdapterRegistry} to their blocking counterparts.

209	 * <p>View implementations capable of taking advantage of reactive types

210	 * can override this method if needed.

211	 * @return {@code Mono} for the completion of async attributes resolution

212	 */

213	protected Mono<Void> resolveAsyncAttributes(Map<String, Object> model) {

214

215		List<String> names = new ArrayList<>();

216		List<Mono<?>> valueMonos = new ArrayList<>();

217

218		for (Map.Entry<String, ?> entry : model.entrySet()) {

219			Object value =  entry.getValue();

220			if (value == null) {

221				continue;

222			}

223			ReactiveAdapter adapter = this.adapterRegistry.getAdapter(null, value);

224			if (adapter != null) {

225				names.add(entry.getKey());

226				if (adapter.isMultiValue()) {

227					Flux<Object> fluxValue = Flux.from(adapter.toPublisher(value));

228					valueMonos.add(fluxValue.collectList().defaultIfEmpty(Collections.emptyList()));

229				}

230				else {

231					Mono<Object> monoValue = Mono.from(adapter.toPublisher(value));

232					valueMonos.add(monoValue.defaultIfEmpty(NO_VALUE));

233				}

234			}

235		}

236

237		if (names.isEmpty()) {

238			return Mono.empty();

239		}

240

241		return Mono.zip(valueMonos,

242				values -> {

243					for (int i=0; i < values.length; i++) {

244						if (values[i] != NO_VALUE) {

245							model.put(names.get(i), values[i]);

246						}

247						else {

248							model.remove(names.get(i));

249						}

250					}

251					return NO_VALUE;

252				})

253				.then();

254	}

256	/**

257	 * Create a RequestContext to expose under the specified attribute name.

258	 * <p>The default implementation creates a standard RequestContext instance

259	 * for the given request and model. Can be overridden in subclasses for

260	 * custom instances.

261	 * @param exchange current exchange

262	 * @param model combined output Map (never {@code null}),

263	 * with dynamic values taking precedence over static attributes

264	 * @return the RequestContext instance

265	 * @see #setRequestContextAttribute

266	 */

267	protected RequestContext createRequestContext(ServerWebExchange exchange, Map<String, Object> model) {

268		return new RequestContext(exchange, model, obtainApplicationContext(), getRequestDataValueProcessor());

269	}

270

271	/**

272	 * Return the {@link RequestDataValueProcessor} to use.

273	 * <p>The default implementation looks in the {@link #getApplicationContext()

274	 * Spring configuration} for a {@code RequestDataValueProcessor} bean with

275	 * the name {@link #REQUEST_DATA_VALUE_PROCESSOR_BEAN_NAME}.

276	 * @return the RequestDataValueProcessor, or null if there is none at the

277	 * application context.

278	 */

279	@Nullable

280	protected RequestDataValueProcessor getRequestDataValueProcessor() {

281		ApplicationContext context = getApplicationContext();

282		if (context != null && context.containsBean(REQUEST_DATA_VALUE_PROCESSOR_BEAN_NAME)) {

283			return context.getBean(REQUEST_DATA_VALUE_PROCESSOR_BEAN_NAME, RequestDataValueProcessor.class);

284		}

285		return null;

286	}

288	/**

289	 * Subclasses must implement this method to actually render the view.

290	 * @param renderAttributes combined output Map (never {@code null}),

291	 * with dynamic values taking precedence over static attributes

292	 * @param contentType the content type selected to render with which should

293	 * match one of the {@link #getSupportedMediaTypes() supported media types}.

294	 *@param exchange current exchange  @return {@code Mono} to represent when

295	 * and if rendering succeeds

296	 */

297	protected abstract Mono<Void> renderInternal(Map<String, Object> renderAttributes,

298			@Nullable MediaType contentType, ServerWebExchange exchange);

300

301	@Override

302	public String toString() {

303		return getClass().getName();

304	}
