50/**

51 * View that redirects to an absolute, context relative, or current request

52 * relative URL. The URL may be a URI template in which case the URI template

53 * variables will be replaced with values available in the model. By default

54 * all primitive model attributes (or collections thereof) are exposed as HTTP

55 * query parameters (assuming they've not been used as URI template variables),

56 * but this behavior can be changed by overriding the

57 * {@link #isEligibleProperty(String, Object)} method.

58 *

59 * <p>A URL for this view is supposed to be a HTTP redirect URL, i.e.

60 * suitable for HttpServletResponse's {@code sendRedirect} method, which

61 * is what actually does the redirect if the HTTP 1.0 flag is on, or via sending

62 * back an HTTP 303 code - if the HTTP 1.0 compatibility flag is off.

63 *

64 * <p>Note that while the default value for the "contextRelative" flag is off,

65 * you will probably want to almost always set it to true. With the flag off,

66 * URLs starting with "/" are considered relative to the web server root, while

67 * with the flag on, they are considered relative to the web application root.

68 * Since most web applications will never know or care what their context path

69 * actually is, they are much better off setting this flag to true, and submitting

70 * paths which are to be considered relative to the web application root.

71 *

72 * <p><b>NOTE when using this redirect view in a Portlet environment:</b> Make sure

73 * that your controller respects the Portlet {@code sendRedirect} constraints.

74 *

75 * @author Rod Johnson

76 * @author Juergen Hoeller

77 * @author Colin Sampaleanu

78 * @author Sam Brannen

79 * @author Arjen Poutsma

80 * @author Rossen Stoyanchev

81 * @see #setContextRelative

82 * @see #setHttp10Compatible

83 * @see #setExposeModelAttributes

84 * @see javax.servlet.http.HttpServletResponse#sendRedirect

85 */

86public class RedirectView extends AbstractUrlBasedView implements SmartView {

299	 * Convert model to request parameters and redirect to the given URL.

300	 * @see #appendQueryProperties

301	 * @see #sendRedirect

302	 */

303	@Override

304	protected void renderMergedOutputModel(Map<String, Object> model, HttpServletRequest request,

305			HttpServletResponse response) throws IOException {

306

307		String targetUrl = createTargetUrl(model, request);

308		targetUrl = updateTargetUrl(targetUrl, model, request, response);

309

310		// Save flash attributes

311		RequestContextUtils.saveOutputFlashMap(targetUrl, request, response);

312

313		// Redirect

314		sendRedirect(request, response, targetUrl, this.http10Compatible);

315	}

317	/**

318	 * Create the target URL by checking if the redirect string is a URI template first,

319	 * expanding it with the given model, and then optionally appending simple type model

320	 * attributes as query String parameters.

321	 */

322	protected final String createTargetUrl(Map<String, Object> model, HttpServletRequest request)

323			throws UnsupportedEncodingException {

324

325		// Prepare target URL.

326		StringBuilder targetUrl = new StringBuilder();

327		String url = getUrl();

328		Assert.state(url != null, "'url' not set");

329

330		if (this.contextRelative && getUrl().startsWith("/")) {

331			// Do not apply context path to relative URLs.

332			targetUrl.append(request.getContextPath());

333		}

334		targetUrl.append(getUrl());

335

336		String enc = this.encodingScheme;

337		if (enc == null) {

338			enc = request.getCharacterEncoding();

339		}

340		if (enc == null) {

341			enc = WebUtils.DEFAULT_CHARACTER_ENCODING;

342		}

343

344		if (this.expandUriTemplateVariables && StringUtils.hasText(targetUrl)) {

345			Map<String, String> variables = getCurrentRequestUriVariables(request);

346			targetUrl = replaceUriTemplateVariables(targetUrl.toString(), model, variables, enc);

347		}

348		if (isPropagateQueryProperties()) {

349		 	appendCurrentQueryParams(targetUrl, request);

350		}

351		if (this.exposeModelAttributes) {

352			appendQueryProperties(targetUrl, model, enc);

353		}

354

355		return targetUrl.toString();

356	}

358	/**

359	 * Replace URI template variables in the target URL with encoded model

360	 * attributes or URI variables from the current request. Model attributes

361	 * referenced in the URL are removed from the model.

362	 * @param targetUrl the redirect URL

363	 * @param model Map that contains model attributes

364	 * @param currentUriVariables current request URI variables to use

365	 * @param encodingScheme the encoding scheme to use

366	 * @throws UnsupportedEncodingException if string encoding failed

367	 */

368	protected StringBuilder replaceUriTemplateVariables(

369			String targetUrl, Map<String, Object> model, Map<String, String> currentUriVariables, String encodingScheme)

370			throws UnsupportedEncodingException {

371

372		StringBuilder result = new StringBuilder();

373		Matcher matcher = URI_TEMPLATE_VARIABLE_PATTERN.matcher(targetUrl);

374		int endLastMatch = 0;

375		while (matcher.find()) {

376			String name = matcher.group(1);

377			Object value = (model.containsKey(name) ? model.remove(name) : currentUriVariables.get(name));

378			if (value == null) {

379				throw new IllegalArgumentException("Model has no value for key '" + name + "'");

380			}

381			result.append(targetUrl.substring(endLastMatch, matcher.start()));

382			result.append(UriUtils.encodePathSegment(value.toString(), encodingScheme));

383			endLastMatch = matcher.end();

384		}

385		result.append(targetUrl.substring(endLastMatch, targetUrl.length()));

386		return result;

387	}

389	@SuppressWarnings("unchecked")

390	private Map<String, String> getCurrentRequestUriVariables(HttpServletRequest request) {

391		String name = HandlerMapping.URI_TEMPLATE_VARIABLES_ATTRIBUTE;

392		Map<String, String> uriVars = (Map<String, String>) request.getAttribute(name);

393		return (uriVars != null) ? uriVars : Collections.<String, String> emptyMap();

394	}

396	/**

397	 * Append the query string of the current request to the target redirect URL.

398	 * @param targetUrl the StringBuilder to append the properties to

399	 * @param request the current request

400	 * @since 4.1

401	 */

402	protected void appendCurrentQueryParams(StringBuilder targetUrl, HttpServletRequest request) {

403		String query = request.getQueryString();

404		if (StringUtils.hasText(query)) {

405			// Extract anchor fragment, if any.

406			String fragment = null;

407			int anchorIndex = targetUrl.indexOf("#");

408			if (anchorIndex > -1) {

409				fragment = targetUrl.substring(anchorIndex);

410				targetUrl.delete(anchorIndex, targetUrl.length());

411			}

412

413			if (targetUrl.toString().indexOf('?') < 0) {

414				targetUrl.append('?').append(query);

415			}

416			else {

417				targetUrl.append('&').append(query);

418			}

419			// Append anchor fragment, if any, to end of URL.

420			if (fragment != null) {

421				targetUrl.append(fragment);

422			}

423		}

424	}

426	/**

427	 * Append query properties to the redirect URL.

428	 * Stringifies, URL-encodes and formats model attributes as query properties.

429	 * @param targetUrl the StringBuilder to append the properties to

430	 * @param model Map that contains model attributes

431	 * @param encodingScheme the encoding scheme to use

432	 * @throws UnsupportedEncodingException if string encoding failed

433	 * @see #queryProperties

434	 */

435	@SuppressWarnings("unchecked")

436	protected void appendQueryProperties(StringBuilder targetUrl, Map<String, Object> model, String encodingScheme)

437			throws UnsupportedEncodingException {

438

439		// Extract anchor fragment, if any.

440		String fragment = null;

441		int anchorIndex = targetUrl.indexOf("#");

442		if (anchorIndex > -1) {

443			fragment = targetUrl.substring(anchorIndex);

444			targetUrl.delete(anchorIndex, targetUrl.length());

445		}

446

447		// If there aren't already some parameters, we need a "?".

448		boolean first = (targetUrl.toString().indexOf('?') < 0);

449		for (Map.Entry<String, Object> entry : queryProperties(model).entrySet()) {

450			Object rawValue = entry.getValue();

451			Iterator<Object> valueIter;

452			if (rawValue != null && rawValue.getClass().isArray()) {

453				valueIter = Arrays.asList(ObjectUtils.toObjectArray(rawValue)).iterator();

454			}

455			else if (rawValue instanceof Collection) {

456				valueIter = ((Collection<Object>) rawValue).iterator();

457			}

458			else {

459				valueIter = Collections.singleton(rawValue).iterator();

460			}

461			while (valueIter.hasNext()) {

462				Object value = valueIter.next();

463				if (first) {

464					targetUrl.append('?');

465					first = false;

466				}

467				else {

468					targetUrl.append('&');

469				}

470				String encodedKey = urlEncode(entry.getKey(), encodingScheme);

471				String encodedValue = (value != null ? urlEncode(value.toString(), encodingScheme) : "");

472				targetUrl.append(encodedKey).append('=').append(encodedValue);

473			}

474		}

475

476		// Append anchor fragment, if any, to end of URL.

477		if (fragment != null) {

478			targetUrl.append(fragment);

479		}

480	}

482	/**

483	 * Determine name-value pairs for query strings, which will be stringified,

484	 * URL-encoded and formatted by {@link #appendQueryProperties}.

485	 * <p>This implementation filters the model through checking

486	 * {@link #isEligibleProperty(String, Object)} for each element,

487	 * by default accepting Strings, primitives and primitive wrappers only.

488	 * @param model the original model Map

489	 * @return the filtered Map of eligible query properties

490	 * @see #isEligibleProperty(String, Object)

491	 */

492	protected Map<String, Object> queryProperties(Map<String, Object> model) {

493		Map<String, Object> result = new LinkedHashMap<>();

494		model.forEach((name, value) -> {

495			if (isEligibleProperty(name, value)) {

496				result.put(name, value);

497			}

498		});

499		return result;

500	}

502	/**

503	 * Determine whether the given model element should be exposed

504	 * as a query property.

505	 * <p>The default implementation considers Strings and primitives

506	 * as eligible, and also arrays and Collections/Iterables with

507	 * corresponding elements. This can be overridden in subclasses.

508	 * @param key the key of the model element

509	 * @param value the value of the model element

510	 * @return whether the element is eligible as query property

511	 */

512	protected boolean isEligibleProperty(String key, @Nullable Object value) {

513		if (value == null) {

514			return false;

515		}

516		if (isEligibleValue(value)) {

517			return true;

518		}

519		if (value.getClass().isArray()) {

520			int length = Array.getLength(value);

521			if (length == 0) {

522				return false;

523			}

524			for (int i = 0; i < length; i++) {

525				Object element = Array.get(value, i);

526				if (!isEligibleValue(element)) {

527					return false;

528				}

529			}

530			return true;

531		}

532		if (value instanceof Collection) {

533			Collection<?> coll = (Collection<?>) value;

534			if (coll.isEmpty()) {

535				return false;

536			}

537			for (Object element : coll) {

538				if (!isEligibleValue(element)) {

539					return false;

540				}

541			}

542			return true;

543		}

544		return false;

545	}

548	 * Determine whether the given model element value is eligible for exposure.

549	 * <p>The default implementation considers primitives, Strings, Numbers, Dates,

550	 * URIs, URLs and Locale objects as eligible. This can be overridden in subclasses.

551	 * @param value the model element value

573	/**

574	 * Find the registered {@link RequestDataValueProcessor}, if any, and allow

575	 * it to update the redirect target URL.

576	 * @param targetUrl the given redirect URL

577	 * @return the updated URL or the same as URL as the one passed in

578	 */

579	protected String updateTargetUrl(String targetUrl, Map<String, Object> model,

580			HttpServletRequest request, HttpServletResponse response) {

581

582		WebApplicationContext wac = getWebApplicationContext();

583		if (wac == null) {

584			wac = RequestContextUtils.findWebApplicationContext(request, getServletContext());

585		}

586

587		if (wac != null && wac.containsBean(RequestContextUtils.REQUEST_DATA_VALUE_PROCESSOR_BEAN_NAME)) {

588			RequestDataValueProcessor processor = wac.getBean(

589					RequestContextUtils.REQUEST_DATA_VALUE_PROCESSOR_BEAN_NAME, RequestDataValueProcessor.class);

590			return processor.processUrl(request, targetUrl);

591		}

592

593		return targetUrl;

594	}

596	/**

597	 * Send a redirect back to the HTTP client

598	 * @param request current HTTP request (allows for reacting to request method)

599	 * @param response current HTTP response (for sending response headers)

600	 * @param targetUrl the target URL to redirect to

601	 * @param http10Compatible whether to stay compatible with HTTP 1.0 clients

602	 * @throws IOException if thrown by response methods

603	 */

604	protected void sendRedirect(HttpServletRequest request, HttpServletResponse response,

605			String targetUrl, boolean http10Compatible) throws IOException {

606

607		String encodedURL = (isRemoteHost(targetUrl) ? targetUrl : response.encodeRedirectURL(targetUrl));

608		if (http10Compatible) {

609			HttpStatus attributeStatusCode = (HttpStatus) request.getAttribute(View.RESPONSE_STATUS_ATTRIBUTE);

610			if (this.statusCode != null) {

611				response.setStatus(this.statusCode.value());

612				response.setHeader("Location", encodedURL);

613			}

614			else if (attributeStatusCode != null) {

615				response.setStatus(attributeStatusCode.value());

616				response.setHeader("Location", encodedURL);

617			}

618			else {

619				// Send status code 302 by default.

620				response.sendRedirect(encodedURL);

621			}

622		}

623		else {

624			HttpStatus statusCode = getHttp11StatusCode(request, response, targetUrl);

625			response.setStatus(statusCode.value());

626			response.setHeader("Location", encodedURL);

627		}

628	}

630	/**

631	 * Whether the given targetUrl has a host that is a "foreign" system in which

632	 * case {@link HttpServletResponse#encodeRedirectURL} will not be applied.

633	 * This method returns {@code true} if the {@link #setHosts(String[])}

634	 * property is configured and the target URL has a host that does not match.

635	 * @param targetUrl the target redirect URL

636	 * @return {@code true} the target URL has a remote host, {@code false} if it

637	 * the URL does not have a host or the "host" property is not configured.

638	 * @since 4.3

639	 */

640	protected boolean isRemoteHost(String targetUrl) {

641		if (ObjectUtils.isEmpty(getHosts())) {

642			return false;

643		}

644		String targetHost = UriComponentsBuilder.fromUriString(targetUrl).build().getHost();

645		if (StringUtils.isEmpty(targetHost)) {

646			return false;

647		}

648		for (String host : getHosts()) {

649			if (targetHost.equals(host)) {

650				return false;

651			}

652		}

653		return true;

654	}

656	/**

657	 * Determines the status code to use for HTTP 1.1 compatible requests.

658	 * <p>The default implementation returns the {@link #setStatusCode(HttpStatus) statusCode}

659	 * property if set, or the value of the {@link #RESPONSE_STATUS_ATTRIBUTE} attribute.

660	 * If neither are set, it defaults to {@link HttpStatus#SEE_OTHER} (303).

661	 * @param request the request to inspect

662	 * @param response the servlet response

663	 * @param targetUrl the target URL

664	 * @return the response status

665	 */

666	protected HttpStatus getHttp11StatusCode(

667			HttpServletRequest request, HttpServletResponse response, String targetUrl) {

668

669		if (this.statusCode != null) {

670			return this.statusCode;
