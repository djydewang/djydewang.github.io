43/**

44 * Production implementation of the {@link JavaMailSender} interface,

45 * supporting both JavaMail {@link MimeMessage MimeMessages} and Spring

46 * {@link SimpleMailMessage SimpleMailMessages}. Can also be used as a

47 * plain {@link org.springframework.mail.MailSender} implementation.

48 *

49 * <p>Allows for defining all settings locally as bean properties.

50 * Alternatively, a pre-configured JavaMail {@link javax.mail.Session} can be

51 * specified, possibly pulled from an application server's JNDI environment.

52 *

53 * <p>Non-default properties in this object will always override the settings

54 * in the JavaMail {@code Session}. Note that if overriding all values locally,

55 * there is no added value in setting a pre-configured {@code Session}.

56 *

57 * @author Dmitriy Kopylenko

58 * @author Juergen Hoeller

59 * @since 10.09.2003

60 * @see javax.mail.internet.MimeMessage

61 * @see javax.mail.Session

62 * @see #setSession

63 * @see #setJavaMailProperties

64 * @see #setHost

65 * @see #setPort

66 * @see #setUsername

67 * @see #setPassword

68 */

69public class JavaMailSenderImpl implements JavaMailSender {

310		send(new SimpleMailMessage[] {simpleMessage});

313	@Override

314	public void send(SimpleMailMessage... simpleMessages) throws MailException {

315		List<MimeMessage> mimeMessages = new ArrayList<>(simpleMessages.length);

316		for (SimpleMailMessage simpleMessage : simpleMessages) {

317			MimeMailMessage message = new MimeMailMessage(createMimeMessage());

318			simpleMessage.copyTo(message);

319			mimeMessages.add(message.getMimeMessage());

320		}

321		doSend(mimeMessages.toArray(new MimeMessage[0]), simpleMessages);

322	}

367	@Override

368	public void send(MimeMessagePreparator... mimeMessagePreparators) throws MailException {

369		try {

370			List<MimeMessage> mimeMessages = new ArrayList<>(mimeMessagePreparators.length);

371			for (MimeMessagePreparator preparator : mimeMessagePreparators) {

372				MimeMessage mimeMessage = createMimeMessage();

373				preparator.prepare(mimeMessage);

374				mimeMessages.add(mimeMessage);

375			}

376			send(mimeMessages.toArray(new MimeMessage[0]));

377		}

378		catch (MailException ex) {

379			throw ex;

380		}

381		catch (MessagingException ex) {

382			throw new MailParseException(ex);

383		}

384		catch (Exception ex) {

385			throw new MailPreparationException(ex);

386		}

387	}

389	/**

390	 * Validate that this instance can connect to the server that it is configured

391	 * for. Throws a {@link MessagingException} if the connection attempt failed.

392	 */

393	public void testConnection() throws MessagingException {

394		Transport transport = null;

395		try {

396			transport = connectTransport();

397		}

398		finally {

399			if (transport != null) {

400				transport.close();

401			}

402		}

403	}

405	/**

406	 * Actually send the given array of MimeMessages via JavaMail.

407	 * @param mimeMessages MimeMessage objects to send

408	 * @param originalMessages corresponding original message objects

409	 * that the MimeMessages have been created from (with same array

410	 * length and indices as the "mimeMessages" array), if any

411	 * @throws org.springframework.mail.MailAuthenticationException

412	 * in case of authentication failure

413	 * @throws org.springframework.mail.MailSendException

414	 * in case of failure when sending a message

415	 */

416	protected void doSend(MimeMessage[] mimeMessages, @Nullable Object[] originalMessages) throws MailException {

417		Map<Object, Exception> failedMessages = new LinkedHashMap<>();

418		Transport transport = null;

419

420		try {

421			for (int i = 0; i < mimeMessages.length; i++) {

422

423				// Check transport connection first...

424				if (transport == null || !transport.isConnected()) {

425					if (transport != null) {

426						try {

427							transport.close();

428						}

429						catch (Exception ex) {

430							// Ignore - we're reconnecting anyway

431						}

432						transport = null;

433					}

434					try {

435						transport = connectTransport();

436					}

437					catch (AuthenticationFailedException ex) {

438						throw new MailAuthenticationException(ex);

439					}

440					catch (Exception ex) {

441						// Effectively, all remaining messages failed...

442						for (int j = i; j < mimeMessages.length; j++) {

443							Object original = (originalMessages != null ? originalMessages[j] : mimeMessages[j]);

444							failedMessages.put(original, ex);

445						}

446						throw new MailSendException("Mail server connection failed", ex, failedMessages);

447					}

448				}

449

450				// Send message via current transport...

451				MimeMessage mimeMessage = mimeMessages[i];

452				try {

453					if (mimeMessage.getSentDate() == null) {

454						mimeMessage.setSentDate(new Date());

455					}

456					String messageId = mimeMessage.getMessageID();

457					mimeMessage.saveChanges();

458					if (messageId != null) {

459						// Preserve explicitly specified message id...

460						mimeMessage.setHeader(HEADER_MESSAGE_ID, messageId);

461					}

462					transport.sendMessage(mimeMessage, mimeMessage.getAllRecipients());

463				}

464				catch (Exception ex) {

465					Object original = (originalMessages != null ? originalMessages[i] : mimeMessage);

466					failedMessages.put(original, ex);

467				}

468			}

469		}

470		finally {

471			try {

472				if (transport != null) {

473					transport.close();

474				}

475			}

476			catch (Exception ex) {

477				if (!failedMessages.isEmpty()) {

478					throw new MailSendException("Failed to close server connection after message failures", ex,

479							failedMessages);

480				}

481				else {

482					throw new MailSendException("Failed to close server connection after message sending", ex);

483				}

484			}

485		}

486

487		if (!failedMessages.isEmpty()) {

488			throw new MailSendException(failedMessages);

489		}

490	}

492	/**

493	 * Obtain and connect a Transport from the underlying JavaMail Session,

494	 * passing in the specified host, port, username, and password.

495	 * @return the connected Transport object

496	 * @throws MessagingException if the connect attempt failed

497	 * @since 4.1.2

498	 * @see #getTransport

499	 * @see #getHost()

500	 * @see #getPort()

501	 * @see #getUsername()

502	 * @see #getPassword()

503	 */

504	protected Transport connectTransport() throws MessagingException {

505		String username = getUsername();

506		String password = getPassword();

507		if ("".equals(username)) {  // probably from a placeholder

508			username = null;

509			if ("".equals(password)) {  // in conjunction with "" username, this means no password to use

510				password = null;

511			}

512		}

513

514		Transport transport = getTransport(getSession());

515		transport.connect(getHost(), getPort(), username, password);

516		return transport;

517	}

519	/**

520	 * Obtain a Transport object from the given JavaMail Session,

521	 * using the configured protocol.

522	 * <p>Can be overridden in subclasses, e.g. to return a mock Transport object.

523	 * @see javax.mail.Session#getTransport(String)

524	 * @see #getSession()

525	 * @see #getProtocol()

526	 */

527	protected Transport getTransport(Session session) throws NoSuchProviderException {

528		String protocol	= getProtocol();

529		if (protocol == null) {

530			protocol = session.getProperty("mail.transport.protocol");

531			if (protocol == null) {

532				protocol = DEFAULT_PROTOCOL;

533			}

534		}

535		return session.getTransport(protocol);

536	}
