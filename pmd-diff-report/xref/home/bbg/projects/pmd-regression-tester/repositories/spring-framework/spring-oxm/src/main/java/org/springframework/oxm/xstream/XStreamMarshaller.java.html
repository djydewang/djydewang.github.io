88/**

89 * Implementation of the {@code Marshaller} interface for XStream.

90 *

91 * <p>By default, XStream does not require any further configuration and can (un)marshal

92 * any class on the classpath. As such, it is <b>not recommended to use the

93 * {@code XStreamMarshaller} to unmarshal XML from external sources</b> (i.e. the Web),

94 * as this can result in <b>security vulnerabilities</b>. If you do use the

95 * {@code XStreamMarshaller} to unmarshal external XML, set the

96 * {@link #setSupportedClasses(Class[]) supportedClasses} and

97 * {@link #setConverters(ConverterMatcher[]) converters} properties (possibly using

98 * a {@link CatchAllConverter}) or override the {@link #customizeXStream(XStream)}

99 * method to make sure it only accepts the classes you want it to support.

100 *

101 * <p>Due to XStream's API, it is required to set the encoding used for writing to

102 * OutputStreams. It defaults to {@code UTF-8}.

103 *

104 * <p><b>NOTE:</b> XStream is an XML serialization library, not a data binding library.

105 * Therefore, it has limited namespace support. As such, it is rather unsuitable for

106 * usage within Web Services.

107 *

108 * <p>This marshaller requires XStream 1.4.5 or higher, as of Spring 4.3.

109 * Note that {@link XStream} construction has been reworked in 4.0, with the

110 * stream driver and the class loader getting passed into XStream itself now.

111 *

112 * @author Peter Meijer

113 * @author Arjen Poutsma

114 * @author Juergen Hoeller

115 * @since 3.0

116 */

117public class XStreamMarshaller extends AbstractMarshaller implements BeanClassLoaderAware, InitializingBean {

414	 * and {@link #customizeXStream}.

423	/**

424	 * Construct an XStream instance, either using one of the

425	 * standard constructors or creating a custom subclass.

426	 * @return the {@code XStream} instance

427	 */

428	protected XStream constructXStream() {

429		return new XStream(this.reflectionProvider, getDefaultDriver(), new ClassLoaderReference(this.beanClassLoader),

430				this.mapper, this.converterLookup, this.converterRegistry) {

431			@Override

432			protected MapperWrapper wrapMapper(MapperWrapper next) {

433				MapperWrapper mapperToWrap = next;

434				if (mapperWrappers != null) {

435					for (Class<? extends MapperWrapper> mapperWrapper : mapperWrappers) {

436						Constructor<? extends MapperWrapper> ctor;

437						try {

438							ctor = mapperWrapper.getConstructor(Mapper.class);

439						}

440						catch (NoSuchMethodException ex) {

441							try {

442								ctor = mapperWrapper.getConstructor(MapperWrapper.class);

443							}

444							catch (NoSuchMethodException ex2) {

445								throw new IllegalStateException("No appropriate MapperWrapper constructor found: " + mapperWrapper);

446							}

447						}

448						try {

449							mapperToWrap = ctor.newInstance(mapperToWrap);

450						}

451						catch (Throwable ex) {

452							throw new IllegalStateException("Failed to construct MapperWrapper: " + mapperWrapper);

453						}

454					}

455				}

456				return mapperToWrap;

457			}

458		};

459	}

461	/**

462	 * Configure the XStream instance with this marshaller's bean properties.

463	 * @param xstream the {@code XStream} instance

464	 */

465	protected void configureXStream(XStream xstream) {

466		if (this.converters != null) {

467			for (int i = 0; i < this.converters.length; i++) {

468				if (this.converters[i] instanceof Converter) {

469					xstream.registerConverter((Converter) this.converters[i], i);

470				}

471				else if (this.converters[i] instanceof SingleValueConverter) {

472					xstream.registerConverter((SingleValueConverter) this.converters[i], i);

473				}

474				else {

475					throw new IllegalArgumentException("Invalid ConverterMatcher [" + this.converters[i] + "]");

476				}

477			}

478		}

479

480		if (this.marshallingStrategy != null) {

481			xstream.setMarshallingStrategy(this.marshallingStrategy);

482		}

483		if (this.mode != null) {

484			xstream.setMode(this.mode);

485		}

486

487		try {

488			if (this.aliases != null) {

489				Map<String, Class<?>> classMap = toClassMap(this.aliases);

490				classMap.forEach(xstream::alias);

491			}

492			if (this.aliasesByType != null) {

493				Map<String, Class<?>> classMap = toClassMap(this.aliasesByType);

494				classMap.forEach(xstream::aliasType);

495			}

496			if (this.fieldAliases != null) {

497				for (Map.Entry<String, String> entry : this.fieldAliases.entrySet()) {

498					String alias = entry.getValue();

499					String field = entry.getKey();

500					int idx = field.lastIndexOf('.');

501					if (idx != -1) {

502						String className = field.substring(0, idx);

503						Class<?> clazz = ClassUtils.forName(className, this.beanClassLoader);

504						String fieldName = field.substring(idx + 1);

505						xstream.aliasField(alias, clazz, fieldName);

506					}

507					else {

508						throw new IllegalArgumentException("Field name [" + field + "] does not contain '.'");

509					}

510				}

511			}

512		}

513		catch (ClassNotFoundException ex) {

514			throw new IllegalStateException("Failed to load specified alias class", ex);

515		}

516

517		if (this.useAttributeForTypes != null) {

518			for (Class<?> type : this.useAttributeForTypes) {

519				xstream.useAttributeFor(type);

520			}

521		}

522		if (this.useAttributeFor != null) {

523			for (Map.Entry<?, ?> entry : this.useAttributeFor.entrySet()) {

524				if (entry.getKey() instanceof String) {

525					if (entry.getValue() instanceof Class) {

526						xstream.useAttributeFor((String) entry.getKey(), (Class<?>) entry.getValue());

527					}

528					else {

529						throw new IllegalArgumentException(

530								"'useAttributesFor' takes Map<String, Class> when using a map key of type String");

531					}

532				}

533				else if (entry.getKey() instanceof Class) {

534					Class<?> key = (Class<?>) entry.getKey();

535					if (entry.getValue() instanceof String) {

536						xstream.useAttributeFor(key, (String) entry.getValue());

537					}

538					else if (entry.getValue() instanceof List) {

539						@SuppressWarnings("unchecked")

540						List<Object> listValue = (List<Object>) entry.getValue();

541						for (Object element : listValue) {

542							if (element instanceof String) {

543								xstream.useAttributeFor(key, (String) element);

544							}

545						}

546					}

547					else {

548						throw new IllegalArgumentException("'useAttributesFor' property takes either Map<Class, String> " +

549								"or Map<Class, List<String>> when using a map key of type Class");

550					}

551				}

552				else {

553					throw new IllegalArgumentException(

554							"'useAttributesFor' property takes either a map key of type String or Class");

555				}

556			}

557		}

558

559		if (this.implicitCollections != null) {

560			this.implicitCollections.forEach((key, fields) -> {

561				String[] collectionFields = StringUtils.commaDelimitedListToStringArray(fields);

562				for (String collectionField : collectionFields) {

563					xstream.addImplicitCollection(key, collectionField);

564				}

565			});

566		}

567		if (this.omittedFields != null) {

568			this.omittedFields.forEach((key, value) -> {

569				String[] fields = StringUtils.commaDelimitedListToStringArray(value);

570				for (String field : fields) {

571					xstream.omitField(key, field);

572				}

573			});

574		}

575

576		if (this.annotatedClasses != null) {

577			xstream.processAnnotations(this.annotatedClasses);

578		}

579		if (this.autodetectAnnotations) {

580			xstream.autodetectAnnotations(true);

581		}

582	}

584	private Map<String, Class<?>> toClassMap(Map<String, ?> map) throws ClassNotFoundException {

585		Map<String, Class<?>> result = new LinkedHashMap<>(map.size());

586		for (Map.Entry<String, ?> entry : map.entrySet()) {

587			String key = entry.getKey();

588			Object value = entry.getValue();

589			Class<?> type;

590			if (value instanceof Class) {

591				type = (Class<?>) value;

592			}

593			else if (value instanceof String) {

594				String className = (String) value;

595				type = ClassUtils.forName(className, this.beanClassLoader);

596			}

597			else {

598				throw new IllegalArgumentException("Unknown value [" + value + "] - expected String or Class");

599			}

600			result.put(key, type);

601		}

602		return result;

603	}

627	@Override

628	public boolean supports(Class<?> clazz) {

629		if (ObjectUtils.isEmpty(this.supportedClasses)) {

630			return true;

631		}

632		else {

633			for (Class<?> supportedClass : this.supportedClasses) {

634				if (supportedClass.isAssignableFrom(clazz)) {

635					return true;

636				}

637			}

638			return false;

639		}

640	}

645	@Override

646	protected void marshalDomNode(Object graph, Node node) throws XmlMappingException {

647		HierarchicalStreamWriter streamWriter;

648		if (node instanceof Document) {

649			streamWriter = new DomWriter((Document) node, this.nameCoder);

650		}

651		else if (node instanceof Element) {

652			streamWriter = new DomWriter((Element) node, node.getOwnerDocument(), this.nameCoder);

653		}

654		else {

655			throw new IllegalArgumentException("DOMResult contains neither Document nor Element");

656		}

657		doMarshal(graph, streamWriter, null);

658	}

660	@Override

661	protected void marshalXmlEventWriter(Object graph, XMLEventWriter eventWriter) throws XmlMappingException {

662		ContentHandler contentHandler = StaxUtils.createContentHandler(eventWriter);

663		LexicalHandler lexicalHandler = null;

664		if (contentHandler instanceof LexicalHandler) {

665			lexicalHandler = (LexicalHandler) contentHandler;

666		}

667		marshalSaxHandlers(graph, contentHandler, lexicalHandler);

668	}

670	@Override

671	protected void marshalXmlStreamWriter(Object graph, XMLStreamWriter streamWriter) throws XmlMappingException {

672		try {

673			doMarshal(graph, new StaxWriter(new QNameMap(), streamWriter, this.nameCoder), null);

674		}

675		catch (XMLStreamException ex) {

676			throw convertXStreamException(ex, true);

677		}

678	}

680	@Override

681	protected void marshalSaxHandlers(Object graph, ContentHandler contentHandler, @Nullable LexicalHandler lexicalHandler)

682			throws XmlMappingException {

683

684		SaxWriter saxWriter = new SaxWriter(this.nameCoder);

685		saxWriter.setContentHandler(contentHandler);

686		doMarshal(graph, saxWriter, null);

721	/**

722	 * Marshals the given graph to the given XStream HierarchicalStreamWriter.

723	 * Converts exceptions using {@link #convertXStreamException}.

724	 */

725	private void doMarshal(Object graph, HierarchicalStreamWriter streamWriter, @Nullable DataHolder dataHolder) {

726		try {

727			getXStream().marshal(graph, streamWriter, dataHolder);

728		}

729		catch (Exception ex) {

730			throw convertXStreamException(ex, true);

731		}

732		finally {

733			try {

734				streamWriter.flush();

735			}

736			catch (Exception ex) {

737				logger.debug("Could not flush HierarchicalStreamWriter", ex);

738			}

739		}

740	}

820    /**

821     * Unmarshals the given graph to the given XStream HierarchicalStreamWriter.

822     * Converts exceptions using {@link #convertXStreamException}.

823     */

824    private Object doUnmarshal(HierarchicalStreamReader streamReader, @Nullable DataHolder dataHolder) {

825        try {

826            return getXStream().unmarshal(streamReader, null, dataHolder);

827        }

828        catch (Exception ex) {

829            throw convertXStreamException(ex, false);

830        }

831    }

834    /**

835     * Convert the given XStream exception to an appropriate exception from the

836     * {@code org.springframework.oxm} hierarchy.

837     * <p>A boolean flag is used to indicate whether this exception occurs during marshalling or

838     * unmarshalling, since XStream itself does not make this distinction in its exception hierarchy.

839     * @param ex XStream exception that occurred

840     * @param marshalling indicates whether the exception occurs during marshalling ({@code true}),

841     * or unmarshalling ({@code false})

842     * @return the corresponding {@code XmlMappingException}

843     */

844	protected XmlMappingException convertXStreamException(Exception ex, boolean marshalling) {

845		if (ex instanceof StreamException || ex instanceof CannotResolveClassException ||

846				ex instanceof ConversionException) {

847			if (marshalling) {

848				return new MarshallingFailureException("XStream marshalling exception",  ex);

849			}

850			else {

851				return new UnmarshallingFailureException("XStream unmarshalling exception", ex);

852			}

853		}

854		else {

855			// fallback

856			return new UncategorizedMappingException("Unknown XStream exception", ex);

857		}

858	}
