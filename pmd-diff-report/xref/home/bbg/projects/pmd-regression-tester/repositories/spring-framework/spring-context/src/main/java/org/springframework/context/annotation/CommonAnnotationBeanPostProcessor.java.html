75/**

76 * {@link org.springframework.beans.factory.config.BeanPostProcessor} implementation

77 * that supports common Java annotations out of the box, in particular the JSR-250

78 * annotations in the {@code javax.annotation} package. These common Java

79 * annotations are supported in many Java EE 5 technologies (e.g. JSF 1.2),

80 * as well as in Java 6's JAX-WS.

81 *

82 * <p>This post-processor includes support for the {@link javax.annotation.PostConstruct}

83 * and {@link javax.annotation.PreDestroy} annotations - as init annotation

84 * and destroy annotation, respectively - through inheriting from

85 * {@link InitDestroyAnnotationBeanPostProcessor} with pre-configured annotation types.

86 *

87 * <p>The central element is the {@link javax.annotation.Resource} annotation

88 * for annotation-driven injection of named beans, by default from the containing

89 * Spring BeanFactory, with only {@code mappedName} references resolved in JNDI.

90 * The {@link #setAlwaysUseJndiLookup "alwaysUseJndiLookup" flag} enforces JNDI lookups

91 * equivalent to standard Java EE 5 resource injection for {@code name} references

92 * and default names as well. The target beans can be simple POJOs, with no special

93 * requirements other than the type having to match.

94 *

95 * <p>The JAX-WS {@link javax.xml.ws.WebServiceRef} annotation is supported too,

96 * analogous to {@link javax.annotation.Resource} but with the capability of creating

97 * specific JAX-WS service endpoints. This may either point to an explicitly defined

98 * resource by name or operate on a locally specified JAX-WS service class. Finally,

99 * this post-processor also supports the EJB 3 {@link javax.ejb.EJB} annotation,

100 * analogous to {@link javax.annotation.Resource} as well, with the capability to

101 * specify both a local bean name and a global JNDI name for fallback retrieval.

102 * The target beans can be plain POJOs as well as EJB 3 Session Beans in this case.

103 *

104 * <p>The common annotations supported by this post-processor are available in

105 * Java 6 (JDK 1.6) as well as in Java EE 5/6 (which provides a standalone jar for

106 * its common annotations as well, allowing for use in any Java 5 based application).

107 *

108 * <p>For default usage, resolving resource names as Spring bean names,

109 * simply define the following in your application context:

110 *

111 * <pre class="code">

112 * &lt;bean class="org.springframework.context.annotation.CommonAnnotationBeanPostProcessor"/&gt;</pre>

113 *

114 * For direct JNDI access, resolving resource names as JNDI resource references

115 * within the Java EE application's "java:comp/env/" namespace, use the following:

116 *

117 * <pre class="code">

118 * &lt;bean class="org.springframework.context.annotation.CommonAnnotationBeanPostProcessor"&gt;

119 *   &lt;property name="alwaysUseJndiLookup" value="true"/&gt;

120 * &lt;/bean&gt;</pre>

121 *

122 * {@code mappedName} references will always be resolved in JNDI,

123 * allowing for global JNDI names (including "java:" prefix) as well. The

124 * "alwaysUseJndiLookup" flag just affects {@code name} references and

125 * default names (inferred from the field name / property name).

126 *

127 * <p><b>NOTE:</b> A default CommonAnnotationBeanPostProcessor will be registered

128 * by the "context:annotation-config" and "context:component-scan" XML tags.

129 * Remove or turn off the default annotation configuration there if you intend

130 * to specify a custom CommonAnnotationBeanPostProcessor bean definition!

131 * <p><b>NOTE:</b> Annotation injection will be performed <i>before</i> XML injection; thus

132 * the latter configuration will override the former for properties wired through

133 * both approaches.

134 *

135 * @author Juergen Hoeller

136 * @since 2.5

137 * @see #setAlwaysUseJndiLookup

138 * @see #setResourceFactory

139 * @see org.springframework.beans.factory.annotation.InitDestroyAnnotationBeanPostProcessor

140 * @see org.springframework.beans.factory.annotation.AutowiredAnnotationBeanPostProcessor

141 */

142@SuppressWarnings("serial")

143public class CommonAnnotationBeanPostProcessor extends InitDestroyAnnotationBeanPostProcessor

144		implements InstantiationAwareBeanPostProcessor, BeanFactoryAware, Serializable {

327	private InjectionMetadata findResourceMetadata(String beanName, final Class<?> clazz, @Nullable PropertyValues pvs) {

328		// Fall back to class name as cache key, for backwards compatibility with custom callers.

329		String cacheKey = (StringUtils.hasLength(beanName) ? beanName : clazz.getName());

330		// Quick check on the concurrent map first, with minimal locking.

331		InjectionMetadata metadata = this.injectionMetadataCache.get(cacheKey);

332		if (InjectionMetadata.needsRefresh(metadata, clazz)) {

333			synchronized (this.injectionMetadataCache) {

334				metadata = this.injectionMetadataCache.get(cacheKey);

335				if (InjectionMetadata.needsRefresh(metadata, clazz)) {

336					if (metadata != null) {

337						metadata.clear(pvs);

338					}

339					metadata = buildResourceMetadata(clazz);

340					this.injectionMetadataCache.put(cacheKey, metadata);

341				}

342			}

343		}

344		return metadata;

345	}

347	private InjectionMetadata buildResourceMetadata(final Class<?> clazz) {

348		LinkedList<InjectionMetadata.InjectedElement> elements = new LinkedList<>();

349		Class<?> targetClass = clazz;

350

351		do {

352			final LinkedList<InjectionMetadata.InjectedElement> currElements =

353					new LinkedList<>();

354

355			ReflectionUtils.doWithLocalFields(targetClass, field -> {

356				if (webServiceRefClass != null && field.isAnnotationPresent(webServiceRefClass)) {

357					if (Modifier.isStatic(field.getModifiers())) {

358						throw new IllegalStateException("@WebServiceRef annotation is not supported on static fields");

359					}

360					currElements.add(new WebServiceRefElement(field, field, null));

361				}

362				else if (ejbRefClass != null && field.isAnnotationPresent(ejbRefClass)) {

363					if (Modifier.isStatic(field.getModifiers())) {

364						throw new IllegalStateException("@EJB annotation is not supported on static fields");

365					}

366					currElements.add(new EjbRefElement(field, field, null));

367				}

368				else if (field.isAnnotationPresent(Resource.class)) {

369					if (Modifier.isStatic(field.getModifiers())) {

370						throw new IllegalStateException("@Resource annotation is not supported on static fields");

371					}

372					if (!ignoredResourceTypes.contains(field.getType().getName())) {

373						currElements.add(new ResourceElement(field, field, null));

374					}

375				}

376			});

377

378			ReflectionUtils.doWithLocalMethods(targetClass, method -> {

379				Method bridgedMethod = BridgeMethodResolver.findBridgedMethod(method);

380				if (!BridgeMethodResolver.isVisibilityBridgeMethodPair(method, bridgedMethod)) {

381					return;

382				}

383				if (method.equals(ClassUtils.getMostSpecificMethod(method, clazz))) {

384					if (webServiceRefClass != null && bridgedMethod.isAnnotationPresent(webServiceRefClass)) {

385						if (Modifier.isStatic(method.getModifiers())) {

386							throw new IllegalStateException("@WebServiceRef annotation is not supported on static methods");

387						}

388						if (method.getParameterCount() != 1) {

389							throw new IllegalStateException("@WebServiceRef annotation requires a single-arg method: " + method);

390						}

391						PropertyDescriptor pd = BeanUtils.findPropertyForMethod(bridgedMethod, clazz);

392						currElements.add(new WebServiceRefElement(method, bridgedMethod, pd));

393					}

394					else if (ejbRefClass != null && bridgedMethod.isAnnotationPresent(ejbRefClass)) {

395						if (Modifier.isStatic(method.getModifiers())) {

396							throw new IllegalStateException("@EJB annotation is not supported on static methods");

397						}

398						if (method.getParameterCount() != 1) {

399							throw new IllegalStateException("@EJB annotation requires a single-arg method: " + method);

400						}

401						PropertyDescriptor pd = BeanUtils.findPropertyForMethod(bridgedMethod, clazz);

402						currElements.add(new EjbRefElement(method, bridgedMethod, pd));

403					}

404					else if (bridgedMethod.isAnnotationPresent(Resource.class)) {

405						if (Modifier.isStatic(method.getModifiers())) {

406							throw new IllegalStateException("@Resource annotation is not supported on static methods");

407						}

408						Class<?>[] paramTypes = method.getParameterTypes();

409						if (paramTypes.length != 1) {

410							throw new IllegalStateException("@Resource annotation requires a single-arg method: " + method);

411						}

412						if (!ignoredResourceTypes.contains(paramTypes[0].getName())) {

413							PropertyDescriptor pd = BeanUtils.findPropertyForMethod(bridgedMethod, clazz);

414							currElements.add(new ResourceElement(method, bridgedMethod, pd));

415						}

416					}

417				}

418			});

419

420			elements.addAll(0, currElements);

421			targetClass = targetClass.getSuperclass();

422		}

423		while (targetClass != null && targetClass != Object.class);

424

425		return new InjectionMetadata(clazz, elements);

426	}

428	/**

429	 * Obtain a lazily resolving resource proxy for the given name and type,

430	 * delegating to {@link #getResource} on demand once a method call comes in.

431	 * @param element the descriptor for the annotated field/method

432	 * @param requestingBeanName the name of the requesting bean

433	 * @return the resource object (never {@code null})

434	 * @since 4.2

435	 * @see #getResource

436	 * @see Lazy

437	 */

438	protected Object buildLazyResourceProxy(final LookupElement element, final @Nullable String requestingBeanName) {

439		TargetSource ts = new TargetSource() {

440			@Override

441			public Class<?> getTargetClass() {

442				return element.lookupType;

443			}

444			@Override

445			public boolean isStatic() {

446				return false;

447			}

448			@Override

449			public Object getTarget() {

450				return getResource(element, requestingBeanName);

451			}

452			@Override

453			public void releaseTarget(Object target) {

454			}

455		};

456		ProxyFactory pf = new ProxyFactory();

457		pf.setTargetSource(ts);

458		if (element.lookupType.isInterface()) {

459			pf.addInterface(element.lookupType);

460		}

461		ClassLoader classLoader = (this.beanFactory instanceof ConfigurableBeanFactory ?

462				((ConfigurableBeanFactory) this.beanFactory).getBeanClassLoader() : null);

463		return pf.getProxy(classLoader);

464	}

466	/**

467	 * Obtain the resource object for the given name and type.

468	 * @param element the descriptor for the annotated field/method

469	 * @param requestingBeanName the name of the requesting bean

470	 * @return the resource object (never {@code null})

471	 * @throws BeansException if we failed to obtain the target resource

472	 */

473	protected Object getResource(LookupElement element, @Nullable String requestingBeanName) throws BeansException {

474		if (StringUtils.hasLength(element.mappedName)) {

475			return this.jndiFactory.getBean(element.mappedName, element.lookupType);

476		}

477		if (this.alwaysUseJndiLookup) {

478			return this.jndiFactory.getBean(element.name, element.lookupType);

479		}

480		if (this.resourceFactory == null) {

481			throw new NoSuchBeanDefinitionException(element.lookupType,

482					"No resource factory configured - specify the 'resourceFactory' property");

483		}

484		return autowireResource(this.resourceFactory, element, requestingBeanName);

485	}

487	/**

488	 * Obtain a resource object for the given name and type through autowiring

489	 * based on the given factory.

490	 * @param factory the factory to autowire against

491	 * @param element the descriptor for the annotated field/method

492	 * @param requestingBeanName the name of the requesting bean

493	 * @return the resource object (never {@code null})

494	 * @throws BeansException if we failed to obtain the target resource

495	 */

496	protected Object autowireResource(BeanFactory factory, LookupElement element, @Nullable String requestingBeanName)

497			throws BeansException {

498

499		Object resource;

500		Set<String> autowiredBeanNames;

501		String name = element.name;

502

503		if (this.fallbackToDefaultTypeMatch && element.isDefaultName &&

504				factory instanceof AutowireCapableBeanFactory && !factory.containsBean(name)) {

505			autowiredBeanNames = new LinkedHashSet<>();

506			resource = ((AutowireCapableBeanFactory) factory).resolveDependency(

507					element.getDependencyDescriptor(), requestingBeanName, autowiredBeanNames, null);

508			if (resource == null) {

509				throw new NoSuchBeanDefinitionException(element.getLookupType(), "No resolvable resource object");

510			}

511		}

512		else {

513			resource = factory.getBean(name, element.lookupType);

514			autowiredBeanNames = Collections.singleton(name);

515		}

516

517		if (factory instanceof ConfigurableBeanFactory) {

518			ConfigurableBeanFactory beanFactory = (ConfigurableBeanFactory) factory;

519			for (String autowiredBeanName : autowiredBeanNames) {

520				if (requestingBeanName != null && beanFactory.containsBean(autowiredBeanName)) {

521					beanFactory.registerDependentBean(autowiredBeanName, requestingBeanName);

522				}

523			}

524		}

525

526		return resource;

527	}

531	 * Class representing generic injection information about an annotated field

532	 * or setter method, supporting @Resource and related annotations.

533	 */

534	protected abstract class LookupElement extends InjectionMetadata.InjectedElement {

535

536		protected String name = "";

537

538		protected boolean isDefaultName = false;

539

540		protected Class<?> lookupType = Object.class;

541

542		@Nullable

543		protected String mappedName;

544

545		public LookupElement(Member member, @Nullable PropertyDescriptor pd) {

546			super(member, pd);

547		}

548

549		/**

550		 * Return the resource name for the lookup.

551		 */

552		public final String getName() {

553			return this.name;

554		}

555

556		/**

557		 * Return the desired type for the lookup.

558		 */

559		public final Class<?> getLookupType() {

560			return this.lookupType;

561		}

562

563		/**

564		 * Build a DependencyDescriptor for the underlying field/method.

565		 */

566		public final DependencyDescriptor getDependencyDescriptor() {

567			if (this.isField) {

568				return new LookupDependencyDescriptor((Field) this.member, this.lookupType);

569			}

570			else {

571				return new LookupDependencyDescriptor((Method) this.member, this.lookupType);

572			}

573		}

574	}

577	/**

578	 * Class representing injection information about an annotated field

579	 * or setter method, supporting the @Resource annotation.

580	 */

581	private class ResourceElement extends LookupElement {

582

583		private final boolean lazyLookup;

584

585		public ResourceElement(Member member, AnnotatedElement ae, @Nullable PropertyDescriptor pd) {

586			super(member, pd);

587			Resource resource = ae.getAnnotation(Resource.class);

588			String resourceName = resource.name();

589			Class<?> resourceType = resource.type();

590			this.isDefaultName = !StringUtils.hasLength(resourceName);

591			if (this.isDefaultName) {

592				resourceName = this.member.getName();

593				if (this.member instanceof Method && resourceName.startsWith("set") && resourceName.length() > 3) {

594					resourceName = Introspector.decapitalize(resourceName.substring(3));

595				}

596			}

597			else if (embeddedValueResolver != null) {

598				resourceName = embeddedValueResolver.resolveStringValue(resourceName);

599			}

600			if (Object.class != resourceType) {

601				checkResourceType(resourceType);

602			}

603			else {

604				// No resource type specified... check field/method.

605				resourceType = getResourceType();

606			}

607			this.name = (resourceName != null ? resourceName : "");

608			this.lookupType = resourceType;

609			String lookupValue = resource.lookup();

610			this.mappedName = (StringUtils.hasLength(lookupValue) ? lookupValue : resource.mappedName());

611			Lazy lazy = ae.getAnnotation(Lazy.class);

612			this.lazyLookup = (lazy != null && lazy.value());

613		}

614

615		@Override

616		protected Object getResourceToInject(Object target, @Nullable String requestingBeanName) {

617			return (this.lazyLookup ? buildLazyResourceProxy(this, requestingBeanName) :

618					getResource(this, requestingBeanName));

619		}

620	}

623	/**

624	 * Class representing injection information about an annotated field

625	 * or setter method, supporting the @WebServiceRef annotation.

626	 */

627	private class WebServiceRefElement extends LookupElement {

628

629		private final Class<?> elementType;

630

631		private final String wsdlLocation;

632

633		public WebServiceRefElement(Member member, AnnotatedElement ae, @Nullable PropertyDescriptor pd) {

634			super(member, pd);

635			WebServiceRef resource = ae.getAnnotation(WebServiceRef.class);

636			String resourceName = resource.name();

637			Class<?> resourceType = resource.type();

638			this.isDefaultName = !StringUtils.hasLength(resourceName);

639			if (this.isDefaultName) {

640				resourceName = this.member.getName();

641				if (this.member instanceof Method && resourceName.startsWith("set") && resourceName.length() > 3) {

642					resourceName = Introspector.decapitalize(resourceName.substring(3));

643				}

644			}

645			if (Object.class != resourceType) {

646				checkResourceType(resourceType);

647			}

648			else {

649				// No resource type specified... check field/method.

650				resourceType = getResourceType();

651			}

652			this.name = resourceName;

653			this.elementType = resourceType;

654			if (Service.class.isAssignableFrom(resourceType)) {

655				this.lookupType = resourceType;

656			}

657			else {

658				this.lookupType = resource.value();

659			}

660			this.mappedName = resource.mappedName();

661			this.wsdlLocation = resource.wsdlLocation();

662		}

663

664		@Override

665		protected Object getResourceToInject(Object target, @Nullable String requestingBeanName) {

666			Service service;

667			try {

668				service = (Service) getResource(this, requestingBeanName);

669			}

670			catch (NoSuchBeanDefinitionException notFound) {

671				// Service to be created through generated class.

672				if (Service.class == this.lookupType) {

673					throw new IllegalStateException("No resource with name '" + this.name + "' found in context, " +

674							"and no specific JAX-WS Service subclass specified. The typical solution is to either specify " +

675							"a LocalJaxWsServiceFactoryBean with the given name or to specify the (generated) Service " +

676							"subclass as @WebServiceRef(...) value.");

677				}

678				if (StringUtils.hasLength(this.wsdlLocation)) {

679					try {

680						Constructor<?> ctor = this.lookupType.getConstructor(URL.class, QName.class);

681						WebServiceClient clientAnn = this.lookupType.getAnnotation(WebServiceClient.class);

682						if (clientAnn == null) {

683							throw new IllegalStateException("JAX-WS Service class [" + this.lookupType.getName() +

684									"] does not carry a WebServiceClient annotation");

685						}

686						service = (Service) BeanUtils.instantiateClass(ctor,

687								new URL(this.wsdlLocation), new QName(clientAnn.targetNamespace(), clientAnn.name()));

688					}

689					catch (NoSuchMethodException ex) {

690						throw new IllegalStateException("JAX-WS Service class [" + this.lookupType.getName() +

691								"] does not have a (URL, QName) constructor. Cannot apply specified WSDL location [" +

692								this.wsdlLocation + "].");

693					}

694					catch (MalformedURLException ex) {

695						throw new IllegalArgumentException(

696								"Specified WSDL location [" + this.wsdlLocation + "] isn't a valid URL");

697					}

698				}

699				else {

700					service = (Service) BeanUtils.instantiateClass(this.lookupType);

701				}

702			}

703			return service.getPort(this.elementType);

704		}

705	}

708	/**

709	 * Class representing injection information about an annotated field

710	 * or setter method, supporting the @EJB annotation.

711	 */

712	private class EjbRefElement extends LookupElement {

713

714		private final String beanName;

715

716		public EjbRefElement(Member member, AnnotatedElement ae, @Nullable PropertyDescriptor pd) {

717			super(member, pd);

718			EJB resource = ae.getAnnotation(EJB.class);

719			String resourceBeanName = resource.beanName();

720			String resourceName = resource.name();

721			this.isDefaultName = !StringUtils.hasLength(resourceName);

722			if (this.isDefaultName) {

723				resourceName = this.member.getName();

724				if (this.member instanceof Method && resourceName.startsWith("set") && resourceName.length() > 3) {

725					resourceName = Introspector.decapitalize(resourceName.substring(3));

726				}

727			}

728			Class<?> resourceType = resource.beanInterface();

729			if (Object.class != resourceType) {

730				checkResourceType(resourceType);

731			}

732			else {

733				// No resource type specified... check field/method.

734				resourceType = getResourceType();

735			}

736			this.beanName = resourceBeanName;

737			this.name = resourceName;

738			this.lookupType = resourceType;

739			this.mappedName = resource.mappedName();

740		}

741

742		@Override

743		protected Object getResourceToInject(Object target, @Nullable String requestingBeanName) {

744			if (StringUtils.hasLength(this.beanName)) {

745				if (beanFactory != null && beanFactory.containsBean(this.beanName)) {

746					// Local match found for explicitly specified local bean name.

747					Object bean = beanFactory.getBean(this.beanName, this.lookupType);

748					if (requestingBeanName != null && beanFactory instanceof ConfigurableBeanFactory) {

749						((ConfigurableBeanFactory) beanFactory).registerDependentBean(this.beanName, requestingBeanName);

750					}

751					return bean;

752				}

753				else if (this.isDefaultName && !StringUtils.hasLength(this.mappedName)) {

754					throw new NoSuchBeanDefinitionException(this.beanName,

755							"Cannot resolve 'beanName' in local BeanFactory. Consider specifying a general 'name' value instead.");

756				}

757			}

758			// JNDI name lookup - may still go to a local BeanFactory.

759			return getResource(this, requestingBeanName);

760		}

761	}
