242	/**

243	 * Compile this JdbcInsert using provided parameters and meta-data plus other settings.

244	 * This finalizes the configuration for this object and subsequent attempts to compile are

245	 * ignored. This will be implicitly called the first time an un-compiled insert is executed.

246	 * @throws InvalidDataAccessApiUsageException if the object hasn't been correctly initialized,

247	 * for example if no DataSource has been provided

248	 */

249	public final synchronized void compile() throws InvalidDataAccessApiUsageException {

250		if (!isCompiled()) {

251			if (getTableName() == null) {

252				throw new InvalidDataAccessApiUsageException("Table name is required");

253			}

254			try {

255				this.jdbcTemplate.afterPropertiesSet();

256			}

257			catch (IllegalArgumentException ex) {

258				throw new InvalidDataAccessApiUsageException(ex.getMessage());

259			}

260			compileInternal();

261			this.compiled = true;

262			if (logger.isDebugEnabled()) {

263				logger.debug("JdbcInsert for table [" + getTableName() + "] compiled");

264			}

265		}

266	}

267

268	/**

269	 * Delegate method to perform the actual compilation.

270	 * <p>Subclasses can override this template method to perform  their own compilation.

271	 * Invoked after this base class's compilation is complete.

272	 */

273	protected void compileInternal() {

274		DataSource dataSource = getJdbcTemplate().getDataSource();

275		Assert.state(dataSource != null, "No DataSource set");

276		this.tableMetaDataContext.processMetaData(dataSource, getColumnNames(), getGeneratedKeyNames());

277		this.insertString = this.tableMetaDataContext.createInsertString(getGeneratedKeyNames());

278		this.insertTypes = this.tableMetaDataContext.createInsertTypes();

279		if (logger.isDebugEnabled()) {

280			logger.debug("Compiled insert object: insert string is [" + this.insertString + "]");

281		}

282		onCompileInternal();

283	}

286	 * Hook method that subclasses may override to react to compilation.

287	 * <p>This implementation is empty.

288	 */

289	protected void onCompileInternal() {

290	}

300	/**

301	 * Check whether this operation has been compiled already;

302	 * lazily compile it if not already compiled.

303	 * <p>Automatically called by {@code validateParameters}.

304	 */

305	protected void checkCompiled() {

306		if (!isCompiled()) {

307			logger.debug("JdbcInsert not compiled before execution - invoking compile");

308			compile();

309		}

310	}

312	/**

313	 * Method to check whether we are allowed to make any configuration changes at this time.

314	 * If the class has been compiled, then no further changes to the configuration are allowed.

315	 */

316	protected void checkIfConfigurationModificationIsAllowed() {

317		if (isCompiled()) {

318			throw new InvalidDataAccessApiUsageException(

319					"Configuration cannot be altered once the class has been compiled or used");

320		}

321	}

408	/**

409	 * Delegate method to execute the insert, generating a single key.

410	 */

411	private Number executeInsertAndReturnKeyInternal(final List<?> values) {

412		KeyHolder kh = executeInsertAndReturnKeyHolderInternal(values);

413		if (kh.getKey() != null) {

414			return kh.getKey();

415		}

416		else {

417			throw new DataIntegrityViolationException(

418					"Unable to retrieve the generated key for the insert: " + getInsertString());

419		}

420	}

421

422	/**

423	 * Delegate method to execute the insert, generating any number of keys.

424	 */

425	private KeyHolder executeInsertAndReturnKeyHolderInternal(final List<?> values) {

426		if (logger.isDebugEnabled()) {

427			logger.debug("The following parameters are used for call " + getInsertString() + " with: " + values);

428		}

429		final KeyHolder keyHolder = new GeneratedKeyHolder();

430

431		if (this.tableMetaDataContext.isGetGeneratedKeysSupported()) {

432			getJdbcTemplate().update(

433					con -> {

434						PreparedStatement ps = prepareStatementForGeneratedKeys(con);

435						setParameterValues(ps, values, getInsertTypes());

436						return ps;

437					},

438					keyHolder);

439		}

440

441		else {

442			if (!this.tableMetaDataContext.isGetGeneratedKeysSimulated()) {

443				throw new InvalidDataAccessResourceUsageException(

444						"The getGeneratedKeys feature is not supported by this database");

445			}

446			if (getGeneratedKeyNames().length < 1) {

447				throw new InvalidDataAccessApiUsageException("Generated Key Name(s) not specified. " +

448						"Using the generated keys features requires specifying the name(s) of the generated column(s)");

449			}

450			if (getGeneratedKeyNames().length > 1) {

451				throw new InvalidDataAccessApiUsageException(

452						"Current database only supports retrieving the key for a single column. There are " +

453						getGeneratedKeyNames().length  + " columns specified: " + Arrays.asList(getGeneratedKeyNames()));

454			}

455

456			Assert.state(getTableName() != null, "No table name set");

457			final String keyQuery = this.tableMetaDataContext.getSimpleQueryForGetGeneratedKey(

458					getTableName(), getGeneratedKeyNames()[0]);

459			Assert.state(keyQuery != null, "Query for simulating get generated keys must not be null");

460

461			// This is a hack to be able to get the generated key from a database that doesn't support

462			// get generated keys feature. HSQL is one, PostgreSQL is another. Postgres uses a RETURNING

463			// clause while HSQL uses a second query that has to be executed with the same connection.

464

465			if (keyQuery.toUpperCase().startsWith("RETURNING")) {

466				Long key = getJdbcTemplate().queryForObject(

467						getInsertString() + " " + keyQuery, values.toArray(), Long.class);

468				Map<String, Object> keys = new HashMap<>(1);

469				keys.put(getGeneratedKeyNames()[0], key);

470				keyHolder.getKeyList().add(keys);

471			}

472			else {

473				getJdbcTemplate().execute((ConnectionCallback<Object>) con -> {

474					// Do the insert

475					PreparedStatement ps = null;

476					try {

477						ps = con.prepareStatement(getInsertString());

478						setParameterValues(ps, values, getInsertTypes());

479						ps.executeUpdate();

480					}

481					finally {

482						JdbcUtils.closeStatement(ps);

483					}

484					//Get the key

485					Statement keyStmt = null;

486					ResultSet rs = null;

487					Map<String, Object> keys = new HashMap<>(1);

488					try {

489						keyStmt = con.createStatement();

490						rs = keyStmt.executeQuery(keyQuery);

491						if (rs.next()) {

492							long key = rs.getLong(1);

493							keys.put(getGeneratedKeyNames()[0], key);

494							keyHolder.getKeyList().add(keys);

495						}

496					}

497					finally {

498						JdbcUtils.closeResultSet(rs);

499						JdbcUtils.closeStatement(keyStmt);

500					}

501					return null;

502				});

503			}

504		}

505

506		return keyHolder;

507	}

509	/**

510	 * Create a PreparedStatement to be used for an insert operation with generated keys.

511	 * @param con the Connection to use

512	 * @return the PreparedStatement

513	 */

514	private PreparedStatement prepareStatementForGeneratedKeys(Connection con) throws SQLException {

515		if (getGeneratedKeyNames().length < 1) {

516			throw new InvalidDataAccessApiUsageException("Generated Key Name(s) not specified. " +

517					"Using the generated keys features requires specifying the name(s) of the generated column(s).");

518		}

519		PreparedStatement ps;

520		if (this.tableMetaDataContext.isGeneratedKeysColumnNameArraySupported()) {

521			if (logger.isDebugEnabled()) {

522				logger.debug("Using generated keys support with array of column names.");

523			}

524			ps = con.prepareStatement(getInsertString(), getGeneratedKeyNames());

525		}

526		else {

527			if (logger.isDebugEnabled()) {

528				logger.debug("Using generated keys support with Statement.RETURN_GENERATED_KEYS.");

529			}

530			ps = con.prepareStatement(getInsertString(), Statement.RETURN_GENERATED_KEYS);

531		}

532		return ps;

533	}

535	/**

536	 * Delegate method that executes a batch insert using the passed-in Maps of parameters.

537	 * @param batch array of Maps with parameter names and values to be used in batch insert

538	 * @return array of number of rows affected

539	 */

540	@SuppressWarnings("unchecked")

541	protected int[] doExecuteBatch(Map<String, ?>... batch) {

542		checkCompiled();

543		List<List<Object>> batchValues = new ArrayList<>(batch.length);

544		for (Map<String, ?> args : batch) {

545			batchValues.add(matchInParameterValuesWithInsertColumns(args));

546		}

547		return executeBatchInternal(batchValues);

548	}

550	/**

551	 * Delegate method that executes a batch insert using the passed-in {@link SqlParameterSource}s.

552	 * @param batch array of SqlParameterSource with parameter names and values to be used in insert

553	 * @return array of number of rows affected

554	 */

555	protected int[] doExecuteBatch(SqlParameterSource... batch) {

556		checkCompiled();

557		List<List<Object>> batchValues = new ArrayList<>(batch.length);

558		for (SqlParameterSource parameterSource : batch) {

559			batchValues.add(matchInParameterValuesWithInsertColumns(parameterSource));

560		}

561		return executeBatchInternal(batchValues);

562	}

564	/**

565	 * Delegate method to execute the batch insert.

566	 */

567	private int[] executeBatchInternal(final List<List<Object>> batchValues) {

568		if (logger.isDebugEnabled()) {

569			logger.debug("Executing statement " + getInsertString() + " with batch of size: " + batchValues.size());

570		}

571		return getJdbcTemplate().batchUpdate(getInsertString(),

572				new BatchPreparedStatementSetter() {

573					@Override

574					public void setValues(PreparedStatement ps, int i) throws SQLException {

575						setParameterValues(ps, batchValues.get(i), getInsertTypes());

576					}

577					@Override

578					public int getBatchSize() {

579						return batchValues.size();

580					}

581				});

582	}

584	/**

585	 * Internal implementation for setting parameter values

586	 * @param preparedStatement the PreparedStatement

587	 * @param values the values to be set

588	 */

589	private void setParameterValues(PreparedStatement preparedStatement, List<?> values, @Nullable int... columnTypes)

590			throws SQLException {

591

592		int colIndex = 0;

593		for (Object value : values) {

594			colIndex++;

595			if (columnTypes == null || colIndex > columnTypes.length) {

596				StatementCreatorUtils.setParameterValue(preparedStatement, colIndex, SqlTypeValue.TYPE_UNKNOWN, value);

597			}

598			else {

599				StatementCreatorUtils.setParameterValue(preparedStatement, colIndex, columnTypes[colIndex - 1], value);

600			}

601		}

602	}
