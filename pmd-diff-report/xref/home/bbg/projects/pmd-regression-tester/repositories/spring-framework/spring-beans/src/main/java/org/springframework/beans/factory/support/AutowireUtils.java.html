52	/**

53	 * Sort the given constructors, preferring public constructors and "greedy" ones with

54	 * a maximum number of arguments. The result will contain public constructors first,

55	 * with decreasing number of arguments, then non-public constructors, again with

56	 * decreasing number of arguments.

57	 * @param constructors the constructor array to sort

58	 */

59	public static void sortConstructors(Constructor<?>[] constructors) {

60		Arrays.sort(constructors, (c1, c2) -> {

61            boolean p1 = Modifier.isPublic(c1.getModifiers());

62            boolean p2 = Modifier.isPublic(c2.getModifiers());

63            if (p1 != p2) {

64                return (p1 ? -1 : 1);

65            }

66            int c1pl = c1.getParameterCount();

67            int c2pl = c2.getParameterCount();

68            return (c1pl < c2pl ? 1 : (c1pl > c2pl ? -1 : 0));

69        });

70	}

71

72	/**

73	 * Sort the given factory methods, preferring public methods and "greedy" ones

74	 * with a maximum of arguments. The result will contain public methods first,

75	 * with decreasing number of arguments, then non-public methods, again with

76	 * decreasing number of arguments.

77	 * @param factoryMethods the factory method array to sort

78	 */

79	public static void sortFactoryMethods(Method[] factoryMethods) {

80		Arrays.sort(factoryMethods, (fm1, fm2) -> {

81            boolean p1 = Modifier.isPublic(fm1.getModifiers());

82            boolean p2 = Modifier.isPublic(fm2.getModifiers());

83            if (p1 != p2) {

84                return (p1 ? -1 : 1);

85            }

86            int c1pl = fm1.getParameterCount();

87            int c2pl = fm2.getParameterCount();

88            return (c1pl < c2pl ? 1 : (c1pl > c2pl ? -1 : 0));

89        });

90	}

92	/**

93	 * Determine whether the given bean property is excluded from dependency checks.

94	 * <p>This implementation excludes properties defined by CGLIB.

95	 * @param pd the PropertyDescriptor of the bean property

96	 * @return whether the bean property is excluded

97	 */

98	public static boolean isExcludedFromDependencyCheck(PropertyDescriptor pd) {

99		Method wm = pd.getWriteMethod();

100		if (wm == null) {

101			return false;

102		}

103		if (!wm.getDeclaringClass().getName().contains("$$")) {

104			// Not a CGLIB method so it's OK.

105			return false;

106		}

107		// It was declared by CGLIB, but we might still want to autowire it

108		// if it was actually declared by the superclass.

109		Class<?> superclass = wm.getDeclaringClass().getSuperclass();

110		return !ClassUtils.hasMethod(superclass, wm.getName(), wm.getParameterTypes());

111	}

112

113	/**

114	 * Return whether the setter method of the given bean property is defined

115	 * in any of the given interfaces.

116	 * @param pd the PropertyDescriptor of the bean property

117	 * @param interfaces the Set of interfaces (Class objects)

118	 * @return whether the setter method is defined by an interface

119	 */

120	public static boolean isSetterDefinedInInterface(PropertyDescriptor pd, Set<Class<?>> interfaces) {

121		Method setter = pd.getWriteMethod();

122		if (setter != null) {

123			Class<?> targetClass = setter.getDeclaringClass();

124			for (Class<?> ifc : interfaces) {

125				if (ifc.isAssignableFrom(targetClass) &&

126						ClassUtils.hasMethod(ifc, setter.getName(), setter.getParameterTypes())) {

127					return true;

128				}

129			}

130		}

131		return false;

132	}

134	/**

135	 * Resolve the given autowiring value against the given required type,

136	 * e.g. an {@link ObjectFactory} value to its actual object result.

137	 * @param autowiringValue the value to resolve

138	 * @param requiredType the type to assign the result to

139	 * @return the resolved value

140	 */

141	public static Object resolveAutowiringValue(Object autowiringValue, Class<?> requiredType) {

142		if (autowiringValue instanceof ObjectFactory && !requiredType.isInstance(autowiringValue)) {

143			ObjectFactory<?> factory = (ObjectFactory<?>) autowiringValue;

144			if (autowiringValue instanceof Serializable && requiredType.isInterface()) {

145				autowiringValue = Proxy.newProxyInstance(requiredType.getClassLoader(),

146						new Class<?>[] {requiredType}, new ObjectFactoryDelegatingInvocationHandler(factory));

147			}

148			else {

149				return factory.getObject();

150			}

151		}

152		return autowiringValue;

153	}

155	/**

156	 * Determine the target type for the generic return type of the given

157	 * <em>generic factory method</em>, where formal type variables are declared

158	 * on the given method itself.

159	 * <p>For example, given a factory method with the following signature, if

160	 * {@code resolveReturnTypeForFactoryMethod()} is invoked with the reflected

161	 * method for {@code creatProxy()} and an {@code Object[]} array containing

162	 * {@code MyService.class}, {@code resolveReturnTypeForFactoryMethod()} will

163	 * infer that the target return type is {@code MyService}.

164	 * <pre class="code">{@code public static <T> T createProxy(Class<T> clazz)}</pre>

165	 * <h4>Possible Return Values</h4>

166	 * <ul>

167	 * <li>the target return type, if it can be inferred</li>

168	 * <li>the {@linkplain Method#getReturnType() standard return type}, if

169	 * the given {@code method} does not declare any {@linkplain

170	 * Method#getTypeParameters() formal type variables}</li>

171	 * <li>the {@linkplain Method#getReturnType() standard return type}, if the

172	 * target return type cannot be inferred (e.g., due to type erasure)</li>

173	 * <li>{@code null}, if the length of the given arguments array is shorter

174	 * than the length of the {@linkplain

175	 * Method#getGenericParameterTypes() formal argument list} for the given

176	 * method</li>

177	 * </ul>

178	 * @param method the method to introspect (never {@code null})

179	 * @param args the arguments that will be supplied to the method when it is

180	 * invoked (never {@code null})

181	 * @param classLoader the ClassLoader to resolve class names against,

182	 * if necessary (never {@code null})

183	 * @return the resolved target return type or the standard method return type

184	 * @since 3.2.5

185	 */

186	public static Class<?> resolveReturnTypeForFactoryMethod(

187			Method method, Object[] args, @Nullable ClassLoader classLoader) {

188

189		Assert.notNull(method, "Method must not be null");

190		Assert.notNull(args, "Argument array must not be null");

191

192		TypeVariable<Method>[] declaredTypeVariables = method.getTypeParameters();

193		Type genericReturnType = method.getGenericReturnType();

194		Type[] methodParameterTypes = method.getGenericParameterTypes();

195		Assert.isTrue(args.length == methodParameterTypes.length, "Argument array does not match parameter count");

196

197		// Ensure that the type variable (e.g., T) is declared directly on the method

198		// itself (e.g., via <T>), not on the enclosing class or interface.

199		boolean locallyDeclaredTypeVariableMatchesReturnType = false;

200		for (TypeVariable<Method> currentTypeVariable : declaredTypeVariables) {

201			if (currentTypeVariable.equals(genericReturnType)) {

202				locallyDeclaredTypeVariableMatchesReturnType = true;

203				break;

204			}

205		}

206

207		if (locallyDeclaredTypeVariableMatchesReturnType) {

208			for (int i = 0; i < methodParameterTypes.length; i++) {

209				Type methodParameterType = methodParameterTypes[i];

210				Object arg = args[i];

211				if (methodParameterType.equals(genericReturnType)) {

212					if (arg instanceof TypedStringValue) {

213						TypedStringValue typedValue = ((TypedStringValue) arg);

214						if (typedValue.hasTargetType()) {

215							return typedValue.getTargetType();

216						}

217						try {

218							Class<?> resolvedType = typedValue.resolveTargetType(classLoader);

219							if (resolvedType != null) {

220								return resolvedType;

221							}

222						}

223						catch (ClassNotFoundException ex) {

224							throw new IllegalStateException("Failed to resolve value type [" +

225									typedValue.getTargetTypeName() + "] for factory method argument", ex);

226						}

227					}

228					else if (arg != null && !(arg instanceof BeanMetadataElement)) {

229						// Only consider argument type if it is a simple value...

230						return arg.getClass();

231					}

232					return method.getReturnType();

233				}

234				else if (methodParameterType instanceof ParameterizedType) {

235					ParameterizedType parameterizedType = (ParameterizedType) methodParameterType;

236					Type[] actualTypeArguments = parameterizedType.getActualTypeArguments();

237					for (Type typeArg : actualTypeArguments) {

238						if (typeArg.equals(genericReturnType)) {

239							if (arg instanceof Class) {

240								return (Class<?>) arg;

241							}

242							else {

243								String className = null;

244								if (arg instanceof String) {

245									className = (String) arg;

246								}

247								else if (arg instanceof TypedStringValue) {

248									TypedStringValue typedValue = ((TypedStringValue) arg);

249									String targetTypeName = typedValue.getTargetTypeName();

250									if (targetTypeName == null || Class.class.getName().equals(targetTypeName)) {

251										className = typedValue.getValue();

252									}

253								}

254								if (className != null) {

255									try {

256										return ClassUtils.forName(className, classLoader);

257									}

258									catch (ClassNotFoundException ex) {

259										throw new IllegalStateException("Could not resolve class name [" + arg +

260												"] for factory method argument", ex);

261									}

262								}

263								// Consider adding logic to determine the class of the typeArg, if possible.

264								// For now, just fall back...

265								return method.getReturnType();

266							}

267						}

268					}

269				}

270			}

271		}

272

273		// Fall back...

274		return method.getReturnType();

275	}

278	/**

279	 * Reflective InvocationHandler for lazy access to the current target object.

280	 */

281	@SuppressWarnings("serial")

282	private static class ObjectFactoryDelegatingInvocationHandler implements InvocationHandler, Serializable {

283

284		private final ObjectFactory<?> objectFactory;

285

286		public ObjectFactoryDelegatingInvocationHandler(ObjectFactory<?> objectFactory) {

287			this.objectFactory = objectFactory;

288		}

289

290		@Override

291		public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {

292			String methodName = method.getName();

293			if (methodName.equals("equals")) {

294				// Only consider equal when proxies are identical.

295				return (proxy == args[0]);

296			}

297			else if (methodName.equals("hashCode")) {

298				// Use hashCode of proxy.

299				return System.identityHashCode(proxy);

300			}

301			else if (methodName.equals("toString")) {

302				return this.objectFactory.toString();

303			}

304			try {

305				return method.invoke(this.objectFactory.getObject(), args);

306			}

307			catch (InvocationTargetException ex) {

308				throw ex.getTargetException();

309			}

310		}

311	}
