382	private void updateContentTypeHeader() {

383		if (StringUtils.hasLength(this.contentType)) {

384			StringBuilder sb = new StringBuilder(this.contentType);

385			if (!this.contentType.toLowerCase().contains(CHARSET_PREFIX) &&

386					StringUtils.hasLength(this.characterEncoding)) {

387				sb.append(";").append(CHARSET_PREFIX).append(this.characterEncoding);

388			}

389			doAddHeaderValue(HttpHeaders.CONTENT_TYPE, sb.toString(), true);

390		}

391	}

436		return new String(this.content, this.characterEncoding);

449	public void setContentType(String contentType) {

450		this.contentType = contentType;

451		if (contentType != null) {

452			try {

453				MediaType mediaType = MediaType.parseMediaType(contentType);

454				if (mediaType.getCharset() != null) {

455					this.characterEncoding = mediaType.getCharset().name();

456				}

457			}

458			catch (IllegalArgumentException ex) {

459				// Try to get charset value anyway

460				int charsetIndex = contentType.toLowerCase().indexOf(CHARSET_PREFIX);

461				if (charsetIndex != -1) {

462					this.characterEncoding = contentType.substring(charsetIndex + CHARSET_PREFIX.length());

463				}

464			}

465			updateContentTypeHeader();

466		}

467	}

503	/**

504	 * Set all provided parameters <strong>replacing</strong> any existing

505	 * values for the provided parameter names. To add without replacing

506	 * existing values, use {@link #addParameters(java.util.Map)}.

507	 */

508	public void setParameters(Map<String, ?> params) {

509		Assert.notNull(params, "Parameter map must not be null");

510		for (String key : params.keySet()) {

511			Object value = params.get(key);

512			if (value instanceof String) {

513				setParameter(key, (String) value);

514			}

515			else if (value instanceof String[]) {

516				setParameter(key, (String[]) value);

517			}

518			else {

519				throw new IllegalArgumentException(

520						"Parameter map value must be single value " + " or array of type [" + String.class.getName() + "]");

521			}

522		}

523	}

526	 * Add a single value for the specified HTTP parameter.

527	 * <p>If there are already one or more values registered for the given

528	 * parameter name, the given value will be added to the end of the list.

529	 */

530	public void addParameter(String name, String value) {

531		addParameter(name, new String[] {value});

532	}

534	/**

535	 * Add an array of values for the specified HTTP parameter.

536	 * <p>If there are already one or more values registered for the given

537	 * parameter name, the given values will be added to the end of the list.

538	 */

539	public void addParameter(String name, String... values) {

540		Assert.notNull(name, "Parameter name must not be null");

541		String[] oldArr = this.parameters.get(name);

542		if (oldArr != null) {

543			String[] newArr = new String[oldArr.length + values.length];

544			System.arraycopy(oldArr, 0, newArr, 0, oldArr.length);

545			System.arraycopy(values, 0, newArr, oldArr.length, values.length);

546			this.parameters.put(name, newArr);

547		}

548		else {

549			this.parameters.put(name, values);

550		}

551	}

635	@Override

636	public String getServerName() {

637		String host = getHeader(HttpHeaders.HOST);

638		if (host != null) {

639			host = host.trim();

640			if (host.startsWith("[")) {

641				host = host.substring(1, host.indexOf(']'));

642			}

643			else if (host.contains(":")) {

644				host = host.substring(0, host.indexOf(':'));

645			}

646			return host;

647		}

648

649		// else

650		return this.serverName;

651	}

657	@Override

658	public int getServerPort() {

659		String host = getHeader(HttpHeaders.HOST);

660		if (host != null) {

661			host = host.trim();

662			int idx;

663			if (host.startsWith("[")) {

664				idx = host.indexOf(':', host.indexOf(']'));

665			}

666			else {

667				idx = host.indexOf(':');

668			}

669			if (idx != -1) {

670				return Integer.parseInt(host.substring(idx + 1));

671			}

672		}

673

674		// else

675		return this.serverPort;

676	}

759	private void updateAcceptLanguageHeader() {

760		HttpHeaders headers = new HttpHeaders();

761		headers.setAcceptLanguageAsLocales(this.locales);

762		doAddHeaderValue(HttpHeaders.ACCEPT_LANGUAGE, headers.getFirst(HttpHeaders.ACCEPT_LANGUAGE), true);

947	/**

948	 * Add a header entry for the given name.

949	 * <p>While this method can take any {@code Object} as a parameter, it

950	 * is recommended to use the following types:

951	 * <ul>

952	 * <li>String or any Object to be converted using {@code toString()}; see {@link #getHeader}.</li>

953	 * <li>String, Number, or Date for date headers; see {@link #getDateHeader}.</li>

954	 * <li>String or Number for integer headers; see {@link #getIntHeader}.</li>

955	 * <li>{@code String[]} or {@code Collection<String>} for multiple values; see {@link #getHeaders}.</li>

956	 * </ul>

957	 * @see #getHeaderNames

958	 * @see #getHeaders

959	 * @see #getHeader

960	 * @see #getDateHeader

961	 */

962	public void addHeader(String name, Object value) {

963		if (HttpHeaders.CONTENT_TYPE.equalsIgnoreCase(name) &&

964				!this.headers.containsKey(HttpHeaders.CONTENT_TYPE)) {

965			setContentType(value.toString());

966		}

967		else if (HttpHeaders.ACCEPT_LANGUAGE.equalsIgnoreCase(name) &&

968				!this.headers.containsKey(HttpHeaders.ACCEPT_LANGUAGE)) {

969			try {

970				HttpHeaders headers = new HttpHeaders();

971				headers.add(HttpHeaders.ACCEPT_LANGUAGE, value.toString());

972				setPreferredLocales(headers.getAcceptLanguageAsLocales());

973			}

974			catch (IllegalArgumentException ex) {

975				// Invalid Accept-Language format -> store plain header instead

976				doAddHeaderValue(name, value, true);

977			}

978		}

979		else {

980			doAddHeaderValue(name, value, false);

981		}

982	}

983

984	private void doAddHeaderValue(String name, Object value, boolean replace) {

985		HeaderValueHolder header = HeaderValueHolder.getByName(this.headers, name);

986		Assert.notNull(value, "Header value must not be null");

987		if (header == null || replace) {

988			header = new HeaderValueHolder();

989			this.headers.put(name, header);

990		}

991		if (value instanceof Collection) {

992			header.addValues((Collection<?>) value);

993		}

994		else if (value.getClass().isArray()) {

995			header.addValueArray(value);

996		}

997		else {

998			header.addValue(value);

999		}

1000	}

1002	/**

1003	 * Return the long timestamp for the date header with the given {@code name}.

1004	 * <p>If the internal value representation is a String, this method will try

1005	 * to parse it as a date using the supported date formats:

1006	 * <ul>

1007	 * <li>"EEE, dd MMM yyyy HH:mm:ss zzz"</li>

1008	 * <li>"EEE, dd-MMM-yy HH:mm:ss zzz"</li>

1009	 * <li>"EEE MMM dd HH:mm:ss yyyy"</li>

1010	 * </ul>

1011	 * @param name the header name

1012	 * @see <a href="https://tools.ietf.org/html/rfc7231#section-7.1.1.1">Section 7.1.1.1 of RFC 7231</a>

1013	 */

1014	@Override

1015	public long getDateHeader(String name) {

1016		HeaderValueHolder header = HeaderValueHolder.getByName(this.headers, name);

1017		Object value = (header != null ? header.getValue() : null);

1018		if (value instanceof Date) {

1019			return ((Date) value).getTime();

1020		}

1021		else if (value instanceof Number) {

1022			return ((Number) value).longValue();

1023		}

1024		else if (value instanceof String) {

1025			return parseDateHeader(name, (String) value);

1026		}

1027		else if (value != null) {

1028			throw new IllegalArgumentException(

1029					"Value for header '" + name + "' is not a Date, Number, or String: " + value);

1030		}

1031		else {

1032			return -1L;

1033		}

1034	}

1035

1036	private long parseDateHeader(String name, String value) {

1037		for (String dateFormat : DATE_FORMATS) {

1038			SimpleDateFormat simpleDateFormat = new SimpleDateFormat(dateFormat, Locale.US);

1039			simpleDateFormat.setTimeZone(GMT);

1040			try {

1041				return simpleDateFormat.parse(value).getTime();

1042			}

1043			catch (ParseException ex) {

1044				// ignore

1045			}

1046		}

1047		throw new IllegalArgumentException("Cannot parse date value '" + value + "' for '" + name + "' header");

1048	}

1172	@Override

1173	public StringBuffer getRequestURL() {

1174		String scheme = getScheme();

1175		String server = getServerName();

1176		int port = getServerPort();

1177		String uri = getRequestURI();

1178

1179		StringBuffer url = new StringBuffer(scheme).append("://").append(server);

1180		if (port > 0 && ((HTTP.equalsIgnoreCase(scheme) && port != 80) ||

1181				(HTTPS.equalsIgnoreCase(scheme) && port != 443))) {

1182			url.append(':').append(port);

1183		}

1184		if (StringUtils.hasText(uri)) {

1185			url.append(uri);

1186		}

1187		return url;

1188	}
