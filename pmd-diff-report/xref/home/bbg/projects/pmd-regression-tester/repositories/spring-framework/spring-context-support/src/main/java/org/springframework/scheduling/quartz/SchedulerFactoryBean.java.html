51/**

52 * {@link FactoryBean} that creates and configures a Quartz {@link org.quartz.Scheduler},

53 * manages its lifecycle as part of the Spring application context, and exposes the

54 * Scheduler as bean reference for dependency injection.

55 *

56 * <p>Allows registration of JobDetails, Calendars and Triggers, automatically

57 * starting the scheduler on initialization and shutting it down on destruction.

58 * In scenarios that just require static registration of jobs at startup, there

59 * is no need to access the Scheduler instance itself in application code.

60 *

61 * <p>For dynamic registration of jobs at runtime, use a bean reference to

62 * this SchedulerFactoryBean to get direct access to the Quartz Scheduler

63 * ({@code org.quartz.Scheduler}). This allows you to create new jobs

64 * and triggers, and also to control and monitor the entire Scheduler.

65 *

66 * <p>Note that Quartz instantiates a new Job for each execution, in

67 * contrast to Timer which uses a TimerTask instance that is shared

68 * between repeated executions. Just JobDetail descriptors are shared.

69 *

70 * <p>When using persistent jobs, it is strongly recommended to perform all

71 * operations on the Scheduler within Spring-managed (or plain JTA) transactions.

72 * Else, database locking will not properly work and might even break.

73 * (See {@link #setDataSource setDataSource} javadoc for details.)

74 *

75 * <p>The preferred way to achieve transactional execution is to demarcate

76 * declarative transactions at the business facade level, which will

77 * automatically apply to Scheduler operations performed within those scopes.

78 * Alternatively, you may add transactional advice for the Scheduler itself.

79 *

80 * <p>Compatible with Quartz 2.1.4 and higher, as of Spring 4.1.

81 *

82 * @author Juergen Hoeller

83 * @since 18.02.2004

84 * @see #setDataSource

85 * @see org.quartz.Scheduler

86 * @see org.quartz.SchedulerFactory

87 * @see org.quartz.impl.StdSchedulerFactory

88 * @see org.springframework.transaction.interceptor.TransactionProxyFactoryBean

89 */

90public class SchedulerFactoryBean extends SchedulerAccessor implements FactoryBean<Scheduler>,

91		BeanNameAware, ApplicationContextAware, InitializingBean, DisposableBean, SmartLifecycle {

479	@Override

480	public void afterPropertiesSet() throws Exception {

481		if (this.dataSource == null && this.nonTransactionalDataSource != null) {

482			this.dataSource = this.nonTransactionalDataSource;

483		}

484

485		if (this.applicationContext != null && this.resourceLoader == null) {

486			this.resourceLoader = this.applicationContext;

487		}

488

489		// Initialize the SchedulerFactory instance...

490		SchedulerFactory schedulerFactory = prepareSchedulerFactory();

491

492		if (this.resourceLoader != null) {

493			// Make given ResourceLoader available for SchedulerFactory configuration.

494			configTimeResourceLoaderHolder.set(this.resourceLoader);

495		}

496		if (this.taskExecutor != null) {

497			// Make given TaskExecutor available for SchedulerFactory configuration.

498			configTimeTaskExecutorHolder.set(this.taskExecutor);

499		}

500		if (this.dataSource != null) {

501			// Make given DataSource available for SchedulerFactory configuration.

502			configTimeDataSourceHolder.set(this.dataSource);

503		}

504		if (this.nonTransactionalDataSource != null) {

505			// Make given non-transactional DataSource available for SchedulerFactory configuration.

506			configTimeNonTransactionalDataSourceHolder.set(this.nonTransactionalDataSource);

507		}

508

509		// Get Scheduler instance from SchedulerFactory.

510		try {

511			this.scheduler = createScheduler(schedulerFactory, this.schedulerName);

512			populateSchedulerContext();

513

514			if (!this.jobFactorySet && !(this.scheduler instanceof RemoteScheduler)) {

515				// Use AdaptableJobFactory as default for a local Scheduler, unless when

516				// explicitly given a null value through the "jobFactory" bean property.

517				this.jobFactory = new AdaptableJobFactory();

518			}

519			if (this.jobFactory != null) {

520				if (this.jobFactory instanceof SchedulerContextAware) {

521					((SchedulerContextAware) this.jobFactory).setSchedulerContext(this.scheduler.getContext());

522				}

523				this.scheduler.setJobFactory(this.jobFactory);

524			}

525		}

526

527		finally {

528			if (this.resourceLoader != null) {

529				configTimeResourceLoaderHolder.remove();

530			}

531			if (this.taskExecutor != null) {

532				configTimeTaskExecutorHolder.remove();

533			}

534			if (this.dataSource != null) {

535				configTimeDataSourceHolder.remove();

536			}

537			if (this.nonTransactionalDataSource != null) {

538				configTimeNonTransactionalDataSourceHolder.remove();

539			}

540		}

541

542		registerListeners();

543		registerJobsAndTriggers();

544	}

547	/**

548	 * Create a SchedulerFactory if necessary and apply locally defined Quartz properties to it.

549	 * @return the initialized SchedulerFactory

550	 */

551	private SchedulerFactory prepareSchedulerFactory() throws SchedulerException, IOException {

552		SchedulerFactory schedulerFactory = this.schedulerFactory;

553		if (schedulerFactory == null) {

554			// Create local SchedulerFactory instance (typically a StdSchedulerFactory)

555			schedulerFactory = BeanUtils.instantiateClass(this.schedulerFactoryClass);

556			if (schedulerFactory instanceof StdSchedulerFactory) {

557				initSchedulerFactory((StdSchedulerFactory) schedulerFactory);

558			}

559			else if (this.configLocation != null || this.quartzProperties != null ||

560					this.taskExecutor != null || this.dataSource != null) {

561				throw new IllegalArgumentException(

562						"StdSchedulerFactory required for applying Quartz properties: " + schedulerFactory);

563			}

564			// Otherwise, no local settings to be applied via StdSchedulerFactory.initialize(Properties)

565		}

566		// Otherwise, assume that externally provided factory has been initialized with appropriate settings

567		return schedulerFactory;

568	}

570	/**

571	 * Initialize the given SchedulerFactory, applying locally defined Quartz properties to it.

572	 * @param schedulerFactory the SchedulerFactory to initialize

573	 */

574	private void initSchedulerFactory(StdSchedulerFactory schedulerFactory) throws SchedulerException, IOException {

575		Properties mergedProps = new Properties();

576		if (this.resourceLoader != null) {

577			mergedProps.setProperty(StdSchedulerFactory.PROP_SCHED_CLASS_LOAD_HELPER_CLASS,

578					ResourceLoaderClassLoadHelper.class.getName());

579		}

580

581		if (this.taskExecutor != null) {

582			mergedProps.setProperty(StdSchedulerFactory.PROP_THREAD_POOL_CLASS,

583					LocalTaskExecutorThreadPool.class.getName());

584		}

585		else {

586			// Set necessary default properties here, as Quartz will not apply

587			// its default configuration when explicitly given properties.

588			mergedProps.setProperty(StdSchedulerFactory.PROP_THREAD_POOL_CLASS, SimpleThreadPool.class.getName());

589			mergedProps.setProperty(PROP_THREAD_COUNT, Integer.toString(DEFAULT_THREAD_COUNT));

590		}

591

592		if (this.configLocation != null) {

593			if (logger.isInfoEnabled()) {

594				logger.info("Loading Quartz config from [" + this.configLocation + "]");

595			}

596			PropertiesLoaderUtils.fillProperties(mergedProps, this.configLocation);

597		}

598

599		CollectionUtils.mergePropertiesIntoMap(this.quartzProperties, mergedProps);

600		if (this.dataSource != null) {

601			mergedProps.put(StdSchedulerFactory.PROP_JOB_STORE_CLASS, LocalDataSourceJobStore.class.getName());

602		}

603		if (this.schedulerName != null) {

604			mergedProps.put(StdSchedulerFactory.PROP_SCHED_INSTANCE_NAME, this.schedulerName);

605		}

606

607		schedulerFactory.initialize(mergedProps);

608	}

610	/**

611	 * Create the Scheduler instance for the given factory and scheduler name.

612	 * Called by {@link #afterPropertiesSet}.

613	 * <p>The default implementation invokes SchedulerFactory's {@code getScheduler}

614	 * method. Can be overridden for custom Scheduler creation.

615	 * @param schedulerFactory the factory to create the Scheduler with

616	 * @param schedulerName the name of the scheduler to create

617	 * @return the Scheduler instance

618	 * @throws SchedulerException if thrown by Quartz methods

619	 * @see #afterPropertiesSet

620	 * @see org.quartz.SchedulerFactory#getScheduler

621	 */

622	protected Scheduler createScheduler(SchedulerFactory schedulerFactory, @Nullable String schedulerName)

623			throws SchedulerException {

624

625		// Override thread context ClassLoader to work around naive Quartz ClassLoadHelper loading.

626		Thread currentThread = Thread.currentThread();

627		ClassLoader threadContextClassLoader = currentThread.getContextClassLoader();

628		boolean overrideClassLoader = (this.resourceLoader != null &&

629				this.resourceLoader.getClassLoader() != threadContextClassLoader);

630		if (overrideClassLoader) {

631			currentThread.setContextClassLoader(this.resourceLoader.getClassLoader());

632		}

633		try {

634			SchedulerRepository repository = SchedulerRepository.getInstance();

635			synchronized (repository) {

636				Scheduler existingScheduler = (schedulerName != null ? repository.lookup(schedulerName) : null);

637				Scheduler newScheduler = schedulerFactory.getScheduler();

638				if (newScheduler == existingScheduler) {

639					throw new IllegalStateException("Active Scheduler of name '" + schedulerName + "' already registered " +

640							"in Quartz SchedulerRepository. Cannot create a new Spring-managed Scheduler of the same name!");

641				}

642				if (!this.exposeSchedulerInRepository) {

643					// Need to remove it in this case, since Quartz shares the Scheduler instance by default!

644					SchedulerRepository.getInstance().remove(newScheduler.getSchedulerName());

645				}

646				return newScheduler;

647			}

648		}

649		finally {

650			if (overrideClassLoader) {

651				// Reset original thread context ClassLoader.

652				currentThread.setContextClassLoader(threadContextClassLoader);

653			}

654		}

655	}

656

657	/**

658	 * Expose the specified context attributes and/or the current

659	 * ApplicationContext in the Quartz SchedulerContext.

660	 */

661	private void populateSchedulerContext() throws SchedulerException {

662		// Put specified objects into Scheduler context.

663		if (this.schedulerContextMap != null) {

664			getScheduler().getContext().putAll(this.schedulerContextMap);

665		}

666

667		// Register ApplicationContext in Scheduler context.

668		if (this.applicationContextSchedulerContextKey != null) {

669			if (this.applicationContext == null) {

670				throw new IllegalStateException(

671					"SchedulerFactoryBean needs to be set up in an ApplicationContext " +

672					"to be able to handle an 'applicationContextSchedulerContextKey'");

673			}

674			getScheduler().getContext().put(this.applicationContextSchedulerContextKey, this.applicationContext);

675		}

676	}

679	/**

680	 * Start the Quartz Scheduler, respecting the "startupDelay" setting.

681	 * @param scheduler the Scheduler to start

682	 * @param startupDelay the number of seconds to wait before starting

683	 * the Scheduler asynchronously

684	 */

685	protected void startScheduler(final Scheduler scheduler, final int startupDelay) throws SchedulerException {

686		if (startupDelay <= 0) {

687			logger.info("Starting Quartz Scheduler now");

688			scheduler.start();

689		}

690		else {

691			if (logger.isInfoEnabled()) {

692				logger.info("Will start Quartz Scheduler [" + scheduler.getSchedulerName() +

693						"] in " + startupDelay + " seconds");

694			}

695			// Not using the Quartz startDelayed method since we explicitly want a daemon

696			// thread here, not keeping the JVM alive in case of all other threads ending.

697			Thread schedulerThread = new Thread() {

698				@Override

699				public void run() {

700					try {

701						Thread.sleep(TimeUnit.SECONDS.toMillis(startupDelay));

702					}

703					catch (InterruptedException ex) {

704						Thread.currentThread().interrupt();

705						// simply proceed

706					}

707					if (logger.isInfoEnabled()) {

708						logger.info("Starting Quartz Scheduler now, after delay of " + startupDelay + " seconds");

709					}

710					try {

711						scheduler.start();

712					}

713					catch (SchedulerException ex) {

714						throw new SchedulingException("Could not start Quartz Scheduler after delay", ex);

715					}

716				}

717			};

718			schedulerThread.setName("Quartz Scheduler [" + scheduler.getSchedulerName() + "]");

719			schedulerThread.setDaemon(true);

720			schedulerThread.start();

721		}

722	}

756	@Override

757	public void start() throws SchedulingException {

758		if (this.scheduler != null) {

759			try {

760				startScheduler(this.scheduler, this.startupDelay);

761			}

762			catch (SchedulerException ex) {

763				throw new SchedulingException("Could not start Quartz Scheduler", ex);

764			}

765		}

766	}

767

768	@Override

769	public void stop() throws SchedulingException {

770		if (this.scheduler != null) {

771			try {

772				this.scheduler.standby();

773			}

774			catch (SchedulerException ex) {

775				throw new SchedulingException("Could not stop Quartz Scheduler", ex);

776			}

777		}

778	}

779

780	@Override

781	public void stop(Runnable callback) throws SchedulingException {

782		stop();

783		callback.run();

784	}

785

786	@Override

787	public boolean isRunning() throws SchedulingException {

788		if (this.scheduler != null) {

789			try {

790				return !this.scheduler.isInStandbyMode();

791			}

792			catch (SchedulerException ex) {

793				return false;

794			}

795		}

796		return false;

797	}
