56/**

57 * <b>This is the central class in the JDBC core package.</b>

58 * It simplifies the use of JDBC and helps to avoid common errors.

59 * It executes core JDBC workflow, leaving application code to provide SQL

60 * and extract results. This class executes SQL queries or updates, initiating

61 * iteration over ResultSets and catching JDBC exceptions and translating

62 * them to the generic, more informative exception hierarchy defined in the

63 * {@code org.springframework.dao} package.

64 *

65 * <p>Code using this class need only implement callback interfaces, giving

66 * them a clearly defined contract. The {@link PreparedStatementCreator} callback

67 * interface creates a prepared statement given a Connection, providing SQL and

68 * any necessary parameters. The {@link ResultSetExtractor} interface extracts

69 * values from a ResultSet. See also {@link PreparedStatementSetter} and

70 * {@link RowMapper} for two popular alternative callback interfaces.

71 *

72 * <p>Can be used within a service implementation via direct instantiation

73 * with a DataSource reference, or get prepared in an application context

74 * and given to services as bean reference. Note: The DataSource should

75 * always be configured as a bean in the application context, in the first case

76 * given to the service directly, in the second case to the prepared template.

77 *

78 * <p>Because this class is parameterizable by the callback interfaces and

79 * the {@link org.springframework.jdbc.support.SQLExceptionTranslator}

80 * interface, there should be no need to subclass it.

81 *

82 * <p>All SQL operations performed by this class are logged at debug level,

83 * using "org.springframework.jdbc.core.JdbcTemplate" as log category.

84 *

85 * <p><b>NOTE: An instance of this class is thread-safe once configured.</b>

86 *

87 * @author Rod Johnson

88 * @author Juergen Hoeller

89 * @author Thomas Risberg

90 * @since May 3, 2001

91 * @see PreparedStatementCreator

92 * @see PreparedStatementSetter

93 * @see CallableStatementCreator

94 * @see PreparedStatementCallback

95 * @see CallableStatementCallback

96 * @see ResultSetExtractor

97 * @see RowCallbackHandler

98 * @see RowMapper

99 * @see org.springframework.jdbc.support.SQLExceptionTranslator

100 */

319	@Override

320	@Nullable

321	public <T> T execute(ConnectionCallback<T> action) throws DataAccessException {

322		Assert.notNull(action, "Callback object must not be null");

323

324		Connection con = DataSourceUtils.getConnection(obtainDataSource());

325		try {

326			// Create close-suppressing Connection proxy, also preparing returned Statements.

327			Connection conToUse = createConnectionProxy(con);

328			return action.doInConnection(conToUse);

329		}

330		catch (SQLException ex) {

331			// Release Connection early, to avoid potential connection pool deadlock

332			// in the case when the exception translator hasn't been initialized yet.

333			String sql = getSql(action);

334			DataSourceUtils.releaseConnection(con, getDataSource());

335			con = null;

336			throw translateException("ConnectionCallback", sql, ex);

337		}

338		finally {

339			DataSourceUtils.releaseConnection(con, getDataSource());

340		}

341	}

342

343	/**

344	 * Create a close-suppressing proxy for the given JDBC Connection.

345	 * Called by the {@code execute} method.

346	 * <p>The proxy also prepares returned JDBC Statements, applying

347	 * statement settings such as fetch size, max rows, and query timeout.

348	 * @param con the JDBC Connection to create a proxy for

349	 * @return the Connection proxy

350	 * @see java.sql.Connection#close()

351	 * @see #execute(ConnectionCallback)

352	 * @see #applyStatementSettings

353	 */

354	protected Connection createConnectionProxy(Connection con) {

355		return (Connection) Proxy.newProxyInstance(

356				ConnectionProxy.class.getClassLoader(),

357				new Class<?>[] {ConnectionProxy.class},

358				new CloseSuppressingInvocationHandler(con));

359	}

366	@Override

367	@Nullable

368	public <T> T execute(StatementCallback<T> action) throws DataAccessException {

369		Assert.notNull(action, "Callback object must not be null");

370

371		Connection con = DataSourceUtils.getConnection(obtainDataSource());

372		Statement stmt = null;

373		try {

374			stmt = con.createStatement();

375			applyStatementSettings(stmt);

376			T result = action.doInStatement(stmt);

377			handleWarnings(stmt);

378			return result;

379		}

380		catch (SQLException ex) {

381			// Release Connection early, to avoid potential connection pool deadlock

382			// in the case when the exception translator hasn't been initialized yet.

383			String sql = getSql(action);

384			JdbcUtils.closeStatement(stmt);

385			stmt = null;

386			DataSourceUtils.releaseConnection(con, getDataSource());

387			con = null;

388			throw translateException("StatementCallback", sql, ex);

389		}

390		finally {

391			JdbcUtils.closeStatement(stmt);

392			DataSourceUtils.releaseConnection(con, getDataSource());

393		}

394	}

517	@Override

518	public int[] batchUpdate(final String... sql) throws DataAccessException {

519		Assert.notEmpty(sql, "SQL array must not be empty");

520		if (logger.isDebugEnabled()) {

521			logger.debug("Executing SQL batch update of " + sql.length + " statements");

522		}

523

524		class BatchUpdateStatementCallback implements StatementCallback<int[]>, SqlProvider {

525

526			@Nullable

527			private String currSql;

528

529			@Override

530			public int[] doInStatement(Statement stmt) throws SQLException, DataAccessException {

531				int[] rowsAffected = new int[sql.length];

532				if (JdbcUtils.supportsBatchUpdates(stmt.getConnection())) {

533					for (String sqlStmt : sql) {

534						this.currSql = appendSql(this.currSql, sqlStmt);

535						stmt.addBatch(sqlStmt);

536					}

537					try {

538						rowsAffected = stmt.executeBatch();

539					}

540					catch (BatchUpdateException ex) {

541						String batchExceptionSql = null;

542						for (int i = 0; i < ex.getUpdateCounts().length; i++) {

543							if (ex.getUpdateCounts()[i] == Statement.EXECUTE_FAILED) {

544								batchExceptionSql = appendSql(batchExceptionSql, sql[i]);

545							}

546						}

547						if (StringUtils.hasLength(batchExceptionSql)) {

548							this.currSql = batchExceptionSql;

549						}

550						throw ex;

551					}

552				}

553				else {

554					for (int i = 0; i < sql.length; i++) {

555						this.currSql = sql[i];

556						if (!stmt.execute(sql[i])) {

557							rowsAffected[i] = stmt.getUpdateCount();

558						}

559						else {

560							throw new InvalidDataAccessApiUsageException("Invalid batch SQL statement: " + sql[i]);

561						}

562					}

563				}

564				return rowsAffected;

565			}

566

567			private String appendSql(@Nullable String sql, String statement) {

568				return (StringUtils.isEmpty(sql) ? statement : sql + "; " + statement);

569			}

570

571			@Override

572			@Nullable

573			public String getSql() {

574				return this.currSql;

575			}

576		}

577

578		int[] result = execute(new BatchUpdateStatementCallback());

579		Assert.state(result != null, "No update counts");

580		return result;

581	}

588	@Override

589	@Nullable

590	public <T> T execute(PreparedStatementCreator psc, PreparedStatementCallback<T> action)

591			throws DataAccessException {

592

593		Assert.notNull(psc, "PreparedStatementCreator must not be null");

594		Assert.notNull(action, "Callback object must not be null");

595		if (logger.isDebugEnabled()) {

596			String sql = getSql(psc);

597			logger.debug("Executing prepared SQL statement" + (sql != null ? " [" + sql + "]" : ""));

598		}

599

600		Connection con = DataSourceUtils.getConnection(obtainDataSource());

601		PreparedStatement ps = null;

602		try {

603			ps = psc.createPreparedStatement(con);

604			applyStatementSettings(ps);

605			T result = action.doInPreparedStatement(ps);

606			handleWarnings(ps);

607			return result;

608		}

609		catch (SQLException ex) {

610			// Release Connection early, to avoid potential connection pool deadlock

611			// in the case when the exception translator hasn't been initialized yet.

612			if (psc instanceof ParameterDisposer) {

613				((ParameterDisposer) psc).cleanupParameters();

614			}

615			String sql = getSql(psc);

616			JdbcUtils.closeStatement(ps);

617			ps = null;

618			DataSourceUtils.releaseConnection(con, getDataSource());

619			con = null;

620			throw translateException("PreparedStatementCallback", sql, ex);

621		}

622		finally {

623			if (psc instanceof ParameterDisposer) {

624				((ParameterDisposer) psc).cleanupParameters();

625			}

626			JdbcUtils.closeStatement(ps);

627			DataSourceUtils.releaseConnection(con, getDataSource());

628		}

629	}

637	/**

638	 * Query using a prepared statement, allowing for a PreparedStatementCreator

639	 * and a PreparedStatementSetter. Most other query methods use this method,

640	 * but application code will always work with either a creator or a setter.

641	 * @param psc Callback handler that can create a PreparedStatement given a

642	 * Connection

643	 * @param pss object that knows how to set values on the prepared statement.

644	 * If this is null, the SQL will be assumed to contain no bind parameters.

645	 * @param rse object that will extract results.

646	 * @return an arbitrary result object, as returned by the ResultSetExtractor

647	 * @throws DataAccessException if there is any problem

648	 */

649	@Nullable

650	public <T> T query(

651			PreparedStatementCreator psc, @Nullable final PreparedStatementSetter pss, final ResultSetExtractor<T> rse)

652			throws DataAccessException {

653

654		Assert.notNull(rse, "ResultSetExtractor must not be null");

655		logger.debug("Executing prepared SQL query");

656

657		return execute(psc, new PreparedStatementCallback<T>() {

658			@Override

659			@Nullable

660			public T doInPreparedStatement(PreparedStatement ps) throws SQLException {

661				ResultSet rs = null;

662				try {

663					if (pss != null) {

664						pss.setValues(ps);

665					}

666					rs = ps.executeQuery();

667					return rse.extractData(rs);

668				}

669				finally {

670					JdbcUtils.closeResultSet(rs);

671					if (pss instanceof ParameterDisposer) {

672						((ParameterDisposer) pss).cleanupParameters();

673					}

674				}

675			}

676		});

677	}

918	@Override

919	public int[] batchUpdate(String sql, final BatchPreparedStatementSetter pss) throws DataAccessException {

920		if (logger.isDebugEnabled()) {

921			logger.debug("Executing SQL batch update [" + sql + "]");

922		}

923

924		int[] result = execute(sql, (PreparedStatementCallback<int[]>) ps -> {

925			try {

926				int batchSize = pss.getBatchSize();

927				InterruptibleBatchPreparedStatementSetter ipss =

928						(pss instanceof InterruptibleBatchPreparedStatementSetter ?

929						(InterruptibleBatchPreparedStatementSetter) pss : null);

930				if (JdbcUtils.supportsBatchUpdates(ps.getConnection())) {

931					for (int i = 0; i < batchSize; i++) {

932						pss.setValues(ps, i);

933						if (ipss != null && ipss.isBatchExhausted(i)) {

934							break;

935						}

936						ps.addBatch();

937					}

938					return ps.executeBatch();

939				}

940				else {

941					List<Integer> rowsAffected = new ArrayList<>();

942					for (int i = 0; i < batchSize; i++) {

943						pss.setValues(ps, i);

944						if (ipss != null && ipss.isBatchExhausted(i)) {

945							break;

946						}

947						rowsAffected.add(ps.executeUpdate());

948					}

949					int[] rowsAffectedArray = new int[rowsAffected.size()];

950					for (int i = 0; i < rowsAffectedArray.length; i++) {

951						rowsAffectedArray[i] = rowsAffected.get(i);

952					}

953					return rowsAffectedArray;

954				}

955			}

956			finally {

957				if (pss instanceof ParameterDisposer) {

958					((ParameterDisposer) pss).cleanupParameters();

959				}

960			}

961		});

962

963		Assert.state(result != null, "No result array");

964		return result;

965	}

977	@Override

978	public <T> int[][] batchUpdate(String sql, final Collection<T> batchArgs, final int batchSize,

979			final ParameterizedPreparedStatementSetter<T> pss) throws DataAccessException {

980

981		if (logger.isDebugEnabled()) {

982			logger.debug("Executing SQL batch update [" + sql + "] with a batch size of " + batchSize);

983		}

984		int[][] result = execute(sql, (PreparedStatementCallback<int[][]>) ps -> {

985			List<int[]> rowsAffected = new ArrayList<>();

986			try {

987				boolean batchSupported = true;

988				if (!JdbcUtils.supportsBatchUpdates(ps.getConnection())) {

989					batchSupported = false;

990					logger.warn("JDBC Driver does not support Batch updates; resorting to single statement execution");

991				}

992				int n = 0;

993				for (T obj : batchArgs) {

994					pss.setValues(ps, obj);

995					n++;

996					if (batchSupported) {

997						ps.addBatch();

998						if (n % batchSize == 0 || n == batchArgs.size()) {

999							if (logger.isDebugEnabled()) {

1000								int batchIdx = (n % batchSize == 0) ? n / batchSize : (n / batchSize) + 1;

1001								int items = n - ((n % batchSize == 0) ? n / batchSize - 1 : (n / batchSize)) * batchSize;

1002								logger.debug("Sending SQL batch update #" + batchIdx + " with " + items + " items");

1003							}

1004							rowsAffected.add(ps.executeBatch());

1005						}

1006					}

1007					else {

1008						int i = ps.executeUpdate();

1009						rowsAffected.add(new int[] {i});

1010					}

1011				}

1012				int[][] result1 = new int[rowsAffected.size()][];

1013				for (int i = 0; i < result1.length; i++) {

1014					result1[i] = rowsAffected.get(i);

1015				}

1016				return result1;

1017			}

1018			finally {

1019				if (pss instanceof ParameterDisposer) {

1020					((ParameterDisposer) pss).cleanupParameters();

1021				}

1022			}

1023		});

1024

1025		Assert.state(result != null, "No result array");

1026		return result;

1027	}

1033	@Override

1034	@Nullable

1035	public <T> T execute(CallableStatementCreator csc, CallableStatementCallback<T> action)

1036			throws DataAccessException {

1037

1038		Assert.notNull(csc, "CallableStatementCreator must not be null");

1039		Assert.notNull(action, "Callback object must not be null");

1040		if (logger.isDebugEnabled()) {

1041			String sql = getSql(csc);

1042			logger.debug("Calling stored procedure" + (sql != null ? " [" + sql  + "]" : ""));

1043		}

1044

1045		Connection con = DataSourceUtils.getConnection(obtainDataSource());

1046		CallableStatement cs = null;

1047		try {

1048			cs = csc.createCallableStatement(con);

1049			applyStatementSettings(cs);

1050			T result = action.doInCallableStatement(cs);

1051			handleWarnings(cs);

1052			return result;

1053		}

1054		catch (SQLException ex) {

1055			// Release Connection early, to avoid potential connection pool deadlock

1056			// in the case when the exception translator hasn't been initialized yet.

1057			if (csc instanceof ParameterDisposer) {

1058				((ParameterDisposer) csc).cleanupParameters();

1059			}

1060			String sql = getSql(csc);

1061			JdbcUtils.closeStatement(cs);

1062			cs = null;

1063			DataSourceUtils.releaseConnection(con, getDataSource());

1064			con = null;

1065			throw translateException("CallableStatementCallback", sql, ex);

1066		}

1067		finally {

1068			if (csc instanceof ParameterDisposer) {

1069				((ParameterDisposer) csc).cleanupParameters();

1070			}

1071			JdbcUtils.closeStatement(cs);

1072			DataSourceUtils.releaseConnection(con, getDataSource());

1073		}

1074	}

1082	@Override

1083	public Map<String, Object> call(CallableStatementCreator csc, List<SqlParameter> declaredParameters)

1084			throws DataAccessException {

1085

1086		final List<SqlParameter> updateCountParameters = new ArrayList<>();

1087		final List<SqlParameter> resultSetParameters = new ArrayList<>();

1088		final List<SqlParameter> callParameters = new ArrayList<>();

1089

1090		for (SqlParameter parameter : declaredParameters) {

1091			if (parameter.isResultsParameter()) {

1092				if (parameter instanceof SqlReturnResultSet) {

1093					resultSetParameters.add(parameter);

1094				}

1095				else {

1096					updateCountParameters.add(parameter);

1097				}

1098			}

1099			else {

1100				callParameters.add(parameter);

1101			}

1102		}

1103

1104		Map<String, Object> result = execute(csc, cs -> {

1105			boolean retVal = cs.execute();

1106			int updateCount = cs.getUpdateCount();

1107			if (logger.isDebugEnabled()) {

1108				logger.debug("CallableStatement.execute() returned '" + retVal + "'");

1109				logger.debug("CallableStatement.getUpdateCount() returned " + updateCount);

1110			}

1111			Map<String, Object> returnedResults = createResultsMap();

1112			if (retVal || updateCount != -1) {

1113				returnedResults.putAll(extractReturnedResults(cs, updateCountParameters, resultSetParameters, updateCount));

1114			}

1115			returnedResults.putAll(extractOutputParameters(cs, callParameters));

1116			return returnedResults;

1117		});

1118

1119		Assert.state(result != null, "No result map");

1120		return result;

1121	}

1123	/**

1124	 * Extract returned ResultSets from the completed stored procedure.

1125	 * @param cs JDBC wrapper for the stored procedure

1126	 * @param updateCountParameters Parameter list of declared update count parameters for the stored procedure

1127	 * @param resultSetParameters Parameter list of declared resultSet parameters for the stored procedure

1128	 * @return Map that contains returned results

1129	 */

1130	protected Map<String, Object> extractReturnedResults(CallableStatement cs,

1131			@Nullable List<SqlParameter> updateCountParameters, @Nullable List<SqlParameter> resultSetParameters,

1132			int updateCount) throws SQLException {

1133

1134		Map<String, Object> returnedResults = new HashMap<>();

1135		int rsIndex = 0;

1136		int updateIndex = 0;

1137		boolean moreResults;

1138		if (!this.skipResultsProcessing) {

1139			do {

1140				if (updateCount == -1) {

1141					if (resultSetParameters != null && resultSetParameters.size() > rsIndex) {

1142						SqlReturnResultSet declaredRsParam = (SqlReturnResultSet) resultSetParameters.get(rsIndex);

1143						returnedResults.putAll(processResultSet(cs.getResultSet(), declaredRsParam));

1144						rsIndex++;

1145					}

1146					else {

1147						if (!this.skipUndeclaredResults) {

1148							String rsName = RETURN_RESULT_SET_PREFIX + (rsIndex + 1);

1149							SqlReturnResultSet undeclaredRsParam = new SqlReturnResultSet(rsName, getColumnMapRowMapper());

1150							if (logger.isDebugEnabled()) {

1151								logger.debug("Added default SqlReturnResultSet parameter named '" + rsName + "'");

1152							}

1153							returnedResults.putAll(processResultSet(cs.getResultSet(), undeclaredRsParam));

1154							rsIndex++;

1155						}

1156					}

1157				}

1158				else {

1159					if (updateCountParameters != null && updateCountParameters.size() > updateIndex) {

1160						SqlReturnUpdateCount ucParam = (SqlReturnUpdateCount) updateCountParameters.get(updateIndex);

1161						String declaredUcName = ucParam.getName();

1162						returnedResults.put(declaredUcName, updateCount);

1163						updateIndex++;

1164					}

1165					else {

1166						if (!this.skipUndeclaredResults) {

1167							String undeclaredName = RETURN_UPDATE_COUNT_PREFIX + (updateIndex + 1);

1168							if (logger.isDebugEnabled()) {

1169								logger.debug("Added default SqlReturnUpdateCount parameter named '" + undeclaredName + "'");

1170							}

1171							returnedResults.put(undeclaredName, updateCount);

1172							updateIndex++;

1173						}

1174					}

1175				}

1176				moreResults = cs.getMoreResults();

1177				updateCount = cs.getUpdateCount();

1178				if (logger.isDebugEnabled()) {

1179					logger.debug("CallableStatement.getUpdateCount() returned " + updateCount);

1180				}

1181			}

1182			while (moreResults || updateCount != -1);

1183		}

1184		return returnedResults;

1185	}

1187	/**

1188	 * Extract output parameters from the completed stored procedure.

1189	 * @param cs JDBC wrapper for the stored procedure

1190	 * @param parameters parameter list for the stored procedure

1191	 * @return Map that contains returned results

1192	 */

1193	protected Map<String, Object> extractOutputParameters(CallableStatement cs, List<SqlParameter> parameters)

1194			throws SQLException {

1195

1196		Map<String, Object> returnedResults = new HashMap<>();

1197		int sqlColIndex = 1;

1198		for (SqlParameter param : parameters) {

1199			if (param instanceof SqlOutParameter) {

1200				SqlOutParameter outParam = (SqlOutParameter) param;

1201				Assert.state(outParam.getName() != null, "Anonymous parameters not allowed");

1202				SqlReturnType returnType = outParam.getSqlReturnType();

1203				if (returnType != null) {

1204					Object out = returnType.getTypeValue(cs, sqlColIndex, outParam.getSqlType(), outParam.getTypeName());

1205					returnedResults.put(outParam.getName(), out);

1206				}

1207				else {

1208					Object out = cs.getObject(sqlColIndex);

1209					if (out instanceof ResultSet) {

1210						if (outParam.isResultSetSupported()) {

1211							returnedResults.putAll(processResultSet((ResultSet) out, outParam));

1212						}

1213						else {

1214							String rsName = outParam.getName();

1215							SqlReturnResultSet rsParam = new SqlReturnResultSet(rsName, getColumnMapRowMapper());

1216							returnedResults.putAll(processResultSet((ResultSet) out, rsParam));

1217							if (logger.isDebugEnabled()) {

1218								logger.debug("Added default SqlReturnResultSet parameter named '" + rsName + "'");

1219							}

1220						}

1221					}

1222					else {

1223						returnedResults.put(outParam.getName(), out);

1224					}

1225				}

1226			}

1227			if (!(param.isResultsParameter())) {

1228				sqlColIndex++;

1229			}

1230		}

1231		return returnedResults;

1232	}

1234	/**

1235	 * Process the given ResultSet from a stored procedure.

1236	 * @param rs the ResultSet to process

1237	 * @param param the corresponding stored procedure parameter

1238	 * @return Map that contains returned results

1239	 */

1240	protected Map<String, Object> processResultSet(

1241			@Nullable ResultSet rs, ResultSetSupportingSqlParameter param) throws SQLException {

1242

1243		if (rs == null) {

1244			return Collections.emptyMap();

1245		}

1246

1247		Map<String, Object> returnedResults = new HashMap<>();

1248		try {

1249			if (param.getRowMapper() != null) {

1250				RowMapper<?> rowMapper = param.getRowMapper();

1251				Object result = (new RowMapperResultSetExtractor<>(rowMapper)).extractData(rs);

1252				returnedResults.put(param.getName(), result);

1253			}

1254			else if (param.getRowCallbackHandler() != null) {

1255				RowCallbackHandler rch = param.getRowCallbackHandler();

1256				(new RowCallbackHandlerResultSetExtractor(rch)).extractData(rs);

1257				returnedResults.put(param.getName(), "ResultSet returned from stored procedure was processed");

1258			}

1259			else if (param.getResultSetExtractor() != null) {

1260				Object result = param.getResultSetExtractor().extractData(rs);

1261				returnedResults.put(param.getName(), result);

1262			}

1263		}

1264		finally {

1265			JdbcUtils.closeResultSet(rs);

1266		}

1267		return returnedResults;

1268	}

1294	/**

1295	 * Create a Map instance to be used as the results map.

1296	 * <p>If {@link #resultsMapCaseInsensitive} has been set to true,

1297	 * a {@link LinkedCaseInsensitiveMap} will be created; otherwise, a

1304		if (isResultsMapCaseInsensitive()) {

1305			return new LinkedCaseInsensitiveMap<>();

1358	 * Throw an SQLWarningException if we're not ignoring warnings,

1359	 * else log the warnings (at debug level).

1360	 * @param stmt the current JDBC statement

1361	 * @throws SQLWarningException if not ignoring warnings

1362	 * @see org.springframework.jdbc.SQLWarningException

1363	 */

1364	protected void handleWarnings(Statement stmt) throws SQLException {

1365		if (isIgnoreWarnings()) {

1366			if (logger.isDebugEnabled()) {

1367				SQLWarning warningToLog = stmt.getWarnings();

1368				while (warningToLog != null) {

1369					logger.debug("SQLWarning ignored: SQL state '" + warningToLog.getSQLState() + "', error code '" +

1370							warningToLog.getErrorCode() + "', message [" + warningToLog.getMessage() + "]");

1371					warningToLog = warningToLog.getNextWarning();

1372				}

1373			}

1374		}

1375		else {

1376			handleWarnings(stmt.getWarnings());

1377		}

1378	}

1434	/**

1435	 * Invocation handler that suppresses close calls on JDBC Connections.

1436	 * Also prepares returned Statement (Prepared/CallbackStatement) objects.

1437	 * @see java.sql.Connection#close()

1438	 */

1439	private class CloseSuppressingInvocationHandler implements InvocationHandler {

1440

1441		private final Connection target;

1442

1443		public CloseSuppressingInvocationHandler(Connection target) {

1444			this.target = target;

1445		}

1446

1447		@Override

1448		@Nullable

1449		public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {

1450			// Invocation on ConnectionProxy interface coming in...

1451

1452			if (method.getName().equals("equals")) {

1453				// Only consider equal when proxies are identical.

1454				return (proxy == args[0]);

1455			}

1456			else if (method.getName().equals("hashCode")) {

1457				// Use hashCode of PersistenceManager proxy.

1458				return System.identityHashCode(proxy);

1459			}

1460			else if (method.getName().equals("unwrap")) {

1461				if (((Class<?>) args[0]).isInstance(proxy)) {

1462					return proxy;

1463				}

1464			}

1465			else if (method.getName().equals("isWrapperFor")) {

1466				if (((Class<?>) args[0]).isInstance(proxy)) {

1467					return true;

1468				}

1469			}

1470			else if (method.getName().equals("close")) {

1471				// Handle close method: suppress, not valid.

1472				return null;

1473			}

1474			else if (method.getName().equals("isClosed")) {

1475				return false;

1476			}

1477			else if (method.getName().equals("getTargetConnection")) {

1478				// Handle getTargetConnection method: return underlying Connection.

1479				return this.target;

1480			}

1481

1482			// Invoke method on target Connection.

1483			try {

1484				Object retVal = method.invoke(this.target, args);

1485

1486				// If return value is a JDBC Statement, apply statement settings

1487				// (fetch size, max rows, transaction timeout).

1488				if (retVal instanceof Statement) {

1489					applyStatementSettings(((Statement) retVal));

1490				}

1491

1492				return retVal;

1493			}

1494			catch (InvocationTargetException ex) {

1495				throw ex.getTargetException();

1496			}

1497		}

1498	}
