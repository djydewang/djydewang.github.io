56/**

57 * This is the central class for {@code javax.validation} (JSR-303) setup in a Spring

58 * application context: It bootstraps a {@code javax.validation.ValidationFactory} and

59 * exposes it through the Spring {@link org.springframework.validation.Validator} interface

60 * as well as through the JSR-303 {@link javax.validation.Validator} interface and the

61 * {@link javax.validation.ValidatorFactory} interface itself.

62 *

63 * <p>When talking to an instance of this bean through the Spring or JSR-303 Validator interfaces,

64 * you'll be talking to the default Validator of the underlying ValidatorFactory. This is very

65 * convenient in that you don't have to perform yet another call on the factory, assuming that

66 * you will almost always use the default Validator anyway. This can also be injected directly

67 * into any target dependency of type {@link org.springframework.validation.Validator}!

68 *

69 * <p><b>As of Spring 5.0, this class requires Bean Validation 1.1+, with special support

70 * for Hibernate Validator 5.x</b> (see {@link #setValidationMessageSource}).

71 * This class is also runtime-compatible with Bean Validation 2.0 and Hibernate Validator 6.0,

72 * with one special note: If you'd like to call BV 2.0's {@code getClockProvider()} method,

73 * obtain the native {@code ValidatorFactory} through {@code #unwrap(ValidatorFactory.class)}

74 * and call the {@code getClockProvider()} method on the returned native reference there.

75 *

76 * <p>This class is also being used by Spring's MVC configuration namespace, in case of the

77 * {@code javax.validation} API being present but no explicit Validator having been configured.

78 *

79 * @author Juergen Hoeller

80 * @since 3.0

81 * @see javax.validation.ValidatorFactory

82 * @see javax.validation.Validator

83 * @see javax.validation.Validation#buildDefaultValidatorFactory()

84 * @see javax.validation.ValidatorFactory#getValidator()

85 */

86public class LocalValidatorFactoryBean extends SpringValidatorAdapter

87		implements ValidatorFactory, ApplicationContextAware, InitializingBean, DisposableBean {

148	/**

149	 * Specify a custom Spring MessageSource for resolving validation messages,

150	 * instead of relying on JSR-303's default "ValidationMessages.properties" bundle

151	 * in the classpath. This may refer to a Spring context's shared "messageSource" bean,

152	 * or to some special MessageSource setup for validation purposes only.

153	 * <p><b>NOTE:</b> This feature requires Hibernate Validator 4.3 or higher on the classpath.

154	 * You may nevertheless use a different validation provider but Hibernate Validator's

155	 * {@link ResourceBundleMessageInterpolator} class must be accessible during configuration.

156	 * <p>Specify either this property or {@link #setMessageInterpolator "messageInterpolator"},

157	 * not both. If you would like to build a custom MessageInterpolator, consider deriving from

158	 * Hibernate Validator's {@link ResourceBundleMessageInterpolator} and passing in a

159	 * Spring-based {@code ResourceBundleLocator} when constructing your interpolator.

160	 * <p>In order for Hibernate's default validation messages to be resolved still, your

161	 * {@link MessageSource} must be configured for optional resolution (usually the default).

162	 * In particular, the {@code MessageSource} instance specified here should not apply

163	 * {@link org.springframework.context.support.AbstractMessageSource#setUseCodeAsDefaultMessage

164	 * "useCodeAsDefaultMessage"} behavior. Please double-check your setup accordingly.

165	 * @see ResourceBundleMessageInterpolator

166	 */

167	public void setValidationMessageSource(MessageSource messageSource) {

168		this.messageInterpolator = HibernateValidatorDelegate.buildMessageInterpolator(messageSource);

169	}

240	@Override

241	@SuppressWarnings({"rawtypes", "unchecked"})

242	public void afterPropertiesSet() {

243		Configuration<?> configuration;

244		if (this.providerClass != null) {

245			ProviderSpecificBootstrap bootstrap = Validation.byProvider(this.providerClass);

246			if (this.validationProviderResolver != null) {

247				bootstrap = bootstrap.providerResolver(this.validationProviderResolver);

248			}

249			configuration = bootstrap.configure();

250		}

251		else {

252			GenericBootstrap bootstrap = Validation.byDefaultProvider();

253			if (this.validationProviderResolver != null) {

254				bootstrap = bootstrap.providerResolver(this.validationProviderResolver);

255			}

256			configuration = bootstrap.configure();

257		}

258

259		// Try Hibernate Validator 5.2's externalClassLoader(ClassLoader) method

260		if (this.applicationContext != null) {

261			try {

262				Method eclMethod = configuration.getClass().getMethod("externalClassLoader", ClassLoader.class);

263				ReflectionUtils.invokeMethod(eclMethod, configuration, this.applicationContext.getClassLoader());

264			}

265			catch (NoSuchMethodException ex) {

266				// Ignore - no Hibernate Validator 5.2+ or similar provider

267			}

268		}

269

270		MessageInterpolator targetInterpolator = this.messageInterpolator;

271		if (targetInterpolator == null) {

272			targetInterpolator = configuration.getDefaultMessageInterpolator();

273		}

274		configuration.messageInterpolator(new LocaleContextMessageInterpolator(targetInterpolator));

275

276		if (this.traversableResolver != null) {

277			configuration.traversableResolver(this.traversableResolver);

278		}

279

280		ConstraintValidatorFactory targetConstraintValidatorFactory = this.constraintValidatorFactory;

281		if (targetConstraintValidatorFactory == null && this.applicationContext != null) {

282			targetConstraintValidatorFactory =

283					new SpringConstraintValidatorFactory(this.applicationContext.getAutowireCapableBeanFactory());

284		}

285		if (targetConstraintValidatorFactory != null) {

286			configuration.constraintValidatorFactory(targetConstraintValidatorFactory);

287		}

288

289		if (this.parameterNameDiscoverer != null) {

290			configureParameterNameProvider(this.parameterNameDiscoverer, configuration);

291		}

292

293		if (this.mappingLocations != null) {

294			for (Resource location : this.mappingLocations) {

295				try {

296					configuration.addMapping(location.getInputStream());

297				}

298				catch (IOException ex) {

299					throw new IllegalStateException("Cannot read mapping resource: " + location);

300				}

301			}

302		}

303

304		this.validationPropertyMap.forEach(configuration::addProperty);

305

306		// Allow for custom post-processing before we actually build the ValidatorFactory.

307		postProcessConfiguration(configuration);

308

309		this.validatorFactory = configuration.buildValidatorFactory();

310		setTargetValidator(this.validatorFactory.getValidator());

311	}

312

313	private void configureParameterNameProvider(ParameterNameDiscoverer discoverer, Configuration<?> configuration) {

314		final ParameterNameProvider defaultProvider = configuration.getDefaultParameterNameProvider();

315		configuration.parameterNameProvider(new ParameterNameProvider() {

316			@Override

317			public List<String> getParameterNames(Constructor<?> constructor) {

318				String[] paramNames = discoverer.getParameterNames(constructor);

319				return (paramNames != null ? Arrays.asList(paramNames) :

320						defaultProvider.getParameterNames(constructor));

321			}

322			@Override

323			public List<String> getParameterNames(Method method) {

324				String[] paramNames = discoverer.getParameterNames(method);

325				return (paramNames != null ? Arrays.asList(paramNames) :

326						defaultProvider.getParameterNames(method));

327			}

328		});

329	}

391	@Override

392	@SuppressWarnings("unchecked")

393	public <T> T unwrap(@Nullable Class<T> type) {

394		if (type == null || !ValidatorFactory.class.isAssignableFrom(type)) {

395			try {

396				return super.unwrap(type);

397			}

398			catch (ValidationException ex) {

399				// ignore - we'll try ValidatorFactory unwrapping next

400			}

401		}

402		if (this.validatorFactory != null) {

403			try {

404				return this.validatorFactory.unwrap(type);

405			}

406			catch (ValidationException ex) {

407				// ignore if just being asked for ValidatorFactory

408				if (ValidatorFactory.class == type) {

409					return (T) this.validatorFactory;

410				}

411				throw ex;

412			}

413		}

414		throw new ValidationException("Cannot unwrap to " + type);

415	}
