550	@Override

551	public void afterPropertiesSet() {

552		// Do this first, it may add ResponseBody advice beans

553		initControllerAdviceCache();

554

555		if (this.argumentResolvers == null) {

556			List<HandlerMethodArgumentResolver> resolvers = getDefaultArgumentResolvers();

557			this.argumentResolvers = new HandlerMethodArgumentResolverComposite().addResolvers(resolvers);

558		}

559		if (this.initBinderArgumentResolvers == null) {

560			List<HandlerMethodArgumentResolver> resolvers = getDefaultInitBinderArgumentResolvers();

561			this.initBinderArgumentResolvers = new HandlerMethodArgumentResolverComposite().addResolvers(resolvers);

562		}

563		if (this.returnValueHandlers == null) {

564			List<HandlerMethodReturnValueHandler> handlers = getDefaultReturnValueHandlers();

565			this.returnValueHandlers = new HandlerMethodReturnValueHandlerComposite().addHandlers(handlers);

566		}

567	}

569	private void initControllerAdviceCache() {

570		if (getApplicationContext() == null) {

571			return;

572		}

573		if (logger.isInfoEnabled()) {

574			logger.info("Looking for @ControllerAdvice: " + getApplicationContext());

575		}

576

577		List<ControllerAdviceBean> adviceBeans = ControllerAdviceBean.findAnnotatedBeans(getApplicationContext());

578		AnnotationAwareOrderComparator.sort(adviceBeans);

579

580		List<Object> requestResponseBodyAdviceBeans = new ArrayList<>();

581

582		for (ControllerAdviceBean adviceBean : adviceBeans) {

583			Class<?> beanType = adviceBean.getBeanType();

584			if (beanType == null) {

585				throw new IllegalStateException("Unresolvable type for ControllerAdviceBean: " + adviceBean);

586			}

587			Set<Method> attrMethods = MethodIntrospector.selectMethods(beanType, MODEL_ATTRIBUTE_METHODS);

588			if (!attrMethods.isEmpty()) {

589				this.modelAttributeAdviceCache.put(adviceBean, attrMethods);

590				if (logger.isInfoEnabled()) {

591					logger.info("Detected @ModelAttribute methods in " + adviceBean);

592				}

593			}

594			Set<Method> binderMethods = MethodIntrospector.selectMethods(beanType, INIT_BINDER_METHODS);

595			if (!binderMethods.isEmpty()) {

596				this.initBinderAdviceCache.put(adviceBean, binderMethods);

597				if (logger.isInfoEnabled()) {

598					logger.info("Detected @InitBinder methods in " + adviceBean);

599				}

600			}

601			if (RequestBodyAdvice.class.isAssignableFrom(beanType)) {

602				requestResponseBodyAdviceBeans.add(adviceBean);

603				if (logger.isInfoEnabled()) {

604					logger.info("Detected RequestBodyAdvice bean in " + adviceBean);

605				}

606			}

607			if (ResponseBodyAdvice.class.isAssignableFrom(beanType)) {

608				requestResponseBodyAdviceBeans.add(adviceBean);

609				if (logger.isInfoEnabled()) {

610					logger.info("Detected ResponseBodyAdvice bean in " + adviceBean);

611				}

612			}

613		}

614

615		if (!requestResponseBodyAdviceBeans.isEmpty()) {

616			this.requestResponseBodyAdvice.addAll(0, requestResponseBodyAdviceBeans);

617		}

618	}

760	@Override

761	protected ModelAndView handleInternal(HttpServletRequest request,

762			HttpServletResponse response, HandlerMethod handlerMethod) throws Exception {

763

764		ModelAndView mav;

765		checkRequest(request);

766

767		// Execute invokeHandlerMethod in synchronized block if required.

768		if (this.synchronizeOnSession) {

769			HttpSession session = request.getSession(false);

770			if (session != null) {

771				Object mutex = WebUtils.getSessionMutex(session);

772				synchronized (mutex) {

773					mav = invokeHandlerMethod(request, response, handlerMethod);

774				}

775			}

776			else {

777				// No HttpSession available -> no mutex necessary

778				mav = invokeHandlerMethod(request, response, handlerMethod);

779			}

780		}

781		else {

782			// No synchronization on session demanded at all...

783			mav = invokeHandlerMethod(request, response, handlerMethod);

784		}

785

786		if (!response.containsHeader(HEADER_CACHE_CONTROL)) {

787			if (getSessionAttributesHandler(handlerMethod).hasSessionAttributes()) {

788				applyCacheSeconds(response, this.cacheSecondsForSessionAttributeHandlers);

789			}

790			else {

791				prepareResponse(response);

792			}

793		}

794

795		return mav;

796	}

809	/**

810	 * Return the {@link SessionAttributesHandler} instance for the given handler type

811	 * (never {@code null}).

812	 */

813	private SessionAttributesHandler getSessionAttributesHandler(HandlerMethod handlerMethod) {

814		Class<?> handlerType = handlerMethod.getBeanType();

815		SessionAttributesHandler sessionAttrHandler = this.sessionAttributesHandlerCache.get(handlerType);

816		if (sessionAttrHandler == null) {

817			synchronized (this.sessionAttributesHandlerCache) {

818				sessionAttrHandler = this.sessionAttributesHandlerCache.get(handlerType);

819				if (sessionAttrHandler == null) {

820					sessionAttrHandler = new SessionAttributesHandler(handlerType, sessionAttributeStore);

821					this.sessionAttributesHandlerCache.put(handlerType, sessionAttrHandler);

822				}

823			}

824		}

825		return sessionAttrHandler;

826	}

828	/**

829	 * Invoke the {@link RequestMapping} handler method preparing a {@link ModelAndView}

830	 * if view resolution is required.

831	 * @since 4.2

832	 * @see #createInvocableHandlerMethod(HandlerMethod)

833	 */

834	@Nullable

835	protected ModelAndView invokeHandlerMethod(HttpServletRequest request,

836			HttpServletResponse response, HandlerMethod handlerMethod) throws Exception {

837

838		ServletWebRequest webRequest = new ServletWebRequest(request, response);

839		try {

840			WebDataBinderFactory binderFactory = getDataBinderFactory(handlerMethod);

841			ModelFactory modelFactory = getModelFactory(handlerMethod, binderFactory);

842

843			ServletInvocableHandlerMethod invocableMethod = createInvocableHandlerMethod(handlerMethod);

844			if (this.argumentResolvers != null) {

845				invocableMethod.setHandlerMethodArgumentResolvers(this.argumentResolvers);

846			}

847			if (this.returnValueHandlers != null) {

848				invocableMethod.setHandlerMethodReturnValueHandlers(this.returnValueHandlers);

849			}

850			invocableMethod.setDataBinderFactory(binderFactory);

851			invocableMethod.setParameterNameDiscoverer(this.parameterNameDiscoverer);

852

853			ModelAndViewContainer mavContainer = new ModelAndViewContainer();

854			mavContainer.addAllAttributes(RequestContextUtils.getInputFlashMap(request));

855			modelFactory.initModel(webRequest, mavContainer, invocableMethod);

856			mavContainer.setIgnoreDefaultModelOnRedirect(this.ignoreDefaultModelOnRedirect);

857

858			AsyncWebRequest asyncWebRequest = WebAsyncUtils.createAsyncWebRequest(request, response);

859			asyncWebRequest.setTimeout(this.asyncRequestTimeout);

860

861			WebAsyncManager asyncManager = WebAsyncUtils.getAsyncManager(request);

862			asyncManager.setTaskExecutor(this.taskExecutor);

863			asyncManager.setAsyncWebRequest(asyncWebRequest);

864			asyncManager.registerCallableInterceptors(this.callableInterceptors);

865			asyncManager.registerDeferredResultInterceptors(this.deferredResultInterceptors);

866

867			if (asyncManager.hasConcurrentResult()) {

868				Object result = asyncManager.getConcurrentResult();

869				mavContainer = (ModelAndViewContainer) asyncManager.getConcurrentResultContext()[0];

870				asyncManager.clearConcurrentResult();

871				if (logger.isDebugEnabled()) {

872					logger.debug("Found concurrent result value [" + result + "]");

873				}

874				invocableMethod = invocableMethod.wrapConcurrentResult(result);

875			}

876

877			invocableMethod.invokeAndHandle(webRequest, mavContainer);

878			if (asyncManager.isConcurrentHandlingStarted()) {

879				return null;

880			}

881

882			return getModelAndView(mavContainer, modelFactory, webRequest);

883		}

884		finally {

885			webRequest.requestCompleted();

886		}

887	}

899	private ModelFactory getModelFactory(HandlerMethod handlerMethod, WebDataBinderFactory binderFactory) {

900		SessionAttributesHandler sessionAttrHandler = getSessionAttributesHandler(handlerMethod);

901		Class<?> handlerType = handlerMethod.getBeanType();

902		Set<Method> methods = this.modelAttributeCache.get(handlerType);

903		if (methods == null) {

904			methods = MethodIntrospector.selectMethods(handlerType, MODEL_ATTRIBUTE_METHODS);

905			this.modelAttributeCache.put(handlerType, methods);

906		}

907		List<InvocableHandlerMethod> attrMethods = new ArrayList<>();

908		// Global methods first

909		this.modelAttributeAdviceCache.forEach((clazz, methodSet) -> {

910			if (clazz.isApplicableToBeanType(handlerType)) {

911				Object bean = clazz.resolveBean();

912				for (Method method : methodSet) {

913					attrMethods.add(createModelAttributeMethod(binderFactory, bean, method));

914				}

915			}

916		});

917		for (Method method : methods) {

918			Object bean = handlerMethod.getBean();

919			attrMethods.add(createModelAttributeMethod(binderFactory, bean, method));

920		}

921		return new ModelFactory(attrMethods, binderFactory, sessionAttrHandler);

922	}

924	private InvocableHandlerMethod createModelAttributeMethod(WebDataBinderFactory factory, Object bean, Method method) {

925		InvocableHandlerMethod attrMethod = new InvocableHandlerMethod(bean, method);

926		if (this.argumentResolvers != null) {

927			attrMethod.setHandlerMethodArgumentResolvers(this.argumentResolvers);

928		}

929		attrMethod.setParameterNameDiscoverer(this.parameterNameDiscoverer);

930		attrMethod.setDataBinderFactory(factory);

931		return attrMethod;

932	}

934	private WebDataBinderFactory getDataBinderFactory(HandlerMethod handlerMethod) throws Exception {

935		Class<?> handlerType = handlerMethod.getBeanType();

936		Set<Method> methods = this.initBinderCache.get(handlerType);

937		if (methods == null) {

938			methods = MethodIntrospector.selectMethods(handlerType, INIT_BINDER_METHODS);

939			this.initBinderCache.put(handlerType, methods);

940		}

941		List<InvocableHandlerMethod> initBinderMethods = new ArrayList<>();

942		// Global methods first

943		this.initBinderAdviceCache.forEach((clazz, methodSet) -> {

944			if (clazz.isApplicableToBeanType(handlerType)) {

945				Object bean = clazz.resolveBean();

946				for (Method method : methodSet) {

947					initBinderMethods.add(createInitBinderMethod(bean, method));

948				}

949			}

950		});

951		for (Method method : methods) {

952			Object bean = handlerMethod.getBean();

953			initBinderMethods.add(createInitBinderMethod(bean, method));

954		}

955		return createDataBinderFactory(initBinderMethods);

956	}

958	private InvocableHandlerMethod createInitBinderMethod(Object bean, Method method) {

959		InvocableHandlerMethod binderMethod = new InvocableHandlerMethod(bean, method);

960		if (this.initBinderArgumentResolvers != null) {

961			binderMethod.setHandlerMethodArgumentResolvers(this.initBinderArgumentResolvers);

962		}

963		binderMethod.setDataBinderFactory(new DefaultDataBinderFactory(this.webBindingInitializer));

964		binderMethod.setParameterNameDiscoverer(this.parameterNameDiscoverer);

965		return binderMethod;

966	}

982	@Nullable

983	private ModelAndView getModelAndView(ModelAndViewContainer mavContainer,

984			ModelFactory modelFactory, NativeWebRequest webRequest) throws Exception {

985

986		modelFactory.updateModel(webRequest, mavContainer);

987		if (mavContainer.isRequestHandled()) {

988			return null;

989		}

990		ModelMap model = mavContainer.getModel();

991		ModelAndView mav = new ModelAndView(mavContainer.getViewName(), model, mavContainer.getStatus());

992		if (!mavContainer.isViewReference()) {

993			mav.setView((View) mavContainer.getView());

994		}

995		if (model instanceof RedirectAttributes) {

996			Map<String, ?> flashAttributes = ((RedirectAttributes) model).getFlashAttributes();

997			HttpServletRequest request = webRequest.getNativeRequest(HttpServletRequest.class);

998			if (request != null) {

999				RequestContextUtils.getOutputFlashMap(request).putAll(flashAttributes);

1000			}

1001		}

1002		return mav;

1003	}
