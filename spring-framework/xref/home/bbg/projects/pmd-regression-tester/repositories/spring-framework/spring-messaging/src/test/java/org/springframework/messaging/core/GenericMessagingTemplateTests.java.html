68	@Test

69	public void sendWithTimeout() {

70		SubscribableChannel channel = mock(SubscribableChannel.class);

71		final AtomicReference<Message<?>> sent = new AtomicReference<>();

72		doAnswer(invocation -> {

73			sent.set(invocation.getArgument(0));

74			return true;

75		}).when(channel).send(any(Message.class), eq(30_000L));

76		Message<?> message = MessageBuilder.withPayload("request")

77				.setHeader(GenericMessagingTemplate.DEFAULT_SEND_TIMEOUT_HEADER, 30_000L)

78				.setHeader(GenericMessagingTemplate.DEFAULT_RECEIVE_TIMEOUT_HEADER, 1L)

79				.build();

80		this.template.send(channel, message);

81		verify(channel).send(any(Message.class), eq(30_000L));

82		assertNotNull(sent.get());

83		assertFalse(sent.get().getHeaders().containsKey(GenericMessagingTemplate.DEFAULT_SEND_TIMEOUT_HEADER));

84		assertFalse(sent.get().getHeaders().containsKey(GenericMessagingTemplate.DEFAULT_RECEIVE_TIMEOUT_HEADER));

85	}

121	@Test

122	public void sendAndReceiveTimeout() throws InterruptedException {

123		final AtomicReference<Throwable> failure = new AtomicReference<Throwable>();

124		final CountDownLatch latch = new CountDownLatch(1);

125

126		this.template.setReceiveTimeout(1);

127		this.template.setSendTimeout(30_000L);

128		this.template.setThrowExceptionOnLateReply(true);

129

130		SubscribableChannel channel = mock(SubscribableChannel.class);

131		MessageHandler handler = createLateReplier(latch, failure);

132		doAnswer(invocation -> {

133			this.executor.execute(() -> {

134				handler.handleMessage(invocation.getArgument(0));

135			});

136			return true;

137		}).when(channel).send(any(Message.class), anyLong());

138

139		assertNull(this.template.convertSendAndReceive(channel, "request", String.class));

140		assertTrue(latch.await(10_000, TimeUnit.MILLISECONDS));

141

142		Throwable ex = failure.get();

143		if (ex != null) {

144			throw new AssertionError(ex);

145		}

146		verify(channel).send(any(Message.class), eq(30_000L));

147	}

181	@Test

182	public void sendAndReceiveVariableTimeoutCustomHeaders() throws InterruptedException {

183		final AtomicReference<Throwable> failure = new AtomicReference<Throwable>();

184		final CountDownLatch latch = new CountDownLatch(1);

185

186		this.template.setSendTimeout(20_000);

187		this.template.setReceiveTimeout(10_000);

188		this.template.setThrowExceptionOnLateReply(true);

189		this.template.setSendTimeoutHeader("sto");

190		this.template.setReceiveTimeoutHeader("rto");

191

192		SubscribableChannel channel = mock(SubscribableChannel.class);

193		MessageHandler handler = createLateReplier(latch, failure);

194		doAnswer(invocation -> {

195			this.executor.execute(() -> {

196				handler.handleMessage(invocation.getArgument(0));

197			});

198			return true;

199		}).when(channel).send(any(Message.class), anyLong());

200

201		Message<?> message = MessageBuilder.withPayload("request")

202				.setHeader("sto", 30_000L)

203				.setHeader("rto", 1L)

204				.build();

205		assertNull(this.template.sendAndReceive(channel, message));

206		assertTrue(latch.await(10_000, TimeUnit.MILLISECONDS));

207

208		Throwable ex = failure.get();

209		if (ex != null) {

210			throw new AssertionError(ex);

211		}

212		verify(channel).send(any(Message.class), eq(30_000L));

213	}

215	private MessageHandler createLateReplier(final CountDownLatch latch, final AtomicReference<Throwable> failure) {

216		MessageHandler handler = message -> {

217			try {

218				Thread.sleep(500);

219				MessageChannel replyChannel = (MessageChannel) message.getHeaders().getReplyChannel();

220				replyChannel.send(new GenericMessage<>("response"));

221				failure.set(new IllegalStateException("Expected exception"));

222			}

223			catch (InterruptedException e) {

224				failure.set(e);

225			}

226			catch (MessageDeliveryException ex) {

227				String expected = "Reply message received but the receiving thread has exited due to a timeout";

228				String actual = ex.getMessage();

229				if (!expected.equals(actual)) {

230					failure.set(new IllegalStateException(

231							"Unexpected error: '" + actual + "'"));

232				}

233			}

234			finally {

235				latch.countDown();

236			}

237		};

238		return handler;

239	}

241	@Test

242	public void convertAndSendWithSimpMessageHeaders() {

243		MessageHeaderAccessor accessor = new MessageHeaderAccessor();

244		accessor.setHeader("key", "value");

245		accessor.setLeaveMutable(true);

246		MessageHeaders headers = accessor.getMessageHeaders();

247

248		this.template.convertAndSend("channel", "data", headers);

249		List<Message<byte[]>> messages = this.messageChannel.getMessages();

250		Message<byte[]> message = messages.get(0);

251

252		assertSame(headers, message.getHeaders());

253		assertFalse(accessor.isMutable());

254	}
