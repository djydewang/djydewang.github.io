86/**

87 * Parses a {@link Configuration} class definition, populating a collection of

88 * {@link ConfigurationClass} objects (parsing a single Configuration class may result in

89 * any number of ConfigurationClass objects because one Configuration class may import

90 * another using the {@link Import} annotation).

91 *

92 * <p>This class helps separate the concern of parsing the structure of a Configuration

93 * class from the concern of registering BeanDefinition objects based on the

94 * content of that model (with the exception of {@code @ComponentScan} annotations which

95 * need to be registered immediately).

96 *

97 * <p>This ASM-based implementation avoids reflection and eager class loading in order to

98 * interoperate effectively with lazy class loading in a Spring ApplicationContext.

99 *

100 * @author Chris Beams

101 * @author Juergen Hoeller

102 * @author Phillip Webb

103 * @author Sam Brannen

104 * @author Stephane Nicoll

105 * @since 3.0

106 * @see ConfigurationClassBeanDefinitionReader

107 */

108class ConfigurationClassParser {

163	public void parse(Set<BeanDefinitionHolder> configCandidates) {

164		this.deferredImportSelectors = new LinkedList<>();

165

166		for (BeanDefinitionHolder holder : configCandidates) {

167			BeanDefinition bd = holder.getBeanDefinition();

168			try {

169				if (bd instanceof AnnotatedBeanDefinition) {

170					parse(((AnnotatedBeanDefinition) bd).getMetadata(), holder.getBeanName());

171				}

172				else if (bd instanceof AbstractBeanDefinition && ((AbstractBeanDefinition) bd).hasBeanClass()) {

173					parse(((AbstractBeanDefinition) bd).getBeanClass(), holder.getBeanName());

174				}

175				else {

176					parse(bd.getBeanClassName(), holder.getBeanName());

177				}

178			}

179			catch (BeanDefinitionStoreException ex) {

180				throw ex;

181			}

182			catch (Throwable ex) {

183				throw new BeanDefinitionStoreException(

184						"Failed to parse configuration class [" + bd.getBeanClassName() + "]", ex);

185			}

186		}

187

188		processDeferredImportSelectors();

189	}

190

191	protected final void parse(@Nullable String className, String beanName) throws IOException {

192		Assert.notNull(className, "No bean class name for configuration class bean definition");

193		MetadataReader reader = this.metadataReaderFactory.getMetadataReader(className);

194		processConfigurationClass(new ConfigurationClass(reader, beanName));

195	}

196

197	protected final void parse(Class<?> clazz, String beanName) throws IOException {

198		processConfigurationClass(new ConfigurationClass(clazz, beanName));

199	}

200

201	protected final void parse(AnnotationMetadata metadata, String beanName) throws IOException {

202		processConfigurationClass(new ConfigurationClass(metadata, beanName));

203	}

220	protected void processConfigurationClass(ConfigurationClass configClass) throws IOException {

221		if (this.conditionEvaluator.shouldSkip(configClass.getMetadata(), ConfigurationPhase.PARSE_CONFIGURATION)) {

222			return;

223		}

224

225		ConfigurationClass existingClass = this.configurationClasses.get(configClass);

226		if (existingClass != null) {

227			if (configClass.isImported()) {

228				if (existingClass.isImported()) {

229					existingClass.mergeImportedBy(configClass);

230				}

231				// Otherwise ignore new imported config class; existing non-imported class overrides it.

232				return;

233			}

234			else {

235				// Explicit bean definition found, probably replacing an import.

236				// Let's remove the old one and go with the new one.

237				this.configurationClasses.remove(configClass);

238				this.knownSuperclasses.values().removeIf(configClass::equals);

239			}

240		}

241

242		// Recursively process the configuration class and its superclass hierarchy.

243		SourceClass sourceClass = asSourceClass(configClass);

244		do {

245			sourceClass = doProcessConfigurationClass(configClass, sourceClass);

246		}

247		while (sourceClass != null);

248

249		this.configurationClasses.put(configClass, configClass);

250	}

252	/**

253	 * Apply processing and build a complete {@link ConfigurationClass} by reading the

254	 * annotations, members and methods from the source class. This method can be called

255	 * multiple times as relevant sources are discovered.

256	 * @param configClass the configuration class being build

257	 * @param sourceClass a source class

258	 * @return the superclass, or {@code null} if none found or previously processed

259	 */

260	@Nullable

261	protected final SourceClass doProcessConfigurationClass(ConfigurationClass configClass, SourceClass sourceClass)

262			throws IOException {

263

264		// Recursively process any member (nested) classes first

265		processMemberClasses(configClass, sourceClass);

266

267		// Process any @PropertySource annotations

268		for (AnnotationAttributes propertySource : AnnotationConfigUtils.attributesForRepeatable(

269				sourceClass.getMetadata(), PropertySources.class,

270				org.springframework.context.annotation.PropertySource.class)) {

271			if (this.environment instanceof ConfigurableEnvironment) {

272				processPropertySource(propertySource);

273			}

274			else {

275				logger.warn("Ignoring @PropertySource annotation on [" + sourceClass.getMetadata().getClassName() +

276						"]. Reason: Environment must implement ConfigurableEnvironment");

277			}

278		}

279

280		// Process any @ComponentScan annotations

281		Set<AnnotationAttributes> componentScans = AnnotationConfigUtils.attributesForRepeatable(

282				sourceClass.getMetadata(), ComponentScans.class, ComponentScan.class);

283		if (!componentScans.isEmpty() &&

284				!this.conditionEvaluator.shouldSkip(sourceClass.getMetadata(), ConfigurationPhase.REGISTER_BEAN)) {

285			for (AnnotationAttributes componentScan : componentScans) {

286				// The config class is annotated with @ComponentScan -> perform the scan immediately

287				Set<BeanDefinitionHolder> scannedBeanDefinitions =

288						this.componentScanParser.parse(componentScan, sourceClass.getMetadata().getClassName());

289				// Check the set of scanned definitions for any further config classes and parse recursively if needed

290				for (BeanDefinitionHolder holder : scannedBeanDefinitions) {

291					if (ConfigurationClassUtils.checkConfigurationClassCandidate(

292							holder.getBeanDefinition(), this.metadataReaderFactory)) {

293						parse(holder.getBeanDefinition().getBeanClassName(), holder.getBeanName());

294					}

295				}

296			}

297		}

298

299		// Process any @Import annotations

300		processImports(configClass, sourceClass, getImports(sourceClass), true);

301

302		// Process any @ImportResource annotations

303		AnnotationAttributes importResource =

304				AnnotationConfigUtils.attributesFor(sourceClass.getMetadata(), ImportResource.class);

305		if (importResource != null) {

306			String[] resources = importResource.getStringArray("locations");

307			Class<? extends BeanDefinitionReader> readerClass = importResource.getClass("reader");

308			for (String resource : resources) {

309				String resolvedResource = this.environment.resolveRequiredPlaceholders(resource);

310				configClass.addImportedResource(resolvedResource, readerClass);

311			}

312		}

313

314		// Process individual @Bean methods

315		Set<MethodMetadata> beanMethods = retrieveBeanMethodMetadata(sourceClass);

316		for (MethodMetadata methodMetadata : beanMethods) {

317			configClass.addBeanMethod(new BeanMethod(methodMetadata, configClass));

318		}

319

320		// Process default methods on interfaces

321		processInterfaces(configClass, sourceClass);

322

323		// Process superclass, if any

324		if (sourceClass.getMetadata().hasSuperClass()) {

325			String superclass = sourceClass.getMetadata().getSuperClassName();

326			if (superclass != null && !superclass.startsWith("java") &&

327					!this.knownSuperclasses.containsKey(superclass)) {

328				this.knownSuperclasses.put(superclass, configClass);

329				// Superclass found, return its annotation metadata and recurse

330				return sourceClass.getSuperClass();

331			}

332		}

333

334		// No superclass -> processing is complete

335		return null;

336	}

338	/**

339	 * Register member (nested) classes that happen to be configuration classes themselves.

340	 */

341	private void processMemberClasses(ConfigurationClass configClass, SourceClass sourceClass) throws IOException {

342		Collection<SourceClass> memberClasses = sourceClass.getMemberClasses();

343		if (!memberClasses.isEmpty()) {

344			List<SourceClass> candidates = new ArrayList<>(memberClasses.size());

345			for (SourceClass memberClass : memberClasses) {

346				if (ConfigurationClassUtils.isConfigurationCandidate(memberClass.getMetadata()) &&

347						!memberClass.getMetadata().getClassName().equals(configClass.getMetadata().getClassName())) {

348					candidates.add(memberClass);

349				}

350			}

351			OrderComparator.sort(candidates);

352			for (SourceClass candidate : candidates) {

353				if (this.importStack.contains(configClass)) {

354					this.problemReporter.error(new CircularImportProblem(configClass, this.importStack));

355				}

356				else {

357					this.importStack.push(configClass);

358					try {

359						processConfigurationClass(candidate.asConfigClass(configClass));

360					}

361					finally {

362						this.importStack.pop();

363					}

364				}

365			}

366		}

367	}

369	/**

370	 * Register default methods on interfaces implemented by the configuration class.

371	 */

372	private void processInterfaces(ConfigurationClass configClass, SourceClass sourceClass) throws IOException {

373		for (SourceClass ifc : sourceClass.getInterfaces()) {

374			Set<MethodMetadata> beanMethods = retrieveBeanMethodMetadata(ifc);

375			for (MethodMetadata methodMetadata : beanMethods) {

376				if (!methodMetadata.isAbstract()) {

377					// A default method or other concrete method on a Java 8+ interface...

378					configClass.addBeanMethod(new BeanMethod(methodMetadata, configClass));

379				}

380			}

381			processInterfaces(configClass, ifc);

382		}

383	}

385	/**

386	 * Retrieve the metadata for all <code>@Bean</code> methods.

387	 */

388	private Set<MethodMetadata> retrieveBeanMethodMetadata(SourceClass sourceClass) {

389		AnnotationMetadata original = sourceClass.getMetadata();

390		Set<MethodMetadata> beanMethods = original.getAnnotatedMethods(Bean.class.getName());

391		if (beanMethods.size() > 1 && original instanceof StandardAnnotationMetadata) {

392			// Try reading the class file via ASM for deterministic declaration order...

393			// Unfortunately, the JVM's standard reflection returns methods in arbitrary

394			// order, even between different runs of the same application on the same JVM.

395			try {

396				AnnotationMetadata asm =

397						this.metadataReaderFactory.getMetadataReader(original.getClassName()).getAnnotationMetadata();

398				Set<MethodMetadata> asmMethods = asm.getAnnotatedMethods(Bean.class.getName());

399				if (asmMethods.size() >= beanMethods.size()) {

400					Set<MethodMetadata> selectedMethods = new LinkedHashSet<>(asmMethods.size());

401					for (MethodMetadata asmMethod : asmMethods) {

402						for (MethodMetadata beanMethod : beanMethods) {

403							if (beanMethod.getMethodName().equals(asmMethod.getMethodName())) {

404								selectedMethods.add(beanMethod);

405								break;

406							}

407						}

408					}

409					if (selectedMethods.size() == beanMethods.size()) {

410						// All reflection-detected methods found in ASM method set -> proceed

411						beanMethods = selectedMethods;

412					}

413				}

414			}

415			catch (IOException ex) {

416				logger.debug("Failed to read class file via ASM for determining @Bean method order", ex);

417				// No worries, let's continue with the reflection metadata we started with...

418			}

419		}

420		return beanMethods;

421	}

424	/**

425	 * Process the given <code>@PropertySource</code> annotation metadata.

426	 * @param propertySource metadata for the <code>@PropertySource</code> annotation found

427	 * @throws IOException if loading a property source failed

428	 */

429	private void processPropertySource(AnnotationAttributes propertySource) throws IOException {

430		String name = propertySource.getString("name");

431		if (!StringUtils.hasLength(name)) {

432			name = null;

433		}

434		String encoding = propertySource.getString("encoding");

435		if (!StringUtils.hasLength(encoding)) {

436			encoding = null;

437		}

438		String[] locations = propertySource.getStringArray("value");

439		Assert.isTrue(locations.length > 0, "At least one @PropertySource(value) location is required");

440		boolean ignoreResourceNotFound = propertySource.getBoolean("ignoreResourceNotFound");

441

442		Class<? extends PropertySourceFactory> factoryClass = propertySource.getClass("factory");

443		PropertySourceFactory factory = (factoryClass == PropertySourceFactory.class ?

444				DEFAULT_PROPERTY_SOURCE_FACTORY : BeanUtils.instantiateClass(factoryClass));

445

446		for (String location : locations) {

447			try {

448				String resolvedLocation = this.environment.resolveRequiredPlaceholders(location);

449				Resource resource = this.resourceLoader.getResource(resolvedLocation);

450				addPropertySource(factory.createPropertySource(name, new EncodedResource(resource, encoding)));

451			}

452			catch (IllegalArgumentException | FileNotFoundException | UnknownHostException ex) {

453				// Placeholders not resolvable or resource not found when trying to open it

454				if (ignoreResourceNotFound) {

455					if (logger.isInfoEnabled()) {

456						logger.info("Properties location [" + location + "] not resolvable: " + ex.getMessage());

457					}

458				}

459				else {

460					throw ex;

461				}

462			}

463		}

464	}

466	private void addPropertySource(PropertySource<?> propertySource) {

467		String name = propertySource.getName();

468		MutablePropertySources propertySources = ((ConfigurableEnvironment) this.environment).getPropertySources();

469

470		if (this.propertySourceNames.contains(name)) {

471			// We've already added a version, we need to extend it

472			PropertySource<?> existing = propertySources.get(name);

473			if (existing != null) {

474				PropertySource<?> newSource = (propertySource instanceof ResourcePropertySource ?

475						((ResourcePropertySource) propertySource).withResourceName() : propertySource);

476				if (existing instanceof CompositePropertySource) {

477					((CompositePropertySource) existing).addFirstPropertySource(newSource);

478				}

479				else {

480					if (existing instanceof ResourcePropertySource) {

481						existing = ((ResourcePropertySource) existing).withResourceName();

482					}

483					CompositePropertySource composite = new CompositePropertySource(name);

484					composite.addPropertySource(newSource);

485					composite.addPropertySource(existing);

486					propertySources.replace(name, composite);

487				}

488				return;

489			}

490		}

491

492		if (this.propertySourceNames.isEmpty()) {

493			propertySources.addLast(propertySource);

494		}

495		else {

496			String firstProcessed = this.propertySourceNames.get(this.propertySourceNames.size() - 1);

497			propertySources.addBefore(firstProcessed, propertySource);

498		}

499		this.propertySourceNames.add(name);

500	}

503	/**

504	 * Returns {@code @Import} class, considering all meta-annotations.

505	 */

506	private Set<SourceClass> getImports(SourceClass sourceClass) throws IOException {

507		Set<SourceClass> imports = new LinkedHashSet<>();

508		Set<SourceClass> visited = new LinkedHashSet<>();

509		collectImports(sourceClass, imports, visited);

510		return imports;

511	}

513	/**

514	 * Recursively collect all declared {@code @Import} values. Unlike most

515	 * meta-annotations it is valid to have several {@code @Import}s declared with

516	 * different values; the usual process of returning values from the first

517	 * meta-annotation on a class is not sufficient.

518	 * <p>For example, it is common for a {@code @Configuration} class to declare direct

519	 * {@code @Import}s in addition to meta-imports originating from an {@code @Enable}

520	 * annotation.

521	 * @param sourceClass the class to search

522	 * @param imports the imports collected so far

523	 * @param visited used to track visited classes to prevent infinite recursion

524	 * @throws IOException if there is any problem reading metadata from the named class

525	 */

526	private void collectImports(SourceClass sourceClass, Set<SourceClass> imports, Set<SourceClass> visited)

527			throws IOException {

528

529		if (visited.add(sourceClass)) {

530			for (SourceClass annotation : sourceClass.getAnnotations()) {

531				String annName = annotation.getMetadata().getClassName();

532				if (!annName.startsWith("java") && !annName.equals(Import.class.getName())) {

533					collectImports(annotation, imports, visited);

534				}

535			}

536			imports.addAll(sourceClass.getAnnotationAttributes(Import.class.getName(), "value"));

537		}

538	}

540	private void processDeferredImportSelectors() {

541		List<DeferredImportSelectorHolder> deferredImports = this.deferredImportSelectors;

542		this.deferredImportSelectors = null;

543		if (deferredImports == null) {

544			return;

545		}

546

547		deferredImports.sort(DEFERRED_IMPORT_COMPARATOR);

548		Map<Object, DeferredImportSelectorGrouping> groupings = new LinkedHashMap<>();

549		Map<AnnotationMetadata, ConfigurationClass> configurationClasses = new HashMap<>();

550		for (DeferredImportSelectorHolder deferredImport : deferredImports) {

551			Class<? extends Group> group = deferredImport.getImportSelector().getImportGroup();

552			DeferredImportSelectorGrouping grouping = groupings.computeIfAbsent(

553					(group == null ? deferredImport : group),

554					(key) -> new DeferredImportSelectorGrouping(createGroup(group)));

555			grouping.add(deferredImport);

556			configurationClasses.put(deferredImport.getConfigurationClass().getMetadata(),

557					deferredImport.getConfigurationClass());

558		}

559		for (DeferredImportSelectorGrouping grouping : groupings.values()) {

560			grouping.getImports().forEach((entry) -> {

561				ConfigurationClass configurationClass = configurationClasses.get(

562						entry.getMetadata());

563				try {

564					processImports(configurationClass, asSourceClass(configurationClass),

565							asSourceClasses(entry.getImportClassName()), false);

566				}

567				catch (BeanDefinitionStoreException ex) {

568					throw ex;

569				}

570				catch (Throwable ex) {

571					throw new BeanDefinitionStoreException(

572							"Failed to process import candidates for configuration class [" +

573									configurationClass.getMetadata().getClassName() + "]", ex);

574				}

575			});

576		}

577	}

579	private Group createGroup(@Nullable Class<? extends Group> type) {

580		Class<? extends Group> effectiveType = (type != null ? type

581				: DefaultDeferredImportSelectorGroup.class);

582		Group group = BeanUtils.instantiateClass(effectiveType);

583		ParserStrategyUtils.invokeAwareMethods(group,

584				ConfigurationClassParser.this.environment,

585				ConfigurationClassParser.this.resourceLoader,

586				ConfigurationClassParser.this.registry);

587		return group;

588	}

590	private void processImports(ConfigurationClass configClass, SourceClass currentSourceClass,

591			Collection<SourceClass> importCandidates, boolean checkForCircularImports) throws IOException {

592

593		if (importCandidates.isEmpty()) {

594			return;

595		}

596

597		if (checkForCircularImports && isChainedImportOnStack(configClass)) {

598			this.problemReporter.error(new CircularImportProblem(configClass, this.importStack));

599		}

600		else {

601			this.importStack.push(configClass);

602			try {

603				for (SourceClass candidate : importCandidates) {

604					if (candidate.isAssignable(ImportSelector.class)) {

605						// Candidate class is an ImportSelector -> delegate to it to determine imports

606						Class<?> candidateClass = candidate.loadClass();

607						ImportSelector selector = BeanUtils.instantiateClass(candidateClass, ImportSelector.class);

608						ParserStrategyUtils.invokeAwareMethods(

609								selector, this.environment, this.resourceLoader, this.registry);

610						if (this.deferredImportSelectors != null && selector instanceof DeferredImportSelector) {

611							this.deferredImportSelectors.add(

612									new DeferredImportSelectorHolder(configClass, (DeferredImportSelector) selector));

613						}

614						else {

615							String[] importClassNames = selector.selectImports(currentSourceClass.getMetadata());

616							Collection<SourceClass> importSourceClasses = asSourceClasses(importClassNames);

617							processImports(configClass, currentSourceClass, importSourceClasses, false);

618						}

619					}

620					else if (candidate.isAssignable(ImportBeanDefinitionRegistrar.class)) {

621						// Candidate class is an ImportBeanDefinitionRegistrar ->

622						// delegate to it to register additional bean definitions

623						Class<?> candidateClass = candidate.loadClass();

624						ImportBeanDefinitionRegistrar registrar =

625								BeanUtils.instantiateClass(candidateClass, ImportBeanDefinitionRegistrar.class);

626						ParserStrategyUtils.invokeAwareMethods(

627								registrar, this.environment, this.resourceLoader, this.registry);

628						configClass.addImportBeanDefinitionRegistrar(registrar, currentSourceClass.getMetadata());

629					}

630					else {

631						// Candidate class not an ImportSelector or ImportBeanDefinitionRegistrar ->

632						// process it as an @Configuration class

633						this.importStack.registerImport(

634								currentSourceClass.getMetadata(), candidate.getMetadata().getClassName());

635						processConfigurationClass(candidate.asConfigClass(configClass));

636					}

637				}

638			}

639			catch (BeanDefinitionStoreException ex) {

640				throw ex;

641			}

642			catch (Throwable ex) {

643				throw new BeanDefinitionStoreException(

644						"Failed to process import candidates for configuration class [" +

645						configClass.getMetadata().getClassName() + "]", ex);

646			}

647			finally {

648				this.importStack.pop();

649			}

650		}

651	}

653	private boolean isChainedImportOnStack(ConfigurationClass configClass) {

654		if (this.importStack.contains(configClass)) {

655			String configClassName = configClass.getMetadata().getClassName();

656			AnnotationMetadata importingClass = this.importStack.getImportingClassFor(configClassName);

657			while (importingClass != null) {

658				if (configClassName.equals(importingClass.getClassName())) {

659					return true;

660				}

661				importingClass = this.importStack.getImportingClassFor(importingClass.getClassName());

662			}

663		}

664		return false;

665	}

667	ImportRegistry getImportRegistry() {

668		return this.importStack;

669	}

673	 * Factory method to obtain a {@link SourceClass} from a {@link ConfigurationClass}.

674	 */

675	private SourceClass asSourceClass(ConfigurationClass configurationClass) throws IOException {

676		AnnotationMetadata metadata = configurationClass.getMetadata();

677		if (metadata instanceof StandardAnnotationMetadata) {

678			return asSourceClass(((StandardAnnotationMetadata) metadata).getIntrospectedClass());

679		}

680		return asSourceClass(metadata.getClassName());

681	}

683	/**

684	 * Factory method to obtain a {@link SourceClass} from a {@link Class}.

685	 */

686	SourceClass asSourceClass(@Nullable Class<?> classType) throws IOException {

687		if (classType == null) {

688			return new SourceClass(Object.class);

689		}

690		try {

691			// Sanity test that we can reflectively read annotations,

692			// including Class attributes; if not -> fall back to ASM

693			for (Annotation ann : classType.getAnnotations()) {

694				AnnotationUtils.validateAnnotation(ann);

695			}

696			return new SourceClass(classType);

697		}

698		catch (Throwable ex) {

699			// Enforce ASM via class name resolution

700			return asSourceClass(classType.getName());

701		}

702	}

704	/**

705	 * Factory method to obtain {@link SourceClass}s from class names.

706	 */

707	private Collection<SourceClass> asSourceClasses(String... classNames) throws IOException {

708		List<SourceClass> annotatedClasses = new ArrayList<>(classNames.length);

709		for (String className : classNames) {

710			annotatedClasses.add(asSourceClass(className));

711		}

712		return annotatedClasses;

713	}

714

715	/**

716	 * Factory method to obtain a {@link SourceClass} from a class name.

717	 */

718	SourceClass asSourceClass(@Nullable String className) throws IOException {

719		if (className == null) {

720			return new SourceClass(Object.class);

721		}

722		if (className.startsWith("java")) {

723			// Never use ASM for core java types

724			try {

725				return new SourceClass(ClassUtils.forName(className, this.resourceLoader.getClassLoader()));

726			}

727			catch (ClassNotFoundException ex) {

728				throw new NestedIOException("Failed to load class [" + className + "]", ex);

729			}

730		}

731		return new SourceClass(this.metadataReaderFactory.getMetadataReader(className));

732	}

736	private static class ImportStack extends ArrayDeque<ConfigurationClass> implements ImportRegistry {

737

738		private final MultiValueMap<String, AnnotationMetadata> imports = new LinkedMultiValueMap<>();

739

740		public void registerImport(AnnotationMetadata importingClass, String importedClass) {

741			this.imports.add(importedClass, importingClass);

742		}

743

744		@Override

745		@Nullable

746		public AnnotationMetadata getImportingClassFor(String importedClass) {

747			return CollectionUtils.lastElement(this.imports.get(importedClass));

748		}

749

750		@Override

751		public void removeImportingClass(String importingClass) {

752			for (List<AnnotationMetadata> list : this.imports.values()) {

753				for (Iterator<AnnotationMetadata> iterator = list.iterator(); iterator.hasNext();) {

754					if (iterator.next().getClassName().equals(importingClass)) {

755						iterator.remove();

756						break;

757					}

758				}

759			}

760		}

761

762		/**

763		 * Given a stack containing (in order)

764		 * <ul>

765		 * <li>com.acme.Foo</li>

766		 * <li>com.acme.Bar</li>

767		 * <li>com.acme.Baz</li>

768		 * </ul>

769		 * return "[Foo->Bar->Baz]".

770		 */

771		@Override

772		public String toString() {

773			StringBuilder builder = new StringBuilder("[");

774			Iterator<ConfigurationClass> iterator = iterator();

775			while (iterator.hasNext()) {

776				builder.append(iterator.next().getSimpleName());

777				if (iterator.hasNext()) {

778					builder.append("->");

779				}

780			}

781			return builder.append(']').toString();

782		}

853	/**

854	 * Simple wrapper that allows annotated source classes to be dealt with

855	 * in a uniform manner, regardless of how they are loaded.

856	 */

857	private class SourceClass implements Ordered {

858

859		private final Object source;  // Class or MetadataReader

860

861		private final AnnotationMetadata metadata;

862

863		public SourceClass(Object source) {

864			this.source = source;

865			if (source instanceof Class) {

866				this.metadata = new StandardAnnotationMetadata((Class<?>) source, true);

867			}

868			else {

869				this.metadata = ((MetadataReader) source).getAnnotationMetadata();

870			}

871		}

872

873		public final AnnotationMetadata getMetadata() {

874			return this.metadata;

875		}

876

877		@Override

878		public int getOrder() {

879			Integer order = ConfigurationClassUtils.getOrder(this.metadata);

880			return (order != null ? order : Ordered.LOWEST_PRECEDENCE);

881		}

882

883		public Class<?> loadClass() throws ClassNotFoundException {

884			if (this.source instanceof Class) {

885				return (Class<?>) this.source;

886			}

887			String className = ((MetadataReader) this.source).getClassMetadata().getClassName();

888			return ClassUtils.forName(className, resourceLoader.getClassLoader());

889		}

890

891		public boolean isAssignable(Class<?> clazz) throws IOException {

892			if (this.source instanceof Class) {

893				return clazz.isAssignableFrom((Class<?>) this.source);

894			}

895			return new AssignableTypeFilter(clazz).match((MetadataReader) this.source, metadataReaderFactory);

896		}

897

898		public ConfigurationClass asConfigClass(ConfigurationClass importedBy) throws IOException {

899			if (this.source instanceof Class) {

900				return new ConfigurationClass((Class<?>) this.source, importedBy);

901			}

902			return new ConfigurationClass((MetadataReader) this.source, importedBy);

903		}

904

905		public Collection<SourceClass> getMemberClasses() throws IOException {

906			Object sourceToProcess = this.source;

907			if (sourceToProcess instanceof Class) {

908				Class<?> sourceClass = (Class<?>) sourceToProcess;

909				try {

910					Class<?>[] declaredClasses = sourceClass.getDeclaredClasses();

911					List<SourceClass> members = new ArrayList<>(declaredClasses.length);

912					for (Class<?> declaredClass : declaredClasses) {

913						members.add(asSourceClass(declaredClass));

914					}

915					return members;

916				}

917				catch (NoClassDefFoundError err) {

918					// getDeclaredClasses() failed because of non-resolvable dependencies

919					// -> fall back to ASM below

920					sourceToProcess = metadataReaderFactory.getMetadataReader(sourceClass.getName());

921				}

922			}

923

924			// ASM-based resolution - safe for non-resolvable classes as well

925			MetadataReader sourceReader = (MetadataReader) sourceToProcess;

926			String[] memberClassNames = sourceReader.getClassMetadata().getMemberClassNames();

927			List<SourceClass> members = new ArrayList<>(memberClassNames.length);

928			for (String memberClassName : memberClassNames) {

929				try {

930					members.add(asSourceClass(memberClassName));

931				}

932				catch (IOException ex) {

933					// Let's skip it if it's not resolvable - we're just looking for candidates

934					if (logger.isDebugEnabled()) {

935						logger.debug("Failed to resolve member class [" + memberClassName +

936								"] - not considering it as a configuration class candidate");

937					}

938				}

939			}

940			return members;

941		}

942

943		public SourceClass getSuperClass() throws IOException {

944			if (this.source instanceof Class) {

945				return asSourceClass(((Class<?>) this.source).getSuperclass());

946			}

947			return asSourceClass(((MetadataReader) this.source).getClassMetadata().getSuperClassName());

948		}

949

950		public Set<SourceClass> getInterfaces() throws IOException {

951			Set<SourceClass> result = new LinkedHashSet<>();

952			if (this.source instanceof Class) {

953				Class<?> sourceClass = (Class<?>) this.source;

954				for (Class<?> ifcClass : sourceClass.getInterfaces()) {

955					result.add(asSourceClass(ifcClass));

956				}

957			}

958			else {

959				for (String className : this.metadata.getInterfaceNames()) {

960					result.add(asSourceClass(className));

961				}

962			}

963			return result;

964		}

965

966		public Set<SourceClass> getAnnotations() throws IOException {

967			Set<SourceClass> result = new LinkedHashSet<>();

968			for (String className : this.metadata.getAnnotationTypes()) {

969				try {

970					result.add(getRelated(className));

971				}

972				catch (Throwable ex) {

973					// An annotation not present on the classpath is being ignored

974					// by the JVM's class loading -> ignore here as well.

975				}

976			}

977			return result;

978		}

979

980		public Collection<SourceClass> getAnnotationAttributes(String annType, String attribute) throws IOException {

981			Map<String, Object> annotationAttributes = this.metadata.getAnnotationAttributes(annType, true);

982			if (annotationAttributes == null || !annotationAttributes.containsKey(attribute)) {

983				return Collections.emptySet();

984			}

985			String[] classNames = (String[]) annotationAttributes.get(attribute);

986			Set<SourceClass> result = new LinkedHashSet<>();

987			for (String className : classNames) {

988				result.add(getRelated(className));

989			}

990			return result;

991		}

992

993		private SourceClass getRelated(String className) throws IOException {

994			if (this.source instanceof Class) {

995				try {

996					Class<?> clazz = ((Class<?>) this.source).getClassLoader().loadClass(className);

997					return asSourceClass(clazz);

998				}

999				catch (ClassNotFoundException ex) {

1000					// Ignore -> fall back to ASM next, except for core java types.

1001					if (className.startsWith("java")) {

1002						throw new NestedIOException("Failed to load class [" + className + "]", ex);

1003					}

1004					return new SourceClass(metadataReaderFactory.getMetadataReader(className));

1005				}

1006			}

1007			return asSourceClass(className);

1008		}

1009

1010		@Override

1011		public boolean equals(Object other) {

1012			return (this == other || (other instanceof SourceClass &&

1013					this.metadata.getClassName().equals(((SourceClass) other).metadata.getClassName())));

1014		}

1015

1016		@Override

1017		public int hashCode() {

1018			return this.metadata.getClassName().hashCode();

1019		}

1020

1021		@Override

1022		public String toString() {

1023			return this.metadata.getClassName();

1024		}

1025	}
