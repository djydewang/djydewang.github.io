173	/**

174	 * Specify the portion of the requestURI that represents the context path.

175	 * The context path, if specified, must match to the start of the request URI.

176	 * <p>In most cases, tests can be written by omitting the context path from

177	 * the requestURI. This is because most applications don't actually depend

178	 * on the name under which they're deployed. If specified here, the context

179	 * path must start with a "/" and must not end with a "/".

180	 * @see javax.servlet.http.HttpServletRequest#getContextPath()

181	 */

182	public MockHttpServletRequestBuilder contextPath(String contextPath) {

183		if (StringUtils.hasText(contextPath)) {

184			Assert.isTrue(contextPath.startsWith("/"), "Context path must start with a '/'");

185			Assert.isTrue(!contextPath.endsWith("/"), "Context path must not end with a '/'");

186		}

187		this.contextPath = contextPath;

188		return this;

189	}

191	/**

192	 * Specify the portion of the requestURI that represents the path to which

193	 * the Servlet is mapped. This is typically a portion of the requestURI

194	 * after the context path.

195	 * <p>In most cases, tests can be written by omitting the servlet path from

196	 * the requestURI. This is because most applications don't actually depend

197	 * on the prefix to which a servlet is mapped. For example if a Servlet is

198	 * mapped to {@code "/main/*"}, tests can be written with the requestURI

199	 * {@code "/accounts/1"} as opposed to {@code "/main/accounts/1"}.

200	 * If specified here, the servletPath must start with a "/" and must not

201	 * end with a "/".

202	 * @see javax.servlet.http.HttpServletRequest#getServletPath()

203	 */

204	public MockHttpServletRequestBuilder servletPath(String servletPath) {

205		if (StringUtils.hasText(servletPath)) {

206			Assert.isTrue(servletPath.startsWith("/"), "Servlet path must start with a '/'");

207			Assert.isTrue(!servletPath.endsWith("/"), "Servlet path must not end with a '/'");

208		}

209		this.servletPath = servletPath;

210		return this;

211	}

491	/**

492	 * Merges the properties of the "parent" RequestBuilder accepting values

493	 * only if not already set in "this" instance.

494	 * @param parent the parent {@code RequestBuilder} to inherit properties from

495	 * @return the result of the merge

496	 */

497	@Override

498	public Object merge(@Nullable Object parent) {

499		if (parent == null) {

500			return this;

501		}

502		if (!(parent instanceof MockHttpServletRequestBuilder)) {

503			throw new IllegalArgumentException("Cannot merge with [" + parent.getClass().getName() + "]");

504		}

505		MockHttpServletRequestBuilder parentBuilder = (MockHttpServletRequestBuilder) parent;

506

507		if (!StringUtils.hasText(this.contextPath)) {

508			this.contextPath = parentBuilder.contextPath;

509		}

510		if (!StringUtils.hasText(this.servletPath)) {

511			this.servletPath = parentBuilder.servletPath;

512		}

513		if ("".equals(this.pathInfo)) {

514			this.pathInfo = parentBuilder.pathInfo;

515		}

516

517		if (this.secure == null) {

518			this.secure = parentBuilder.secure;

519		}

520		if (this.principal == null) {

521			this.principal = parentBuilder.principal;

522		}

523		if (this.session == null) {

524			this.session = parentBuilder.session;

525		}

526

527		if (this.characterEncoding == null) {

528			this.characterEncoding = parentBuilder.characterEncoding;

529		}

530		if (this.content == null) {

531			this.content = parentBuilder.content;

532		}

533		if (this.contentType == null) {

534			this.contentType = parentBuilder.contentType;

535		}

536

537		for (String headerName : parentBuilder.headers.keySet()) {

538			if (!this.headers.containsKey(headerName)) {

539				this.headers.put(headerName, parentBuilder.headers.get(headerName));

540			}

541		}

542		for (String paramName : parentBuilder.parameters.keySet()) {

543			if (!this.parameters.containsKey(paramName)) {

544				this.parameters.put(paramName, parentBuilder.parameters.get(paramName));

545			}

546		}

547		for (Cookie cookie : parentBuilder.cookies) {

548			if (!containsCookie(cookie)) {

549				this.cookies.add(cookie);

550			}

551		}

552		for (Locale locale : parentBuilder.locales) {

553			if (!this.locales.contains(locale)) {

554				this.locales.add(locale);

555			}

556		}

557

558		for (String attributeName : parentBuilder.requestAttributes.keySet()) {

559			if (!this.requestAttributes.containsKey(attributeName)) {

560				this.requestAttributes.put(attributeName, parentBuilder.requestAttributes.get(attributeName));

561			}

562		}

563		for (String attributeName : parentBuilder.sessionAttributes.keySet()) {

564			if (!this.sessionAttributes.containsKey(attributeName)) {

565				this.sessionAttributes.put(attributeName, parentBuilder.sessionAttributes.get(attributeName));

566			}

567		}

568		for (String attributeName : parentBuilder.flashAttributes.keySet()) {

569			if (!this.flashAttributes.containsKey(attributeName)) {

570				this.flashAttributes.put(attributeName, parentBuilder.flashAttributes.get(attributeName));

571			}

572		}

573

574		this.postProcessors.addAll(0, parentBuilder.postProcessors);

575

576		return this;

577	}

579	private boolean containsCookie(Cookie cookie) {

580		for (Cookie cookieToCheck : this.cookies) {

581			if (ObjectUtils.nullSafeEquals(cookieToCheck.getName(), cookie.getName())) {

582				return true;

583			}

584		}

585		return false;

586	}

588	/**

589	 * Build a {@link MockHttpServletRequest}.

590	 */

591	@Override

592	public final MockHttpServletRequest buildRequest(ServletContext servletContext) {

593		MockHttpServletRequest request = createServletRequest(servletContext);

594

595		request.setAsyncSupported(true);

596		request.setMethod(this.method);

597

598		String requestUri = this.url.getRawPath();

599		request.setRequestURI(requestUri);

600

601		if (this.url.getScheme() != null) {

602			request.setScheme(this.url.getScheme());

603		}

604		if (this.url.getHost() != null) {

605			request.setServerName(this.url.getHost());

606		}

607		if (this.url.getPort() != -1) {

608			request.setServerPort(this.url.getPort());

609		}

610

611		updatePathRequestProperties(request, requestUri);

612

613		if (this.secure != null) {

614			request.setSecure(this.secure);

615		}

616		if (this.principal != null) {

617			request.setUserPrincipal(this.principal);

618		}

619		if (this.session != null) {

620			request.setSession(this.session);

621		}

622

623		request.setCharacterEncoding(this.characterEncoding);

624		request.setContent(this.content);

625		request.setContentType(this.contentType);

626

627		for (String name : this.headers.keySet()) {

628			for (Object value : this.headers.get(name)) {

629				request.addHeader(name, value);

630			}

631		}

632

633		if (this.url.getRawQuery() != null) {

634			request.setQueryString(this.url.getRawQuery());

635		}

636		addRequestParams(request, UriComponentsBuilder.fromUri(this.url).build().getQueryParams());

637

638		for (String name : this.parameters.keySet()) {

639			for (String value : this.parameters.get(name)) {

640				request.addParameter(name, value);

641			}

642		}

643

644		if (this.content != null && this.content.length > 0) {

645			String requestContentType = request.getContentType();

646			if (requestContentType != null) {

647				MediaType mediaType = MediaType.parseMediaType(requestContentType);

648				if (MediaType.APPLICATION_FORM_URLENCODED.includes(mediaType)) {

649					addRequestParams(request, parseFormData(mediaType));

650				}

651			}

652		}

653

654		if (!ObjectUtils.isEmpty(this.cookies)) {

655			request.setCookies(this.cookies.toArray(new Cookie[0]));

656		}

657		if (!ObjectUtils.isEmpty(this.locales)) {

658			request.setPreferredLocales(this.locales);

659		}

660

661		for (String name : this.requestAttributes.keySet()) {

662			request.setAttribute(name, this.requestAttributes.get(name));

663		}

664		for (String name : this.sessionAttributes.keySet()) {

665			HttpSession session = request.getSession();

666			Assert.state(session != null, "No HttpSession");

667			session.setAttribute(name, this.sessionAttributes.get(name));

668		}

669

670		FlashMap flashMap = new FlashMap();

671		flashMap.putAll(this.flashAttributes);

672		FlashMapManager flashMapManager = getFlashMapManager(request);

673		flashMapManager.saveOutputFlashMap(flashMap, request, new MockHttpServletResponse());

674

675		return request;

676	}

687	/**

688	 * Update the contextPath, servletPath, and pathInfo of the request.

689	 */

690	private void updatePathRequestProperties(MockHttpServletRequest request, String requestUri) {

691		if (!requestUri.startsWith(this.contextPath)) {

692			throw new IllegalArgumentException(

693					"Request URI [" + requestUri + "] does not start with context path [" + this.contextPath + "]");

694		}

695		request.setContextPath(this.contextPath);

696		request.setServletPath(this.servletPath);

697

698		if ("".equals(this.pathInfo)) {

699			if (!requestUri.startsWith(this.contextPath + this.servletPath)) {

700				throw new IllegalArgumentException(

701						"Invalid servlet path [" + this.servletPath + "] for request URI [" + requestUri + "]");

702			}

703			String extraPath = requestUri.substring(this.contextPath.length() + this.servletPath.length());

704			this.pathInfo = (StringUtils.hasText(extraPath) ?

705					urlPathHelper.decodeRequestString(request, extraPath) : null);

706		}

707		request.setPathInfo(this.pathInfo);

708	}

710	private void addRequestParams(MockHttpServletRequest request, MultiValueMap<String, String> map) {

711		map.forEach((key, values) -> values.forEach(value -> {

712			value = (value != null ? UriUtils.decode(value, StandardCharsets.UTF_8) : null);

713			request.addParameter(UriUtils.decode(key, StandardCharsets.UTF_8), value);

714		}));

715	}

717	private MultiValueMap<String, String> parseFormData(final MediaType mediaType) {

718		HttpInputMessage message = new HttpInputMessage() {

719			@Override

720			public InputStream getBody() {

721				return (content != null ? new ByteArrayInputStream(content) : StreamUtils.emptyInput());

722			}

723			@Override

724			public HttpHeaders getHeaders() {

725				HttpHeaders headers = new HttpHeaders();

726				headers.setContentType(mediaType);

727				return headers;

728			}

729		};

730

731		try {

732			return new FormHttpMessageConverter().read(null, message);

733		}

734		catch (IOException ex) {

735			throw new IllegalStateException("Failed to parse form data in request body", ex);

736		}

737	}

739	private FlashMapManager getFlashMapManager(MockHttpServletRequest request) {

740		FlashMapManager flashMapManager = null;

741		try {

742			ServletContext servletContext = request.getServletContext();

743			WebApplicationContext wac = WebApplicationContextUtils.getRequiredWebApplicationContext(servletContext);

744			flashMapManager = wac.getBean(DispatcherServlet.FLASH_MAP_MANAGER_BEAN_NAME, FlashMapManager.class);

745		}

746		catch (IllegalStateException | NoSuchBeanDefinitionException ex) {

747			// ignore

748		}

749		return (flashMapManager != null ? flashMapManager : new SessionFlashMapManager());

750	}
