74	 * Attempt to find a {@link Field field} on the supplied {@link Class} with the

75	 * supplied {@code name}. Searches all superclasses up to {@link Object}.

76	 * @param clazz the class to introspect

77	 * @param name the name of the field

78	 * @return the corresponding Field object, or {@code null} if not found

80	@Nullable

82		return findField(clazz, name, null);

83	}

85	/**

86	 * Attempt to find a {@link Field field} on the supplied {@link Class} with the

87	 * supplied {@code name} and/or {@link Class type}. Searches all superclasses

88	 * up to {@link Object}.

89	 * @param clazz the class to introspect

90	 * @param name the name of the field (may be {@code null} if type is specified)

91	 * @param type the type of the field (may be {@code null} if name is specified)

92	 * @return the corresponding Field object, or {@code null} if not found

93	 */

94	@Nullable

95	public static Field findField(Class<?> clazz, @Nullable String name, @Nullable Class<?> type) {

96		Assert.notNull(clazz, "Class must not be null");

97		Assert.isTrue(name != null || type != null, "Either name or type of the field must be specified");

98		Class<?> searchType = clazz;

99		while (Object.class != searchType && searchType != null) {

100			Field[] fields = getDeclaredFields(searchType);

101			for (Field field : fields) {

102				if ((name == null || name.equals(field.getName())) &&

103						(type == null || type.equals(field.getType()))) {

104					return field;

105				}

106			}

107			searchType = searchType.getSuperclass();

108		}

109		return null;

110	}

112	/**

113	 * Set the field represented by the supplied {@link Field field object} on the

114	 * specified {@link Object target object} to the specified {@code value}.

115	 * In accordance with {@link Field#set(Object, Object)} semantics, the new value

116	 * is automatically unwrapped if the underlying field has a primitive type.

117	 * <p>Thrown exceptions are handled via a call to {@link #handleReflectionException(Exception)}.

118	 * @param field the field to set

119	 * @param target the target object on which to set the field

120	 * @param value the value to set (may be {@code null})

121	 */

122	public static void setField(Field field, @Nullable Object target, @Nullable Object value) {

123		try {

124			field.set(target, value);

125		}

126		catch (IllegalAccessException ex) {

127			handleReflectionException(ex);

128			throw new IllegalStateException(

129					"Unexpected reflection exception - " + ex.getClass().getName() + ": " + ex.getMessage());

130		}

131	}

138	 * <p>Thrown exceptions are handled via a call to {@link #handleReflectionException(Exception)}.

155	/**

156	 * Attempt to find a {@link Method} on the supplied class with the supplied name

157	 * and no parameters. Searches all superclasses up to {@code Object}.

158	 * <p>Returns {@code null} if no {@link Method} can be found.

159	 * @param clazz the class to introspect

160	 * @param name the name of the method

161	 * @return the Method object, or {@code null} if none found

162	 */

163	@Nullable

164	public static Method findMethod(Class<?> clazz, String name) {

165		return findMethod(clazz, name, new Class<?>[0]);

166	}

168	/**

169	 * Attempt to find a {@link Method} on the supplied class with the supplied name

170	 * and parameter types. Searches all superclasses up to {@code Object}.

171	 * <p>Returns {@code null} if no {@link Method} can be found.

172	 * @param clazz the class to introspect

173	 * @param name the name of the method

174	 * @param paramTypes the parameter types of the method

175	 * (may be {@code null} to indicate any signature)

176	 * @return the Method object, or {@code null} if none found

177	 */

178	@Nullable

179	public static Method findMethod(Class<?> clazz, String name, @Nullable Class<?>... paramTypes) {

180		Assert.notNull(clazz, "Class must not be null");

181		Assert.notNull(name, "Method name must not be null");

182		Class<?> searchType = clazz;

183		while (searchType != null) {

184			Method[] methods = (searchType.isInterface() ? searchType.getMethods() : getDeclaredMethods(searchType));

185			for (Method method : methods) {

186				if (name.equals(method.getName()) &&

187						(paramTypes == null || Arrays.equals(paramTypes, method.getParameterTypes()))) {

188					return method;

189				}

190			}

191			searchType = searchType.getSuperclass();

192		}

193		return null;

194	}

210	/**

211	 * Invoke the specified {@link Method} against the supplied target object with the

212	 * supplied arguments. The target object can be {@code null} when invoking a

213	 * static {@link Method}.

214	 * <p>Thrown exceptions are handled via a call to {@link #handleReflectionException}.

215	 * @param method the method to invoke

216	 * @param target the target object to invoke the method on

217	 * @param args the invocation arguments (may be {@code null})

222		try {

223			return method.invoke(target, args);

224		}

225		catch (Exception ex) {

226			handleReflectionException(ex);

227		}

228		throw new IllegalStateException("Should never get here");

229	}

245	/**

246	 * Invoke the specified JDBC API {@link Method} against the supplied target

247	 * object with the supplied arguments.

248	 * @param method the method to invoke

249	 * @param target the target object to invoke the method on

250	 * @param args the invocation arguments (may be {@code null})

251	 * @return the invocation result, if any

252	 * @throws SQLException the JDBC API SQLException to rethrow (if any)

253	 * @see #invokeMethod(java.lang.reflect.Method, Object, Object[])

254	 */

255	@Nullable

256	public static Object invokeJdbcMethod(Method method, @Nullable Object target, @Nullable Object... args)

257			throws SQLException {

258		try {

259			return method.invoke(target, args);

260		}

261		catch (IllegalAccessException ex) {

262			handleReflectionException(ex);

263		}

264		catch (InvocationTargetException ex) {

265			if (ex.getTargetException() instanceof SQLException) {

266				throw (SQLException) ex.getTargetException();

267			}

268			handleInvocationTargetException(ex);

269		}

270		throw new IllegalStateException("Should never get here");

271	}

273	/**

274	 * Handle the given reflection exception. Should only be called if no

275	 * checked exception is expected to be thrown by the target method.

276	 * <p>Throws the underlying RuntimeException or Error in case of an

277	 * InvocationTargetException with such a root cause. Throws an

278	 * IllegalStateException with an appropriate message or

279	 * UndeclaredThrowableException otherwise.

280	 * @param ex the reflection exception to handle

281	 */

282	public static void handleReflectionException(Exception ex) {

283		if (ex instanceof NoSuchMethodException) {

284			throw new IllegalStateException("Method not found: " + ex.getMessage());

285		}

286		if (ex instanceof IllegalAccessException) {

287			throw new IllegalStateException("Could not access method: " + ex.getMessage());

288		}

289		if (ex instanceof InvocationTargetException) {

290			handleInvocationTargetException((InvocationTargetException) ex);

291		}

292		if (ex instanceof RuntimeException) {

293			throw (RuntimeException) ex;

294		}

295		throw new UndeclaredThrowableException(ex);

296	}

298	/**

299	 * Handle the given invocation target exception. Should only be called if no

300	 * checked exception is expected to be thrown by the target method.

301	 * <p>Throws the underlying RuntimeException or Error in case of such a root

302	 * cause. Throws an UndeclaredThrowableException otherwise.

303	 * @param ex the invocation target exception to handle

304	 */

305	public static void handleInvocationTargetException(InvocationTargetException ex) {

306		rethrowRuntimeException(ex.getTargetException());

307	}

309	/**

310	 * Rethrow the given {@link Throwable exception}, which is presumably the

311	 * <em>target exception</em> of an {@link InvocationTargetException}.

312	 * Should only be called if no checked exception is expected to be thrown

313	 * by the target method.

314	 * <p>Rethrows the underlying exception cast to a {@link RuntimeException} or

315	 * {@link Error} if appropriate; otherwise, throws an

316	 * {@link UndeclaredThrowableException}.

317	 * @param ex the exception to rethrow

318	 * @throws RuntimeException the rethrown exception

319	 */

320	public static void rethrowRuntimeException(Throwable ex) {

321		if (ex instanceof RuntimeException) {

322			throw (RuntimeException) ex;

323		}

324		if (ex instanceof Error) {

325			throw (Error) ex;

326		}

327		throw new UndeclaredThrowableException(ex);

328	}

330	/**

331	 * Rethrow the given {@link Throwable exception}, which is presumably the

332	 * <em>target exception</em> of an {@link InvocationTargetException}.

333	 * Should only be called if no checked exception is expected to be thrown

334	 * by the target method.

335	 * <p>Rethrows the underlying exception cast to an {@link Exception} or

336	 * {@link Error} if appropriate; otherwise, throws an

337	 * {@link UndeclaredThrowableException}.

338	 * @param ex the exception to rethrow

339	 * @throws Exception the rethrown exception (in case of a checked exception)

340	 */

341	public static void rethrowException(Throwable ex) throws Exception {

342		if (ex instanceof Exception) {

343			throw (Exception) ex;

344		}

345		if (ex instanceof Error) {

346			throw (Error) ex;

347		}

348		throw new UndeclaredThrowableException(ex);

349	}

351	/**

352	 * Determine whether the given method explicitly declares the given

353	 * exception or one of its superclasses, which means that an exception

354	 * of that type can be propagated as-is within a reflective invocation.

355	 * @param method the declaring method

356	 * @param exceptionType the exception to throw

357	 * @return {@code true} if the exception can be thrown as-is;

358	 * {@code false} if it needs to be wrapped

359	 */

360	public static boolean declaresException(Method method, Class<?> exceptionType) {

361		Assert.notNull(method, "Method must not be null");

362		Class<?>[] declaredExceptions = method.getExceptionTypes();

363		for (Class<?> declaredException : declaredExceptions) {

364			if (declaredException.isAssignableFrom(exceptionType)) {

365				return true;

366			}

367		}

368		return false;

369	}

372	 * Determine whether the given field is a "public static final" constant.

373	 * @param field the field to check

374	 */

375	public static boolean isPublicStaticFinal(Field field) {

376		int modifiers = field.getModifiers();

377		return (Modifier.isPublic(modifiers) && Modifier.isStatic(modifiers) && Modifier.isFinal(modifiers));

380	/**

381	 * Determine whether the given method is an "equals" method.

382	 * @see java.lang.Object#equals(Object)

383	 */

384	public static boolean isEqualsMethod(@Nullable Method method) {

385		if (method == null || !method.getName().equals("equals")) {

386			return false;

387		}

388		Class<?>[] paramTypes = method.getParameterTypes();

389		return (paramTypes.length == 1 && paramTypes[0] == Object.class);

390	}

424	/**

425	 * Determine whether the given method is a CGLIB 'renamed' method,

426	 * following the pattern "CGLIB$methodName$0".

427	 * @param renamedMethod the method to check

428	 * @see org.springframework.cglib.proxy.Enhancer#rename

429	 */

430	public static boolean isCglibRenamedMethod(Method renamedMethod) {

431		String name = renamedMethod.getName();

432		if (name.startsWith(CGLIB_RENAMED_METHOD_PREFIX)) {

433			int i = name.length() - 1;

434			while (i >= 0 && Character.isDigit(name.charAt(i))) {

435				i--;

436			}

437			return ((i > CGLIB_RENAMED_METHOD_PREFIX.length()) &&

438						(i < name.length() - 1) && name.charAt(i) == '$');

439		}

440		return false;

441	}

508	/**

509	 * Perform the given callback operation on all matching methods of the given

510	 * class, as locally declared or equivalent thereof (such as default methods

511	 * on Java 8 based interfaces that the given class implements).

512	 * @param clazz the class to introspect

513	 * @param mc the callback to invoke for each method

514	 * @since 4.2

515	 * @throws IllegalStateException if introspection fails

516	 * @see #doWithMethods

517	 */

518	public static void doWithLocalMethods(Class<?> clazz, MethodCallback mc) {

519		Method[] methods = getDeclaredMethods(clazz);

520		for (Method method : methods) {

521			try {

522				mc.doWith(method);

523			}

524			catch (IllegalAccessException ex) {

525				throw new IllegalStateException("Not allowed to access method '" + method.getName() + "': " + ex);

526			}

527		}

528	}

530	/**

531	 * Perform the given callback operation on all matching methods of the given

532	 * class and superclasses.

533	 * <p>The same named method occurring on subclass and superclass will appear

534	 * twice, unless excluded by a {@link MethodFilter}.

535	 * @param clazz the class to introspect

536	 * @param mc the callback to invoke for each method

537	 * @throws IllegalStateException if introspection fails

538	 * @see #doWithMethods(Class, MethodCallback, MethodFilter)

539	 */

540	public static void doWithMethods(Class<?> clazz, MethodCallback mc) {

541		doWithMethods(clazz, mc, null);

542	}

543

544	/**

545	 * Perform the given callback operation on all matching methods of the given

546	 * class and superclasses (or given interface and super-interfaces).

547	 * <p>The same named method occurring on subclass and superclass will appear

548	 * twice, unless excluded by the specified {@link MethodFilter}.

549	 * @param clazz the class to introspect

550	 * @param mc the callback to invoke for each method

551	 * @param mf the filter that determines the methods to apply the callback to

552	 * @throws IllegalStateException if introspection fails

553	 */

554	public static void doWithMethods(Class<?> clazz, MethodCallback mc, @Nullable MethodFilter mf) {

555		// Keep backing up the inheritance hierarchy.

556		Method[] methods = getDeclaredMethods(clazz);

557		for (Method method : methods) {

558			if (mf != null && !mf.matches(method)) {

559				continue;

560			}

561			try {

562				mc.doWith(method);

563			}

564			catch (IllegalAccessException ex) {

565				throw new IllegalStateException("Not allowed to access method '" + method.getName() + "': " + ex);

566			}

567		}

568		if (clazz.getSuperclass() != null) {

569			doWithMethods(clazz.getSuperclass(), mc, mf);

570		}

571		else if (clazz.isInterface()) {

572			for (Class<?> superIfc : clazz.getInterfaces()) {

573				doWithMethods(superIfc, mc, mf);

574			}

575		}

576	}

578	/**

579	 * Get all declared methods on the leaf class and all superclasses.

580	 * Leaf class methods are included first.

581	 * @param leafClass the class to introspect

582	 * @throws IllegalStateException if introspection fails

583	 */

584	public static Method[] getAllDeclaredMethods(Class<?> leafClass) {

585		final List<Method> methods = new ArrayList<>(32);

586		doWithMethods(leafClass, methods::add);

587		return methods.toArray(new Method[0]);

588	}

589

590	/**

591	 * Get the unique set of declared methods on the leaf class and all superclasses.

592	 * Leaf class methods are included first and while traversing the superclass hierarchy

593	 * any methods found with signatures matching a method already included are filtered out.

594	 * @param leafClass the class to introspect

595	 * @throws IllegalStateException if introspection fails

596	 */

597	public static Method[] getUniqueDeclaredMethods(Class<?> leafClass) {

598		final List<Method> methods = new ArrayList<>(32);

599		doWithMethods(leafClass, method -> {

600			boolean knownSignature = false;

601			Method methodBeingOverriddenWithCovariantReturnType = null;

602			for (Method existingMethod : methods) {

603				if (method.getName().equals(existingMethod.getName()) &&

604						Arrays.equals(method.getParameterTypes(), existingMethod.getParameterTypes())) {

605					// Is this a covariant return type situation?

606					if (existingMethod.getReturnType() != method.getReturnType() &&

607							existingMethod.getReturnType().isAssignableFrom(method.getReturnType())) {

608						methodBeingOverriddenWithCovariantReturnType = existingMethod;

609					}

610					else {

611						knownSignature = true;

612					}

613					break;

614				}

615			}

616			if (methodBeingOverriddenWithCovariantReturnType != null) {

617				methods.remove(methodBeingOverriddenWithCovariantReturnType);

618			}

619			if (!knownSignature && !isCglibRenamedMethod(method)) {

620				methods.add(method);

621			}

622		});

623		return methods.toArray(new Method[0]);

624	}

626	/**

627	 * This variant retrieves {@link Class#getDeclaredMethods()} from a local cache

628	 * in order to avoid the JVM's SecurityManager check and defensive array copying.

629	 * In addition, it also includes Java 8 default methods from locally implemented

630	 * interfaces, since those are effectively to be treated just like declared methods.

631	 * @param clazz the class to introspect

632	 * @return the cached array of methods

633	 * @throws IllegalStateException if introspection fails

634	 * @see Class#getDeclaredMethods()

635	 */

636	private static Method[] getDeclaredMethods(Class<?> clazz) {

637		Assert.notNull(clazz, "Class must not be null");

638		Method[] result = declaredMethodsCache.get(clazz);

639		if (result == null) {

640			try {

641				Method[] declaredMethods = clazz.getDeclaredMethods();

642				List<Method> defaultMethods = findConcreteMethodsOnInterfaces(clazz);

643				if (defaultMethods != null) {

644					result = new Method[declaredMethods.length + defaultMethods.size()];

645					System.arraycopy(declaredMethods, 0, result, 0, declaredMethods.length);

646					int index = declaredMethods.length;

647					for (Method defaultMethod : defaultMethods) {

648						result[index] = defaultMethod;

649						index++;

650					}

651				}

652				else {

653					result = declaredMethods;

654				}

655				declaredMethodsCache.put(clazz, (result.length == 0 ? NO_METHODS : result));

656			}

657			catch (Throwable ex) {

658				throw new IllegalStateException("Failed to introspect Class [" + clazz.getName() +

659						"] from ClassLoader [" + clazz.getClassLoader() + "]", ex);

660			}

661		}

662		return result;

663	}

664

665	@Nullable

666	private static List<Method> findConcreteMethodsOnInterfaces(Class<?> clazz) {

667		List<Method> result = null;

668		for (Class<?> ifc : clazz.getInterfaces()) {

669			for (Method ifcMethod : ifc.getMethods()) {

670				if (!Modifier.isAbstract(ifcMethod.getModifiers())) {

671					if (result == null) {

672						result = new LinkedList<>();

673					}

674					result.add(ifcMethod);

675				}

676			}

677		}

678		return result;

679	}

681	/**

682	 * Invoke the given callback on all fields in the target class, going up the

683	 * class hierarchy to get all declared fields.

684	 * @param clazz the target class to analyze

685	 * @param fc the callback to invoke for each field

686	 * @since 4.2

687	 * @throws IllegalStateException if introspection fails

688	 * @see #doWithFields

689	 */

690	public static void doWithLocalFields(Class<?> clazz, FieldCallback fc) {

691		for (Field field : getDeclaredFields(clazz)) {

692			try {

693				fc.doWith(field);

694			}

695			catch (IllegalAccessException ex) {

696				throw new IllegalStateException("Not allowed to access field '" + field.getName() + "': " + ex);

697			}

698		}

699	}

701	/**

702	 * Invoke the given callback on all fields in the target class, going up the

703	 * class hierarchy to get all declared fields.

704	 * @param clazz the target class to analyze

705	 * @param fc the callback to invoke for each field

706	 * @throws IllegalStateException if introspection fails

707	 */

708	public static void doWithFields(Class<?> clazz, FieldCallback fc) {

709		doWithFields(clazz, fc, null);

710	}

711

712	/**

713	 * Invoke the given callback on all fields in the target class, going up the

714	 * class hierarchy to get all declared fields.

715	 * @param clazz the target class to analyze

716	 * @param fc the callback to invoke for each field

717	 * @param ff the filter that determines the fields to apply the callback to

718	 * @throws IllegalStateException if introspection fails

719	 */

720	public static void doWithFields(Class<?> clazz, FieldCallback fc, @Nullable FieldFilter ff) {

721		// Keep backing up the inheritance hierarchy.

722		Class<?> targetClass = clazz;

723		do {

724			Field[] fields = getDeclaredFields(targetClass);

725			for (Field field : fields) {

726				if (ff != null && !ff.matches(field)) {

727					continue;

728				}

729				try {

730					fc.doWith(field);

731				}

732				catch (IllegalAccessException ex) {

733					throw new IllegalStateException("Not allowed to access field '" + field.getName() + "': " + ex);

734				}

735			}

736			targetClass = targetClass.getSuperclass();

737		}

738		while (targetClass != null && targetClass != Object.class);

739	}

741	/**

742	 * This variant retrieves {@link Class#getDeclaredFields()} from a local cache

743	 * in order to avoid the JVM's SecurityManager check and defensive array copying.

744	 * @param clazz the class to introspect

745	 * @return the cached array of fields

746	 * @throws IllegalStateException if introspection fails

747	 * @see Class#getDeclaredFields()

748	 */

749	private static Field[] getDeclaredFields(Class<?> clazz) {

750		Assert.notNull(clazz, "Class must not be null");

751		Field[] result = declaredFieldsCache.get(clazz);

752		if (result == null) {

753			try {

754				result = clazz.getDeclaredFields();

755				declaredFieldsCache.put(clazz, (result.length == 0 ? NO_FIELDS : result));

756			}

757			catch (Throwable ex) {

758				throw new IllegalStateException("Failed to introspect Class [" + clazz.getName() +

759						"] from ClassLoader [" + clazz.getClassLoader() + "]", ex);

760			}

761		}

762		return result;

763	}

765	/**

766	 * Given the source object and the destination, which must be the same class

767	 * or a subclass, copy all fields, including inherited fields. Designed to

768	 * work on objects with public no-arg constructors.

769	 * @throws IllegalStateException if introspection fails

770	 */

771	public static void shallowCopyFieldState(final Object src, final Object dest) {

772		Assert.notNull(src, "Source for field copy cannot be null");

773		Assert.notNull(dest, "Destination for field copy cannot be null");

774		if (!src.getClass().isAssignableFrom(dest.getClass())) {

775			throw new IllegalArgumentException("Destination class [" + dest.getClass().getName() +

776					"] must be same or subclass as source class [" + src.getClass().getName() + "]");

777		}

778		doWithFields(src.getClass(), field -> {

779			makeAccessible(field);

780			Object srcValue = field.get(src);

781			field.set(dest, srcValue);

782		}, COPYABLE_FIELDS);

783	}

785	/**

786	 * Clear the internal method/field cache.

787	 * @since 4.2.4

788	 */

789	public static void clearCache() {

790		declaredMethodsCache.clear();

791		declaredFieldsCache.clear();

792	}
