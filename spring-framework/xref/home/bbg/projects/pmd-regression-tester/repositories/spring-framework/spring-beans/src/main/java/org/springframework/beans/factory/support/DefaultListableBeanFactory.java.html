81/**

82 * Default implementation of the

83 * {@link org.springframework.beans.factory.ListableBeanFactory} and

84 * {@link BeanDefinitionRegistry} interfaces: a full-fledged bean factory

85 * based on bean definition objects.

86 *

87 * <p>Typical usage is registering all bean definitions first (possibly read

88 * from a bean definition file), before accessing beans. Bean definition lookup

89 * is therefore an inexpensive operation in a local bean definition table,

90 * operating on pre-built bean definition metadata objects.

91 *

92 * <p>Can be used as a standalone bean factory, or as a superclass for custom

93 * bean factories. Note that readers for specific bean definition formats are

94 * typically implemented separately rather than as bean factory subclasses:

95 * see for example {@link PropertiesBeanDefinitionReader} and

96 * {@link org.springframework.beans.factory.xml.XmlBeanDefinitionReader}.

97 *

98 * <p>For an alternative implementation of the

99 * {@link org.springframework.beans.factory.ListableBeanFactory} interface,

100 * have a look at {@link StaticListableBeanFactory}, which manages existing

101 * bean instances rather than creating new ones based on bean definitions.

102 *

103 * @author Rod Johnson

104 * @author Juergen Hoeller

105 * @author Sam Brannen

106 * @author Costin Leau

107 * @author Chris Beams

108 * @author Phillip Webb

109 * @author Stephane Nicoll

110 * @since 16 April 2001

111 * @see StaticListableBeanFactory

112 * @see PropertiesBeanDefinitionReader

113 * @see org.springframework.beans.factory.xml.XmlBeanDefinitionReader

114 */

115@SuppressWarnings("serial")

116public class DefaultListableBeanFactory extends AbstractAutowireCapableBeanFactory

117		implements ConfigurableListableBeanFactory, BeanDefinitionRegistry, Serializable {

302	}

312	@Override

313	public void copyConfigurationFrom(ConfigurableBeanFactory otherFactory) {

314		super.copyConfigurationFrom(otherFactory);

315		if (otherFactory instanceof DefaultListableBeanFactory) {

316			DefaultListableBeanFactory otherListableFactory = (DefaultListableBeanFactory) otherFactory;

317			this.allowBeanDefinitionOverriding = otherListableFactory.allowBeanDefinitionOverriding;

318			this.allowEagerClassLoading = otherListableFactory.allowEagerClassLoading;

319			this.dependencyComparator = otherListableFactory.dependencyComparator;

320			// A clone of the AutowireCandidateResolver since it is potentially BeanFactoryAware...

321			setAutowireCandidateResolver(BeanUtils.instantiateClass(getAutowireCandidateResolver().getClass()));

322			// Make resolvable dependencies (e.g. ResourceLoader) available here as well...

323			this.resolvableDependencies.putAll(otherListableFactory.resolvableDependencies);

324		}

387	@Override

388	public String[] getBeanNamesForType(@Nullable Class<?> type, boolean includeNonSingletons, boolean allowEagerInit) {

389		if (!isConfigurationFrozen() || type == null || !allowEagerInit) {

390			return doGetBeanNamesForType(ResolvableType.forRawClass(type), includeNonSingletons, allowEagerInit);

391		}

392		Map<Class<?>, String[]> cache =

393				(includeNonSingletons ? this.allBeanNamesByType : this.singletonBeanNamesByType);

394		String[] resolvedBeanNames = cache.get(type);

395		if (resolvedBeanNames != null) {

396			return resolvedBeanNames;

397		}

398		resolvedBeanNames = doGetBeanNamesForType(ResolvableType.forRawClass(type), includeNonSingletons, true);

399		if (ClassUtils.isCacheSafe(type, getBeanClassLoader())) {

400			cache.put(type, resolvedBeanNames);

401		}

402		return resolvedBeanNames;

403	}

405	private String[] doGetBeanNamesForType(ResolvableType type, boolean includeNonSingletons, boolean allowEagerInit) {

406		List<String> result = new ArrayList<>();

407

408		// Check all bean definitions.

409		for (String beanName : this.beanDefinitionNames) {

410			// Only consider bean as eligible if the bean name

411			// is not defined as alias for some other bean.

412			if (!isAlias(beanName)) {

413				try {

414					RootBeanDefinition mbd = getMergedLocalBeanDefinition(beanName);

415					// Only check bean definition if it is complete.

416					if (!mbd.isAbstract() && (allowEagerInit ||

417							(mbd.hasBeanClass() || !mbd.isLazyInit() || isAllowEagerClassLoading()) &&

418									!requiresEagerInitForType(mbd.getFactoryBeanName()))) {

419						// In case of FactoryBean, match object created by FactoryBean.

420						boolean isFactoryBean = isFactoryBean(beanName, mbd);

421						BeanDefinitionHolder dbd = mbd.getDecoratedDefinition();

422						boolean matchFound =

423								(allowEagerInit || !isFactoryBean ||

424										(dbd != null && !mbd.isLazyInit()) || containsSingleton(beanName)) &&

425								(includeNonSingletons ||

426										(dbd != null ? mbd.isSingleton() : isSingleton(beanName))) &&

427								isTypeMatch(beanName, type);

428						if (!matchFound && isFactoryBean) {

429							// In case of FactoryBean, try to match FactoryBean instance itself next.

430							beanName = FACTORY_BEAN_PREFIX + beanName;

431							matchFound = (includeNonSingletons || mbd.isSingleton()) && isTypeMatch(beanName, type);

432						}

433						if (matchFound) {

434							result.add(beanName);

435						}

436					}

437				}

438				catch (CannotLoadBeanClassException ex) {

439					if (allowEagerInit) {

440						throw ex;

441					}

442					// Probably contains a placeholder: let's ignore it for type matching purposes.

443					if (this.logger.isDebugEnabled()) {

444						this.logger.debug("Ignoring bean class loading failure for bean '" + beanName + "'", ex);

445					}

446					onSuppressedException(ex);

447				}

448				catch (BeanDefinitionStoreException ex) {

449					if (allowEagerInit) {

450						throw ex;

451					}

452					// Probably contains a placeholder: let's ignore it for type matching purposes.

453					if (this.logger.isDebugEnabled()) {

454						this.logger.debug("Ignoring unresolvable metadata in bean definition '" + beanName + "'", ex);

455					}

456					onSuppressedException(ex);

457				}

458			}

459		}

460

461		// Check manually registered singletons too.

462		for (String beanName : this.manualSingletonNames) {

463			try {

464				// In case of FactoryBean, match object created by FactoryBean.

465				if (isFactoryBean(beanName)) {

466					if ((includeNonSingletons || isSingleton(beanName)) && isTypeMatch(beanName, type)) {

467						result.add(beanName);

468						// Match found for this bean: do not match FactoryBean itself anymore.

469						continue;

470					}

471					// In case of FactoryBean, try to match FactoryBean itself next.

472					beanName = FACTORY_BEAN_PREFIX + beanName;

473				}

474				// Match raw bean instance (might be raw FactoryBean).

475				if (isTypeMatch(beanName, type)) {

476					result.add(beanName);

477				}

478			}

479			catch (NoSuchBeanDefinitionException ex) {

480				// Shouldn't happen - probably a result of circular reference resolution...

481				if (logger.isDebugEnabled()) {

482					logger.debug("Failed to check manually registered singleton with name '" + beanName + "'", ex);

483				}

484			}

485		}

486

487		return StringUtils.toStringArray(result);

488	}

506	@Override

507	@SuppressWarnings("unchecked")

508	public <T> Map<String, T> getBeansOfType(@Nullable Class<T> type, boolean includeNonSingletons, boolean allowEagerInit)

509			throws BeansException {

510

511		String[] beanNames = getBeanNamesForType(type, includeNonSingletons, allowEagerInit);

512		Map<String, T> result = new LinkedHashMap<>(beanNames.length);

513		for (String beanName : beanNames) {

514			try {

515				Object beanInstance = getBean(beanName);

516				result.put(beanName, (beanInstance instanceof NullBean ? null : (T) beanInstance));

517			}

518			catch (BeanCreationException ex) {

519				Throwable rootCause = ex.getMostSpecificCause();

520				if (rootCause instanceof BeanCurrentlyInCreationException) {

521					BeanCreationException bce = (BeanCreationException) rootCause;

522					String exBeanName = bce.getBeanName();

523					if (exBeanName != null && isCurrentlyInCreation(exBeanName)) {

524						if (this.logger.isDebugEnabled()) {

525							this.logger.debug("Ignoring match to currently created bean '" + exBeanName + "': " +

526									ex.getMessage());

527						}

528						onSuppressedException(ex);

529						// Ignore: indicates a circular reference when autowiring constructors.

530						// We want to find matches other than the currently created bean itself.

531						continue;

532					}

533				}

534				throw ex;

535			}

536		}

537		return result;

538	}

540	@Override

541	public String[] getBeanNamesForAnnotation(Class<? extends Annotation> annotationType) {

542		List<String> results = new ArrayList<>();

543		for (String beanName : this.beanDefinitionNames) {

544			BeanDefinition beanDefinition = getBeanDefinition(beanName);

545			if (!beanDefinition.isAbstract() && findAnnotationOnBean(beanName, annotationType) != null) {

546				results.add(beanName);

547			}

548		}

549		for (String beanName : this.manualSingletonNames) {

550			if (!results.contains(beanName) && findAnnotationOnBean(beanName, annotationType) != null) {

551				results.add(beanName);

552			}

553		}

554		return StringUtils.toStringArray(results);

555	}

561		for (String beanName : beanNames) {

567	/**

568	 * Find a {@link Annotation} of {@code annotationType} on the specified

569	 * bean, traversing its interfaces and super classes if no annotation can be

570	 * found on the given class itself, as well as checking its raw bean class

571	 * if not found on the exposed bean reference (e.g. in case of a proxy).

572	 */

573	@Override

574	@Nullable

575	public <A extends Annotation> A findAnnotationOnBean(String beanName, Class<A> annotationType)

576			throws NoSuchBeanDefinitionException{

577

578		A ann = null;

579		Class<?> beanType = getType(beanName);

580		if (beanType != null) {

581			ann = AnnotationUtils.findAnnotation(beanType, annotationType);

582		}

583		if (ann == null && containsBeanDefinition(beanName)) {

584			BeanDefinition bd = getMergedBeanDefinition(beanName);

585			if (bd instanceof AbstractBeanDefinition) {

586				AbstractBeanDefinition abd = (AbstractBeanDefinition) bd;

587				if (abd.hasBeanClass()) {

588					ann = AnnotationUtils.findAnnotation(abd.getBeanClass(), annotationType);

589				}

590			}

591		}

592		return ann;

593	}

619	/**

620	 * Determine whether the specified bean definition qualifies as an autowire candidate,

621	 * to be injected into other beans which declare a dependency of matching type.

622	 * @param beanName the name of the bean definition to check

623	 * @param descriptor the descriptor of the dependency to resolve

624	 * @param resolver the AutowireCandidateResolver to use for the actual resolution algorithm

625	 * @return whether the bean should be considered as autowire candidate

626	 */

627	protected boolean isAutowireCandidate(String beanName, DependencyDescriptor descriptor, AutowireCandidateResolver resolver)

628			throws NoSuchBeanDefinitionException {

629

630		String beanDefinitionName = BeanFactoryUtils.transformedBeanName(beanName);

631		if (containsBeanDefinition(beanDefinitionName)) {

632			return isAutowireCandidate(beanName, getMergedLocalBeanDefinition(beanDefinitionName), descriptor, resolver);

633		}

634		else if (containsSingleton(beanName)) {

635			return isAutowireCandidate(beanName, new RootBeanDefinition(getType(beanName)), descriptor, resolver);

636		}

637

638		BeanFactory parent = getParentBeanFactory();

639		if (parent instanceof DefaultListableBeanFactory) {

640			// No bean definition found in this factory -> delegate to parent.

641			return ((DefaultListableBeanFactory) parent).isAutowireCandidate(beanName, descriptor, resolver);

642		}

643		else if (parent instanceof ConfigurableListableBeanFactory) {

644			// If no DefaultListableBeanFactory, can't pass the resolver along.

645			return ((ConfigurableListableBeanFactory) parent).isAutowireCandidate(beanName, descriptor);

646		}

647		else {

648			return true;

649		}

650	}

652	/**

653	 * Determine whether the specified bean definition qualifies as an autowire candidate,

654	 * to be injected into other beans which declare a dependency of matching type.

655	 * @param beanName the name of the bean definition to check

656	 * @param mbd the merged bean definition to check

657	 * @param descriptor the descriptor of the dependency to resolve

658	 * @param resolver the AutowireCandidateResolver to use for the actual resolution algorithm

659	 * @return whether the bean should be considered as autowire candidate

660	 */

661	protected boolean isAutowireCandidate(String beanName, RootBeanDefinition mbd,

662			DependencyDescriptor descriptor, AutowireCandidateResolver resolver) {

663

664		String beanDefinitionName = BeanFactoryUtils.transformedBeanName(beanName);

665		resolveBeanClass(mbd, beanDefinitionName);

666		if (mbd.isFactoryMethodUnique) {

667			boolean resolve;

668			synchronized (mbd.constructorArgumentLock) {

669				resolve = (mbd.resolvedConstructorOrFactoryMethod == null);

670			}

671			if (resolve) {

672				new ConstructorResolver(this).resolveFactoryMethodIfPossible(mbd);

673			}

674		}

675		return resolver.isAutowireCandidate(

676				new BeanDefinitionHolder(mbd, beanName, getAliases(beanDefinitionName)), descriptor);

677	}

726	@Override

727	public void preInstantiateSingletons() throws BeansException {

728		if (this.logger.isDebugEnabled()) {

729			this.logger.debug("Pre-instantiating singletons in " + this);

730		}

731

732		// Iterate over a copy to allow for init methods which in turn register new bean definitions.

733		// While this may not be part of the regular factory bootstrap, it does otherwise work fine.

734		List<String> beanNames = new ArrayList<>(this.beanDefinitionNames);

735

736		// Trigger initialization of all non-lazy singleton beans...

737		for (String beanName : beanNames) {

738			RootBeanDefinition bd = getMergedLocalBeanDefinition(beanName);

739			if (!bd.isAbstract() && bd.isSingleton() && !bd.isLazyInit()) {

740				if (isFactoryBean(beanName)) {

741					Object bean = getBean(FACTORY_BEAN_PREFIX + beanName);

742					if (bean instanceof FactoryBean) {

743						final FactoryBean<?> factory = (FactoryBean<?>) bean;

744						boolean isEagerInit;

745						if (System.getSecurityManager() != null && factory instanceof SmartFactoryBean) {

746							isEagerInit = AccessController.doPrivileged((PrivilegedAction<Boolean>)

747											((SmartFactoryBean<?>) factory)::isEagerInit,

748									getAccessControlContext());

749						}

750						else {

751							isEagerInit = (factory instanceof SmartFactoryBean &&

752									((SmartFactoryBean<?>) factory).isEagerInit());

753						}

754						if (isEagerInit) {

755							getBean(beanName);

756						}

757					}

758				}

759				else {

760					getBean(beanName);

761				}

762			}

763		}

764

765		// Trigger post-initialization callback for all applicable beans...

766		for (String beanName : beanNames) {

767			Object singletonInstance = getSingleton(beanName);

768			if (singletonInstance instanceof SmartInitializingSingleton) {

769				final SmartInitializingSingleton smartSingleton = (SmartInitializingSingleton) singletonInstance;

770				if (System.getSecurityManager() != null) {

771					AccessController.doPrivileged((PrivilegedAction<Object>) () -> {

772						smartSingleton.afterSingletonsInstantiated();

773						return null;

774					}, getAccessControlContext());

775				}

776				else {

777					smartSingleton.afterSingletonsInstantiated();

778				}

779			}

780		}

788	@Override

789	public void registerBeanDefinition(String beanName, BeanDefinition beanDefinition)

790			throws BeanDefinitionStoreException {

791

792		Assert.hasText(beanName, "Bean name must not be empty");

793		Assert.notNull(beanDefinition, "BeanDefinition must not be null");

794

795		if (beanDefinition instanceof AbstractBeanDefinition) {

796			try {

797				((AbstractBeanDefinition) beanDefinition).validate();

798			}

799			catch (BeanDefinitionValidationException ex) {

800				throw new BeanDefinitionStoreException(beanDefinition.getResourceDescription(), beanName,

801						"Validation of bean definition failed", ex);

802			}

803		}

804

805		BeanDefinition oldBeanDefinition;

806

807		oldBeanDefinition = this.beanDefinitionMap.get(beanName);

808		if (oldBeanDefinition != null) {

809			if (!isAllowBeanDefinitionOverriding()) {

810				throw new BeanDefinitionStoreException(beanDefinition.getResourceDescription(), beanName,

811						"Cannot register bean definition [" + beanDefinition + "] for bean '" + beanName +

812						"': There is already [" + oldBeanDefinition + "] bound.");

813			}

814			else if (oldBeanDefinition.getRole() < beanDefinition.getRole()) {

815				// e.g. was ROLE_APPLICATION, now overriding with ROLE_SUPPORT or ROLE_INFRASTRUCTURE

816				if (this.logger.isWarnEnabled()) {

817					this.logger.warn("Overriding user-defined bean definition for bean '" + beanName +

818							"' with a framework-generated bean definition: replacing [" +

819							oldBeanDefinition + "] with [" + beanDefinition + "]");

820				}

821			}

822			else if (!beanDefinition.equals(oldBeanDefinition)) {

823				if (this.logger.isInfoEnabled()) {

824					this.logger.info("Overriding bean definition for bean '" + beanName +

825							"' with a different definition: replacing [" + oldBeanDefinition +

826							"] with [" + beanDefinition + "]");

827				}

828			}

829			else {

830				if (this.logger.isDebugEnabled()) {

831					this.logger.debug("Overriding bean definition for bean '" + beanName +

832							"' with an equivalent definition: replacing [" + oldBeanDefinition +

833							"] with [" + beanDefinition + "]");

834				}

835			}

836			this.beanDefinitionMap.put(beanName, beanDefinition);

837		}

838		else {

839			if (hasBeanCreationStarted()) {

840				// Cannot modify startup-time collection elements anymore (for stable iteration)

841				synchronized (this.beanDefinitionMap) {

842					this.beanDefinitionMap.put(beanName, beanDefinition);

843					List<String> updatedDefinitions = new ArrayList<>(this.beanDefinitionNames.size() + 1);

844					updatedDefinitions.addAll(this.beanDefinitionNames);

845					updatedDefinitions.add(beanName);

846					this.beanDefinitionNames = updatedDefinitions;

847					if (this.manualSingletonNames.contains(beanName)) {

848						Set<String> updatedSingletons = new LinkedHashSet<>(this.manualSingletonNames);

849						updatedSingletons.remove(beanName);

850						this.manualSingletonNames = updatedSingletons;

851					}

852				}

853			}

854			else {

855				// Still in startup registration phase

856				this.beanDefinitionMap.put(beanName, beanDefinition);

857				this.beanDefinitionNames.add(beanName);

858				this.manualSingletonNames.remove(beanName);

859			}

860			this.frozenBeanDefinitionNames = null;

861		}

862

863		if (oldBeanDefinition != null || containsSingleton(beanName)) {

864			resetBeanDefinition(beanName);

865		}

866	}

868	@Override

869	public void removeBeanDefinition(String beanName) throws NoSuchBeanDefinitionException {

870		Assert.hasText(beanName, "'beanName' must not be empty");

871

872		BeanDefinition bd = this.beanDefinitionMap.remove(beanName);

873		if (bd == null) {

874			if (this.logger.isTraceEnabled()) {

875				this.logger.trace("No bean named '" + beanName + "' found in " + this);

876			}

877			throw new NoSuchBeanDefinitionException(beanName);

878		}

879

880		if (hasBeanCreationStarted()) {

881			// Cannot modify startup-time collection elements anymore (for stable iteration)

882			synchronized (this.beanDefinitionMap) {

883				List<String> updatedDefinitions = new ArrayList<>(this.beanDefinitionNames);

884				updatedDefinitions.remove(beanName);

885				this.beanDefinitionNames = updatedDefinitions;

886			}

887		}

888		else {

889			// Still in startup registration phase

890			this.beanDefinitionNames.remove(beanName);

891		}

892		this.frozenBeanDefinitionNames = null;

893

894		resetBeanDefinition(beanName);

895	}

897	/**

898	 * Reset all bean definition caches for the given bean,

899	 * including the caches of beans that are derived from it.

900	 * @param beanName the name of the bean to reset

901	 */

902	protected void resetBeanDefinition(String beanName) {

903		// Remove the merged bean definition for the given bean, if already created.

904		clearMergedBeanDefinition(beanName);

905

906		// Remove corresponding bean from singleton cache, if any. Shouldn't usually

907		// be necessary, rather just meant for overriding a context's default beans

908		// (e.g. the default StaticMessageSource in a StaticApplicationContext).

909		destroySingleton(beanName);

910

911		// Reset all bean definitions that have the given bean as parent (recursively).

912		for (String bdName : this.beanDefinitionNames) {

913			if (!beanName.equals(bdName)) {

914				BeanDefinition bd = this.beanDefinitionMap.get(bdName);

915				if (beanName.equals(bd.getParentName())) {

916					resetBeanDefinition(bdName);

917				}

918			}

919		}

920	}

995	@SuppressWarnings("unchecked")

996	@Nullable

997	private <T> NamedBeanHolder<T> resolveNamedBean(Class<T> requiredType, @Nullable Object... args) throws BeansException {

998		Assert.notNull(requiredType, "Required type must not be null");

999		String[] candidateNames = getBeanNamesForType(requiredType);

1000

1001		if (candidateNames.length > 1) {

1002			List<String> autowireCandidates = new ArrayList<>(candidateNames.length);

1003			for (String beanName : candidateNames) {

1004				if (!containsBeanDefinition(beanName) || getBeanDefinition(beanName).isAutowireCandidate()) {

1005					autowireCandidates.add(beanName);

1006				}

1007			}

1008			if (!autowireCandidates.isEmpty()) {

1009				candidateNames = StringUtils.toStringArray(autowireCandidates);

1010			}

1011		}

1012

1013		if (candidateNames.length == 1) {

1014			String beanName = candidateNames[0];

1015			return new NamedBeanHolder<>(beanName, getBean(beanName, requiredType, args));

1016		}

1017		else if (candidateNames.length > 1) {

1018			Map<String, Object> candidates = new LinkedHashMap<>(candidateNames.length);

1019			for (String beanName : candidateNames) {

1020				if (containsSingleton(beanName) && args == null) {

1021					Object beanInstance = getBean(beanName);

1022					candidates.put(beanName, (beanInstance instanceof NullBean ? null : beanInstance));

1023				}

1024				else {

1025					candidates.put(beanName, getType(beanName));

1026				}

1027			}

1028			String candidateName = determinePrimaryCandidate(candidates, requiredType);

1029			if (candidateName == null) {

1030				candidateName = determineHighestPriorityCandidate(candidates, requiredType);

1031			}

1032			if (candidateName != null) {

1033				Object beanInstance = candidates.get(candidateName);

1034				if (beanInstance == null || beanInstance instanceof Class) {

1035					beanInstance = getBean(candidateName, requiredType, args);

1036				}

1037				return new NamedBeanHolder<>(candidateName, (T) beanInstance);

1038			}

1039			throw new NoUniqueBeanDefinitionException(requiredType, candidates.keySet());

1040		}

1041

1042		return null;

1043	}

1047	public Object resolveDependency(DependencyDescriptor descriptor, @Nullable String requestingBeanName,

1048			@Nullable Set<String> autowiredBeanNames, @Nullable TypeConverter typeConverter) throws BeansException {

1049

1050		descriptor.initParameterNameDiscovery(getParameterNameDiscoverer());

1051		if (Optional.class == descriptor.getDependencyType()) {

1052			return createOptionalDependency(descriptor, requestingBeanName);

1055				ObjectProvider.class == descriptor.getDependencyType()) {

1056			return new DependencyObjectProvider(descriptor, requestingBeanName);

1058		else if (javaxInjectProviderClass == descriptor.getDependencyType()) {

1059			return new Jsr330ProviderFactory().createDependencyProvider(descriptor, requestingBeanName);

1060		}

1061		else {

1062			Object result = getAutowireCandidateResolver().getLazyResolutionProxyIfNecessary(

1063					descriptor, requestingBeanName);

1064			if (result == null) {

1065				result = doResolveDependency(descriptor, requestingBeanName, autowiredBeanNames, typeConverter);

1066			}

1067			return result;

1068		}

1071	@Nullable

1072	public Object doResolveDependency(DependencyDescriptor descriptor, @Nullable String beanName,

1073			@Nullable Set<String> autowiredBeanNames, @Nullable TypeConverter typeConverter) throws BeansException {

1074

1075		InjectionPoint previousInjectionPoint = ConstructorResolver.setCurrentInjectionPoint(descriptor);

1076		try {

1077			Object shortcut = descriptor.resolveShortcut(this);

1078			if (shortcut != null) {

1079				return shortcut;

1080			}

1081

1082			Class<?> type = descriptor.getDependencyType();

1083			Object value = getAutowireCandidateResolver().getSuggestedValue(descriptor);

1084			if (value != null) {

1085				if (value instanceof String) {

1086					String strVal = resolveEmbeddedValue((String) value);

1087					BeanDefinition bd = (beanName != null && containsBean(beanName) ? getMergedBeanDefinition(beanName) : null);

1088					value = evaluateBeanDefinitionString(strVal, bd);

1089				}

1090				TypeConverter converter = (typeConverter != null ? typeConverter : getTypeConverter());

1091				return (descriptor.getField() != null ?

1092						converter.convertIfNecessary(value, type, descriptor.getField()) :

1093						converter.convertIfNecessary(value, type, descriptor.getMethodParameter()));

1094			}

1095

1096			Object multipleBeans = resolveMultipleBeans(descriptor, beanName, autowiredBeanNames, typeConverter);

1097			if (multipleBeans != null) {

1098				return multipleBeans;

1099			}

1100

1101			Map<String, Object> matchingBeans = findAutowireCandidates(beanName, type, descriptor);

1102			if (matchingBeans.isEmpty()) {

1103				if (isRequired(descriptor)) {

1104					raiseNoMatchingBeanFound(type, descriptor.getResolvableType(), descriptor);

1105				}

1106				return null;

1107			}

1108

1109			String autowiredBeanName;

1110			Object instanceCandidate;

1111

1112			if (matchingBeans.size() > 1) {

1113				autowiredBeanName = determineAutowireCandidate(matchingBeans, descriptor);

1114				if (autowiredBeanName == null) {

1115					if (isRequired(descriptor) || !indicatesMultipleBeans(type)) {

1116						return descriptor.resolveNotUnique(type, matchingBeans);

1117					}

1118					else {

1119						// In case of an optional Collection/Map, silently ignore a non-unique case:

1120						// possibly it was meant to be an empty collection of multiple regular beans

1121						// (before 4.3 in particular when we didn't even look for collection beans).

1122						return null;

1123					}

1124				}

1125				instanceCandidate = matchingBeans.get(autowiredBeanName);

1126			}

1127			else {

1128				// We have exactly one match.

1129				Map.Entry<String, Object> entry = matchingBeans.entrySet().iterator().next();

1130				autowiredBeanName = entry.getKey();

1131				instanceCandidate = entry.getValue();

1132			}

1133

1134			if (autowiredBeanNames != null) {

1135				autowiredBeanNames.add(autowiredBeanName);

1136			}

1137			if (instanceCandidate instanceof Class) {

1138				instanceCandidate = descriptor.resolveCandidate(autowiredBeanName, type, this);

1139			}

1140			Object result = instanceCandidate;

1141			if (result instanceof NullBean) {

1142				if (isRequired(descriptor)) {

1143					raiseNoMatchingBeanFound(type, descriptor.getResolvableType(), descriptor);

1144				}

1145				result = null;

1146			}

1147			if (!ClassUtils.isAssignableValue(type, result)) {

1148				throw new BeanNotOfRequiredTypeException(autowiredBeanName, type, instanceCandidate.getClass());

1149			}

1150			return result;

1151		}

1152		finally {

1153			ConstructorResolver.setCurrentInjectionPoint(previousInjectionPoint);

1154		}

1155	}

1157	@Nullable

1158	private Object resolveMultipleBeans(DependencyDescriptor descriptor, @Nullable String beanName,

1159			@Nullable Set<String> autowiredBeanNames, @Nullable TypeConverter typeConverter) {

1160

1161		Class<?> type = descriptor.getDependencyType();

1162		if (type.isArray()) {

1163			Class<?> componentType = type.getComponentType();

1164			ResolvableType resolvableType = descriptor.getResolvableType();

1165			Class<?> resolvedArrayType = resolvableType.resolve();

1166			if (resolvedArrayType != null && resolvedArrayType != type) {

1167				type = resolvedArrayType;

1168				componentType = resolvableType.getComponentType().resolve();

1169			}

1170			if (componentType == null) {

1171				return null;

1172			}

1173			Map<String, Object> matchingBeans = findAutowireCandidates(beanName, componentType,

1174					new MultiElementDescriptor(descriptor));

1175			if (matchingBeans.isEmpty()) {

1176				return null;

1177			}

1178			if (autowiredBeanNames != null) {

1179				autowiredBeanNames.addAll(matchingBeans.keySet());

1180			}

1181			TypeConverter converter = (typeConverter != null ? typeConverter : getTypeConverter());

1182			Object result = converter.convertIfNecessary(matchingBeans.values(), type);

1183			if (getDependencyComparator() != null && result instanceof Object[]) {

1184				Arrays.sort((Object[]) result, adaptDependencyComparator(matchingBeans));

1185			}

1186			return result;

1187		}

1188		else if (Collection.class.isAssignableFrom(type) && type.isInterface()) {

1189			Class<?> elementType = descriptor.getResolvableType().asCollection().resolveGeneric();

1190			if (elementType == null) {

1191				return null;

1192			}

1193			Map<String, Object> matchingBeans = findAutowireCandidates(beanName, elementType,

1194					new MultiElementDescriptor(descriptor));

1195			if (matchingBeans.isEmpty()) {

1196				return null;

1197			}

1198			if (autowiredBeanNames != null) {

1199				autowiredBeanNames.addAll(matchingBeans.keySet());

1200			}

1201			TypeConverter converter = (typeConverter != null ? typeConverter : getTypeConverter());

1202			Object result = converter.convertIfNecessary(matchingBeans.values(), type);

1203			if (getDependencyComparator() != null && result instanceof List) {

1204				((List<?>) result).sort(adaptDependencyComparator(matchingBeans));

1205			}

1206			return result;

1207		}

1208		else if (Map.class == type) {

1209			ResolvableType mapType = descriptor.getResolvableType().asMap();

1210			Class<?> keyType = mapType.resolveGeneric(0);

1211			if (String.class != keyType) {

1212				return null;

1213			}

1214			Class<?> valueType = mapType.resolveGeneric(1);

1215			if (valueType == null) {

1216				return null;

1217			}

1218			Map<String, Object> matchingBeans = findAutowireCandidates(beanName, valueType,

1219					new MultiElementDescriptor(descriptor));

1220			if (matchingBeans.isEmpty()) {

1221				return null;

1222			}

1223			if (autowiredBeanNames != null) {

1224				autowiredBeanNames.addAll(matchingBeans.keySet());

1225			}

1226			return matchingBeans;

1227		}

1228		else {

1229			return null;

1230		}

1231	}

1260	/**

1261	 * Find bean instances that match the required type.

1262	 * Called during autowiring for the specified bean.

1263	 * @param beanName the name of the bean that is about to be wired

1264	 * @param requiredType the actual type of bean to look for

1265	 * (may be an array component type or collection element type)

1266	 * @param descriptor the descriptor of the dependency to resolve

1267	 * @return a Map of candidate names and candidate instances that match

1268	 * the required type (never {@code null})

1269	 * @throws BeansException in case of errors

1270	 * @see #autowireByType

1271	 * @see #autowireConstructor

1272	 */

1273	protected Map<String, Object> findAutowireCandidates(

1274			@Nullable String beanName, Class<?> requiredType, DependencyDescriptor descriptor) {

1275

1276		String[] candidateNames = BeanFactoryUtils.beanNamesForTypeIncludingAncestors(

1277				this, requiredType, true, descriptor.isEager());

1278		Map<String, Object> result = new LinkedHashMap<>(candidateNames.length);

1279		for (Class<?> autowiringType : this.resolvableDependencies.keySet()) {

1280			if (autowiringType.isAssignableFrom(requiredType)) {

1281				Object autowiringValue = this.resolvableDependencies.get(autowiringType);

1282				autowiringValue = AutowireUtils.resolveAutowiringValue(autowiringValue, requiredType);

1283				if (requiredType.isInstance(autowiringValue)) {

1284					result.put(ObjectUtils.identityToString(autowiringValue), autowiringValue);

1285					break;

1286				}

1287			}

1288		}

1289		for (String candidate : candidateNames) {

1290			if (!isSelfReference(beanName, candidate) && isAutowireCandidate(candidate, descriptor)) {

1291				addCandidateEntry(result, candidate, descriptor, requiredType);

1292			}

1293		}

1294		if (result.isEmpty() && !indicatesMultipleBeans(requiredType)) {

1295			// Consider fallback matches if the first pass failed to find anything...

1296			DependencyDescriptor fallbackDescriptor = descriptor.forFallbackMatch();

1297			for (String candidate : candidateNames) {

1298				if (!isSelfReference(beanName, candidate) && isAutowireCandidate(candidate, fallbackDescriptor)) {

1299					addCandidateEntry(result, candidate, descriptor, requiredType);

1300				}

1301			}

1302			if (result.isEmpty()) {

1303				// Consider self references as a final pass...

1304				// but in the case of a dependency collection, not the very same bean itself.

1305				for (String candidate : candidateNames) {

1306					if (isSelfReference(beanName, candidate) &&

1307							(!(descriptor instanceof MultiElementDescriptor) || !beanName.equals(candidate)) &&

1308							isAutowireCandidate(candidate, fallbackDescriptor)) {

1309						addCandidateEntry(result, candidate, descriptor, requiredType);

1310					}

1311				}

1312			}

1313		}

1314		return result;

1315	}

1317	/**

1318	 * Add an entry to the candidate map: a bean instance if available or just the resolved

1319	 * type, preventing early bean initialization ahead of primary candidate selection.

1320	 */

1321	private void addCandidateEntry(Map<String, Object> candidates, String candidateName,

1322			DependencyDescriptor descriptor, Class<?> requiredType) {

1323

1324		if (descriptor instanceof MultiElementDescriptor || containsSingleton(candidateName)) {

1325			Object beanInstance = descriptor.resolveCandidate(candidateName, requiredType, this);

1326			candidates.put(candidateName, (beanInstance instanceof NullBean ? null : beanInstance));

1327		}

1328		else {

1329			candidates.put(candidateName, getType(candidateName));

1330		}

1331	}

1333	/**

1334	 * Determine the autowire candidate in the given set of beans.

1335	 * <p>Looks for {@code @Primary} and {@code @Priority} (in that order).

1336	 * @param candidates a Map of candidate names and candidate instances

1337	 * that match the required type, as returned by {@link #findAutowireCandidates}

1338	 * @param descriptor the target dependency to match against

1339	 * @return the name of the autowire candidate, or {@code null} if none found

1340	 */

1341	@Nullable

1342	protected String determineAutowireCandidate(Map<String, Object> candidates, DependencyDescriptor descriptor) {

1343		Class<?> requiredType = descriptor.getDependencyType();

1344		String primaryCandidate = determinePrimaryCandidate(candidates, requiredType);

1345		if (primaryCandidate != null) {

1346			return primaryCandidate;

1347		}

1348		String priorityCandidate = determineHighestPriorityCandidate(candidates, requiredType);

1349		if (priorityCandidate != null) {

1350			return priorityCandidate;

1351		}

1352		// Fallback

1353		for (Map.Entry<String, Object> entry : candidates.entrySet()) {

1354			String candidateName = entry.getKey();

1355			Object beanInstance = entry.getValue();

1356			if ((beanInstance != null && this.resolvableDependencies.containsValue(beanInstance)) ||

1357					matchesBeanName(candidateName, descriptor.getDependencyName())) {

1358				return candidateName;

1359			}

1360		}

1361		return null;

1362	}

1364	/**

1365	 * Determine the primary candidate in the given set of beans.

1366	 * @param candidates a Map of candidate names and candidate instances

1367	 * (or candidate classes if not created yet) that match the required type

1368	 * @param requiredType the target dependency type to match against

1369	 * @return the name of the primary candidate, or {@code null} if none found

1370	 * @see #isPrimary(String, Object)

1371	 */

1372	@Nullable

1373	protected String determinePrimaryCandidate(Map<String, Object> candidates, Class<?> requiredType) {

1374		String primaryBeanName = null;

1375		for (Map.Entry<String, Object> entry : candidates.entrySet()) {

1376			String candidateBeanName = entry.getKey();

1377			Object beanInstance = entry.getValue();

1378			if (isPrimary(candidateBeanName, beanInstance)) {

1379				if (primaryBeanName != null) {

1380					boolean candidateLocal = containsBeanDefinition(candidateBeanName);

1381					boolean primaryLocal = containsBeanDefinition(primaryBeanName);

1382					if (candidateLocal && primaryLocal) {

1383						throw new NoUniqueBeanDefinitionException(requiredType, candidates.size(),

1384								"more than one 'primary' bean found among candidates: " + candidates.keySet());

1385					}

1386					else if (candidateLocal) {

1387						primaryBeanName = candidateBeanName;

1388					}

1389				}

1390				else {

1391					primaryBeanName = candidateBeanName;

1392				}

1393			}

1394		}

1395		return primaryBeanName;

1396	}

1398	/**

1399	 * Determine the candidate with the highest priority in the given set of beans.

1400	 * <p>Based on {@code @javax.annotation.Priority}. As defined by the related

1401	 * {@link org.springframework.core.Ordered} interface, the lowest value has

1402	 * the highest priority.

1403	 * @param candidates a Map of candidate names and candidate instances

1404	 * (or candidate classes if not created yet) that match the required type

1405	 * @param requiredType the target dependency type to match against

1406	 * @return the name of the candidate with the highest priority,

1407	 * or {@code null} if none found

1408	 * @see #getPriority(Object)

1409	 */

1410	@Nullable

1411	protected String determineHighestPriorityCandidate(Map<String, Object> candidates, Class<?> requiredType) {

1412		String highestPriorityBeanName = null;

1413		Integer highestPriority = null;

1414		for (Map.Entry<String, Object> entry : candidates.entrySet()) {

1415			String candidateBeanName = entry.getKey();

1416			Object beanInstance = entry.getValue();

1417			if (beanInstance != null) {

1418				Integer candidatePriority = getPriority(beanInstance);

1419				if (candidatePriority != null) {

1420					if (highestPriorityBeanName != null) {

1421						if (candidatePriority.equals(highestPriority)) {

1422							throw new NoUniqueBeanDefinitionException(requiredType, candidates.size(),

1423									"Multiple beans found with the same priority ('" + highestPriority +

1424									"') among candidates: " + candidates.keySet());

1425						}

1426						else if (candidatePriority < highestPriority) {

1427							highestPriorityBeanName = candidateBeanName;

1428							highestPriority = candidatePriority;

1429						}

1430					}

1431					else {

1432						highestPriorityBeanName = candidateBeanName;

1433						highestPriority = candidatePriority;

1434					}

1435				}

1436			}

1437		}

1438		return highestPriorityBeanName;

1439	}

1443	 * marked as a primary bean.

1444	 * @param beanName the name of the bean

1445	 * @param beanInstance the corresponding bean instance (can be null)

1458	 * Return the priority assigned for the given bean instance by

1459	 * the {@code javax.annotation.Priority} annotation.

1460	 * <p>The default implementation delegates to the specified

1461	 * {@link #setDependencyComparator dependency comparator}, checking its

1462	 * {@link OrderComparator#getPriority method} if it is an extension of

1463	 * Spring's common {@link OrderComparator} - typically, an

1464	 * {@link org.springframework.core.annotation.AnnotationAwareOrderComparator}.

1512	/**

1513	 * Raise a BeanNotOfRequiredTypeException for an unresolvable dependency, if applicable,

1514	 * i.e. if the target type of the bean would match but an exposed proxy doesn't.

1515	 */

1516	private void checkBeanNotOfRequiredType(Class<?> type, DependencyDescriptor descriptor) {

1517		for (String beanName : this.beanDefinitionNames) {

1518			RootBeanDefinition mbd = getMergedLocalBeanDefinition(beanName);

1519			Class<?> targetType = mbd.getTargetType();

1520			if (targetType != null && type.isAssignableFrom(targetType) &&

1521					isAutowireCandidate(beanName, mbd, descriptor, getAutowireCandidateResolver())) {

1522				// Probably a proxy interfering with target type match -> throw meaningful exception.

1523				Object beanInstance = getSingleton(beanName, false);

1524				Class<?> beanType = (beanInstance != null && beanInstance.getClass() != NullBean.class) ?

1525						beanInstance.getClass() : predictBeanType(beanName, mbd);

1526				if (beanType != null && !type.isAssignableFrom(beanType)) {

1527					throw new BeanNotOfRequiredTypeException(beanName, type, beanType);

1528				}

1529			}

1530		}

1531

1532		BeanFactory parent = getParentBeanFactory();

1533		if (parent instanceof DefaultListableBeanFactory) {

1534			((DefaultListableBeanFactory) parent).checkBeanNotOfRequiredType(type, descriptor);

1535		}

1536	}
