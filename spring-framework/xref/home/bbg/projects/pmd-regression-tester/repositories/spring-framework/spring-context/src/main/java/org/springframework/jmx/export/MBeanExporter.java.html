72/**

73 * JMX exporter that allows for exposing any <i>Spring-managed bean</i> to a

74 * JMX {@link javax.management.MBeanServer}, without the need to define any

75 * JMX-specific information in the bean classes.

76 *

77 * <p>If a bean implements one of the JMX management interfaces, MBeanExporter can

78 * simply register the MBean with the server through its autodetection process.

79 *

80 * <p>If a bean does not implement one of the JMX management interfaces, MBeanExporter

81 * will create the management information using the supplied {@link MBeanInfoAssembler}.

82 *

83 * <p>A list of {@link MBeanExporterListener MBeanExporterListeners} can be registered

84 * via the {@link #setListeners(MBeanExporterListener[]) listeners} property, allowing

85 * application code to be notified of MBean registration and unregistration events.

86 *

87 * <p>This exporter is compatible with MBeans as well as MXBeans.

88 *

89 * @author Rob Harrop

90 * @author Juergen Hoeller

91 * @author Rick Evans

92 * @author Mark Fisher

93 * @author Stephane Nicoll

94 * @since 1.2

95 * @see #setBeans

96 * @see #setAutodetect

97 * @see #setAssembler

98 * @see #setListeners

99 * @see org.springframework.jmx.export.assembler.MBeanInfoAssembler

100 * @see MBeanExporterListener

101 */

102public class MBeanExporter extends MBeanRegistrationSupport implements MBeanExportOperations,

103		BeanClassLoaderAware, BeanFactoryAware, InitializingBean, SmartInitializingSingleton, DisposableBean {

356	/**

357	 * Set the {@link NotificationListener NotificationListeners} to register

358	 * with the {@link javax.management.MBeanServer}.

359	 * <P>The key of each entry in the {@code Map} is a {@link String}

360	 * representation of the {@link javax.management.ObjectName} or the bean

361	 * name of the MBean the listener should be registered for. Specifying an

362	 * asterisk ({@code *}) for a key will cause the listener to be

363	 * associated with all MBeans registered by this class at startup time.

364	 * <p>The value of each entry is the

365	 * {@link javax.management.NotificationListener} to register. For more

366	 * advanced options such as registering

367	 * {@link javax.management.NotificationFilter NotificationFilters} and

368	 * handback objects see {@link #setNotificationListeners(NotificationListenerBean[])}.

369	 */

370	public void setNotificationListenerMappings(Map<?, ? extends NotificationListener> listeners) {

371		Assert.notNull(listeners, "'listeners' must not be null");

372		List<NotificationListenerBean> notificationListeners =

373				new ArrayList<>(listeners.size());

374

375		listeners.forEach((key, listener) -> {

376			// Get the listener from the map value.

377			NotificationListenerBean bean = new NotificationListenerBean(listener);

378			// Get the ObjectName from the map key.

379			if (key != null && !WILDCARD.equals(key)) {

380				// This listener is mapped to a specific ObjectName.

381				bean.setMappedObjectName(key);

382			}

383			notificationListeners.add(bean);

384		});

385

386		this.notificationListeners = notificationListeners.toArray(new NotificationListenerBean[0]);

387	}

461	@Override

462	public ObjectName registerManagedResource(Object managedResource) throws MBeanExportException {

463		Assert.notNull(managedResource, "Managed resource must not be null");

464		ObjectName objectName;

465		try {

466			objectName = getObjectName(managedResource, null);

467			if (this.ensureUniqueRuntimeObjectNames) {

468				objectName = JmxUtils.appendIdentityToObjectName(objectName, managedResource);

469			}

470		}

471		catch (Throwable ex) {

472			throw new MBeanExportException("Unable to generate ObjectName for MBean [" + managedResource + "]", ex);

473		}

474		registerManagedResource(managedResource, objectName);

475		return objectName;

476	}

477

478	@Override

479	public void registerManagedResource(Object managedResource, ObjectName objectName) throws MBeanExportException {

480		Assert.notNull(managedResource, "Managed resource must not be null");

481		Assert.notNull(objectName, "ObjectName must not be null");

482		try {

483			if (isMBean(managedResource.getClass())) {

484				doRegister(managedResource, objectName);

485			}

486			else {

487				ModelMBean mbean = createAndConfigureMBean(managedResource, managedResource.getClass().getName());

488				doRegister(mbean, objectName);

489				injectNotificationPublisherIfNecessary(managedResource, mbean, objectName);

490			}

491		}

492		catch (JMException ex) {

493			throw new UnableToRegisterMBeanException(

494					"Unable to register MBean [" + managedResource + "] with object name [" + objectName + "]", ex);

495		}

496	}

497

498	@Override

499	public void unregisterManagedResource(ObjectName objectName) {

500		Assert.notNull(objectName, "ObjectName must not be null");

501		doUnregister(objectName);

502	}

509	/**

510	 * Register the defined beans with the {@link MBeanServer}.

511	 * <p>Each bean is exposed to the {@code MBeanServer} via a

512	 * {@code ModelMBean}. The actual implemetation of the

513	 * {@code ModelMBean} interface used depends on the implementation of

514	 * the {@code ModelMBeanProvider} interface that is configured. By

515	 * default the {@code RequiredModelMBean} class that is supplied with

516	 * all JMX implementations is used.

517	 * <p>The management interface produced for each bean is dependent on the

518	 * {@code MBeanInfoAssembler} implementation being used. The

519	 * {@code ObjectName} given to each bean is dependent on the

520	 * implementation of the {@code ObjectNamingStrategy} interface being used.

521	 */

522	protected void registerBeans() {

523		// The beans property may be null, for example if we are relying solely on autodetection.

524		if (this.beans == null) {

525			this.beans = new HashMap<>();

526			// Use AUTODETECT_ALL as default in no beans specified explicitly.

527			if (this.autodetectMode == null) {

528				this.autodetectMode = AUTODETECT_ALL;

529			}

530		}

531

532		// Perform autodetection, if desired.

533		int mode = (this.autodetectMode != null ? this.autodetectMode : AUTODETECT_NONE);

534		if (mode != AUTODETECT_NONE) {

535			if (this.beanFactory == null) {

536				throw new MBeanExportException("Cannot autodetect MBeans if not running in a BeanFactory");

537			}

538			if (mode == AUTODETECT_MBEAN || mode == AUTODETECT_ALL) {

539				// Autodetect any beans that are already MBeans.

540				logger.debug("Autodetecting user-defined JMX MBeans");

541				autodetect(this.beans, (beanClass, beanName) -> isMBean(beanClass));

542			}

543			// Allow the assembler a chance to vote for bean inclusion.

544			if ((mode == AUTODETECT_ASSEMBLER || mode == AUTODETECT_ALL) &&

545					this.assembler instanceof AutodetectCapableMBeanInfoAssembler) {

546				autodetect(this.beans, ((AutodetectCapableMBeanInfoAssembler) this.assembler)::includeBean);

547			}

548		}

549

550		if (!this.beans.isEmpty()) {

551			this.beans.forEach((beanName, instance) -> registerBeanNameOrInstance(instance, beanName));

552		}

553	}

555	/**

556	 * Return whether the specified bean definition should be considered as lazy-init.

557	 * @param beanFactory the bean factory that is supposed to contain the bean definition

558	 * @param beanName the name of the bean to check

559	 * @see org.springframework.beans.factory.config.ConfigurableListableBeanFactory#getBeanDefinition

560	 * @see org.springframework.beans.factory.config.BeanDefinition#isLazyInit

561	 */

562	protected boolean isBeanDefinitionLazyInit(ListableBeanFactory beanFactory, String beanName) {

563		return (beanFactory instanceof ConfigurableListableBeanFactory && beanFactory.containsBeanDefinition(beanName) &&

564				((ConfigurableListableBeanFactory) beanFactory).getBeanDefinition(beanName).isLazyInit());

565	}

567	/**

568	 * Register an individual bean with the {@link #setServer MBeanServer}.

569	 * <p>This method is responsible for deciding <strong>how</strong> a bean

570	 * should be exposed to the {@code MBeanServer}. Specifically, if the

571	 * supplied {@code mapValue} is the name of a bean that is configured

572	 * for lazy initialization, then a proxy to the resource is registered with

573	 * the {@code MBeanServer} so that the lazy load behavior is

574	 * honored. If the bean is already an MBean then it will be registered

575	 * directly with the {@code MBeanServer} without any intervention. For

576	 * all other beans or bean names, the resource itself is registered with

577	 * the {@code MBeanServer} directly.

578	 * @param mapValue the value configured for this bean in the beans map;

579	 * may be either the {@code String} name of a bean, or the bean itself

580	 * @param beanKey the key associated with this bean in the beans map

581	 * @return the {@code ObjectName} under which the resource was registered

582	 * @throws MBeanExportException if the export failed

583	 * @see #setBeans

584	 * @see #registerBeanInstance

585	 * @see #registerLazyInit

586	 */

587	protected ObjectName registerBeanNameOrInstance(Object mapValue, String beanKey) throws MBeanExportException {

588		try {

589			if (mapValue instanceof String) {

590				// Bean name pointing to a potentially lazy-init bean in the factory.

591				if (this.beanFactory == null) {

592					throw new MBeanExportException("Cannot resolve bean names if not running in a BeanFactory");

593				}

594				String beanName = (String) mapValue;

595				if (isBeanDefinitionLazyInit(this.beanFactory, beanName)) {

596					ObjectName objectName = registerLazyInit(beanName, beanKey);

597					replaceNotificationListenerBeanNameKeysIfNecessary(beanName, objectName);

598					return objectName;

599				}

600				else {

601					Object bean = this.beanFactory.getBean(beanName);

602					ObjectName objectName = registerBeanInstance(bean, beanKey);

603					replaceNotificationListenerBeanNameKeysIfNecessary(beanName, objectName);

604					return objectName;

605				}

606			}

607			else {

608				// Plain bean instance -> register it directly.

609				if (this.beanFactory != null) {

610					Map<String, ?> beansOfSameType =

611							this.beanFactory.getBeansOfType(mapValue.getClass(), false, this.allowEagerInit);

612					for (Map.Entry<String, ?> entry : beansOfSameType.entrySet()) {

613						if (entry.getValue() == mapValue) {

614							String beanName = entry.getKey();

615							ObjectName objectName = registerBeanInstance(mapValue, beanKey);

616							replaceNotificationListenerBeanNameKeysIfNecessary(beanName, objectName);

617							return objectName;

618						}

619					}

620				}

621				return registerBeanInstance(mapValue, beanKey);

622			}

623		}

624		catch (Throwable ex) {

625			throw new UnableToRegisterMBeanException(

626					"Unable to register MBean [" + mapValue + "] with key '" + beanKey + "'", ex);

627		}

628	}

630	/**

631	 * Replace any bean names used as keys in the {@code NotificationListener}

632	 * mappings with their corresponding {@code ObjectName} values.

633	 * @param beanName the name of the bean to be registered

634	 * @param objectName the {@code ObjectName} under which the bean will be registered

635	 * with the {@code MBeanServer}

636	 */

638		if (this.notificationListeners != null) {

639			for (NotificationListenerBean notificationListener : this.notificationListeners) {

640				notificationListener.replaceObjectName(beanName, objectName);

641			}

642		}

645	/**

646	 * Registers an existing MBean or an MBean adapter for a plain bean

647	 * with the {@code MBeanServer}.

648	 * @param bean the bean to register, either an MBean or a plain bean

649	 * @param beanKey the key associated with this bean in the beans map

650	 * @return the {@code ObjectName} under which the bean was registered

651	 * with the {@code MBeanServer}

652	 */

653	private ObjectName registerBeanInstance(Object bean, String beanKey) throws JMException {

654		ObjectName objectName = getObjectName(bean, beanKey);

655		Object mbeanToExpose = null;

656		if (isMBean(bean.getClass())) {

657			mbeanToExpose = bean;

658		}

659		else {

660			DynamicMBean adaptedBean = adaptMBeanIfPossible(bean);

661			if (adaptedBean != null) {

662				mbeanToExpose = adaptedBean;

663			}

664		}

665

666		if (mbeanToExpose != null) {

667			if (logger.isInfoEnabled()) {

668				logger.info("Located MBean '" + beanKey + "': registering with JMX server as MBean [" +

669						objectName + "]");

670			}

671			doRegister(mbeanToExpose, objectName);

672		}

673		else {

674			if (logger.isInfoEnabled()) {

675				logger.info("Located managed bean '" + beanKey + "': registering with JMX server as MBean [" +

676						objectName + "]");

677			}

678			ModelMBean mbean = createAndConfigureMBean(bean, beanKey);

679			doRegister(mbean, objectName);

680			injectNotificationPublisherIfNecessary(bean, mbean, objectName);

681		}

682

683		return objectName;

684	}

686	/**

687	 * Register beans that are configured for lazy initialization with the

688	 * {@code MBeanServer} indirectly through a proxy.

689	 * @param beanName the name of the bean in the {@code BeanFactory}

690	 * @param beanKey the key associated with this bean in the beans map

691	 * @return the {@code ObjectName} under which the bean was registered

692	 * with the {@code MBeanServer}

693	 */

694	private ObjectName registerLazyInit(String beanName, String beanKey) throws JMException {

695		Assert.state(this.beanFactory != null, "No BeanFactory set");

696

697		ProxyFactory proxyFactory = new ProxyFactory();

698		proxyFactory.setProxyTargetClass(true);

699		proxyFactory.setFrozen(true);

700

701		if (isMBean(this.beanFactory.getType(beanName))) {

702			// A straight MBean... Let's create a simple lazy-init CGLIB proxy for it.

703			LazyInitTargetSource targetSource = new LazyInitTargetSource();

704			targetSource.setTargetBeanName(beanName);

705			targetSource.setBeanFactory(this.beanFactory);

706			proxyFactory.setTargetSource(targetSource);

707

708			Object proxy = proxyFactory.getProxy(this.beanClassLoader);

709			ObjectName objectName = getObjectName(proxy, beanKey);

710			if (logger.isDebugEnabled()) {

711				logger.debug("Located MBean '" + beanKey + "': registering with JMX server as lazy-init MBean [" +

712						objectName + "]");

713			}

714			doRegister(proxy, objectName);

715			return objectName;

716		}

717

718		else {

719			// A simple bean... Let's create a lazy-init ModelMBean proxy with notification support.

720			NotificationPublisherAwareLazyTargetSource targetSource = new NotificationPublisherAwareLazyTargetSource();

721			targetSource.setTargetBeanName(beanName);

722			targetSource.setBeanFactory(this.beanFactory);

723			proxyFactory.setTargetSource(targetSource);

724

725			Object proxy = proxyFactory.getProxy(this.beanClassLoader);

726			ObjectName objectName = getObjectName(proxy, beanKey);

727			if (logger.isDebugEnabled()) {

728				logger.debug("Located simple bean '" + beanKey + "': registering with JMX server as lazy-init MBean [" +

729						objectName + "]");

730			}

731			ModelMBean mbean = createAndConfigureMBean(proxy, beanKey);

732			targetSource.setModelMBean(mbean);

733			targetSource.setObjectName(objectName);

734			doRegister(mbean, objectName);

735			return objectName;

736		}

737	}

740	 * Retrieve the {@code ObjectName} for a bean.

741	 * <p>If the bean implements the {@code SelfNaming} interface, then the

742	 * {@code ObjectName} will be retrieved using {@code SelfNaming.getObjectName()}.

743	 * Otherwise, the configured {@code ObjectNamingStrategy} is used.

744	 * @param bean the name of the bean in the {@code BeanFactory}

745	 * @param beanKey the key associated with the bean in the beans map

746	 * @return the {@code ObjectName} for the supplied bean

747	 * @throws javax.management.MalformedObjectNameException

748	 * if the retrieved {@code ObjectName} is malformed

760	 * Determine whether the given bean class qualifies as an MBean as-is.

761	 * <p>The default implementation delegates to {@link JmxUtils#isMBean},

762	 * which checks for {@link javax.management.DynamicMBean} classes as well

763	 * as classes with corresponding "*MBean" interface (Standard MBeans)

764	 * or corresponding "*MXBean" interface (Java 6 MXBeans).

765	 * @param beanClass the bean class to analyze

766	 * @return whether the class qualifies as an MBean

767	 * @see org.springframework.jmx.support.JmxUtils#isMBean(Class)

773	/**

774	 * Build an adapted MBean for the given bean instance, if possible.

775	 * <p>The default implementation builds a JMX 1.2 StandardMBean

776	 * for the target's MBean/MXBean interface in case of an AOP proxy,

777	 * delegating the interface's management operations to the proxy.

778	 * @param bean the original bean instance

779	 * @return the adapted MBean, or {@code null} if not possible

780	 */

781	@SuppressWarnings("unchecked")

782	@Nullable

783	protected DynamicMBean adaptMBeanIfPossible(Object bean) throws JMException {

784		Class<?> targetClass = AopUtils.getTargetClass(bean);

785		if (targetClass != bean.getClass()) {

786			Class<?> ifc = JmxUtils.getMXBeanInterface(targetClass);

787			if (ifc != null) {

788				if (!ifc.isInstance(bean)) {

789					throw new NotCompliantMBeanException("Managed bean [" + bean +

790							"] has a target class with an MXBean interface but does not expose it in the proxy");

791				}

792				return new StandardMBean(bean, ((Class<Object>) ifc), true);

793			}

794			else {

795				ifc = JmxUtils.getMBeanInterface(targetClass);

796				if (ifc != null) {

797					if (!ifc.isInstance(bean)) {

798						throw new NotCompliantMBeanException("Managed bean [" + bean +

799								"] has a target class with an MBean interface but does not expose it in the proxy");

800					}

801					return new StandardMBean(bean, ((Class<Object>) ifc));

802				}

803			}

804		}

805		return null;

806	}

809	 * Creates an MBean that is configured with the appropriate management

810	 * interface for the supplied managed resource.

811	 * @param managedResource the resource that is to be exported as an MBean

812	 * @param beanKey the key associated with the managed bean

813	 * @see #createModelMBean()

814	 * @see #getMBeanInfo(Object, String)

824		catch (Throwable ex) {

826					managedResource + "] with key '" + beanKey + "'", ex);

832	 * <p>This method is called to obtain a {@code ModelMBean} instance to

833	 * use when registering a bean. This method is called once per bean during the

834	 * registration phase and must return a new instance of {@code ModelMBean}

843	 * Gets the {@code ModelMBeanInfo} for the bean with the supplied key

844	 * and of the supplied type.

845	 */

846	private ModelMBeanInfo getMBeanInfo(Object managedBean, String beanKey) throws JMException {

861	/**

862	 * Performs the actual autodetection process, delegating to an

863	 * {@code AutodetectCallback} instance to vote on the inclusion of a

864	 * given bean.

865	 * @param callback the {@code AutodetectCallback} to use when deciding

866	 * whether to include a bean or not

867	 */

868	private void autodetect(Map<String, Object> beans, AutodetectCallback callback) {

869		Assert.state(this.beanFactory != null, "No BeanFactory set");

870		Set<String> beanNames = new LinkedHashSet<>(this.beanFactory.getBeanDefinitionCount());

871		Collections.addAll(beanNames, this.beanFactory.getBeanDefinitionNames());

872		if (this.beanFactory instanceof ConfigurableBeanFactory) {

873			Collections.addAll(beanNames, ((ConfigurableBeanFactory) this.beanFactory).getSingletonNames());

874		}

875

876		for (String beanName : beanNames) {

877			if (!isExcluded(beanName) && !isBeanDefinitionAbstract(this.beanFactory, beanName)) {

878				try {

879					Class<?> beanClass = this.beanFactory.getType(beanName);

880					if (beanClass != null && callback.include(beanClass, beanName)) {

881						boolean lazyInit = isBeanDefinitionLazyInit(this.beanFactory, beanName);

882						Object beanInstance = null;

883						if (!lazyInit) {

884							beanInstance = this.beanFactory.getBean(beanName);

885							if (!beanClass.isInstance(beanInstance)) {

886								continue;

887							}

888						}

889						if (!ScopedProxyUtils.isScopedTarget(beanName) && !beans.containsValue(beanName) &&

890								(beanInstance == null ||

891										!CollectionUtils.containsInstance(beans.values(), beanInstance))) {

892							// Not already registered for JMX exposure.

893							beans.put(beanName, (beanInstance != null ? beanInstance : beanName));

894							if (logger.isInfoEnabled()) {

895								logger.info("Bean with name '" + beanName + "' has been autodetected for JMX exposure");

896							}

897						}

898						else {

899							if (logger.isDebugEnabled()) {

900								logger.debug("Bean with name '" + beanName + "' is already registered for JMX exposure");

901							}

902						}

903					}

904				}

905				catch (CannotLoadBeanClassException ex) {

906					if (this.allowEagerInit) {

907						throw ex;

908					}

909					// otherwise ignore beans where the class is not resolvable

910				}

911			}

912		}

913	}

915	/**

916	 * Indicates whether or not a particular bean name is present in the excluded beans list.

917	 */

918	private boolean isExcluded(String beanName) {

919		return (this.excludedBeans.contains(beanName) ||

920					(beanName.startsWith(BeanFactory.FACTORY_BEAN_PREFIX) &&

921							this.excludedBeans.contains(beanName.substring(BeanFactory.FACTORY_BEAN_PREFIX.length()))));

922	}

925	 * Return whether the specified bean definition should be considered as abstract.

926	 */

927	private boolean isBeanDefinitionAbstract(ListableBeanFactory beanFactory, String beanName) {

928		return (beanFactory instanceof ConfigurableListableBeanFactory && beanFactory.containsBeanDefinition(beanName) &&

929				((ConfigurableListableBeanFactory) beanFactory).getBeanDefinition(beanName).isAbstract());

950	/**

951	 * Register the configured {@link NotificationListener NotificationListeners}

952	 * with the {@link MBeanServer}.

953	 */

954	private void registerNotificationListeners() throws MBeanExportException {

955		if (this.notificationListeners != null) {

956			Assert.state(this.server != null, "No MBeanServer available");

957			for (NotificationListenerBean bean : this.notificationListeners) {

958				try {

959					ObjectName[] mappedObjectNames = bean.getResolvedObjectNames();

960					if (mappedObjectNames == null) {

961						// Mapped to all MBeans registered by the MBeanExporter.

962						mappedObjectNames = getRegisteredObjectNames();

963					}

964					if (this.registeredNotificationListeners.put(bean, mappedObjectNames) == null) {

965						for (ObjectName mappedObjectName : mappedObjectNames) {

966							this.server.addNotificationListener(mappedObjectName, bean.getNotificationListener(),

967									bean.getNotificationFilter(), bean.getHandback());

968						}

969					}

970				}

971				catch (Throwable ex) {

972					throw new MBeanExportException("Unable to register NotificationListener", ex);

973				}

974			}

975		}

976	}

977

978	/**

979	 * Unregister the configured {@link NotificationListener NotificationListeners}

980	 * from the {@link MBeanServer}.

981	 */

982	private void unregisterNotificationListeners() {

983		if (this.server != null) {

984			this.registeredNotificationListeners.forEach((bean, mappedObjectNames) -> {

985				for (ObjectName mappedObjectName : mappedObjectNames) {

986					try {

987						this.server.removeNotificationListener(mappedObjectName, bean.getNotificationListener(),

988								bean.getNotificationFilter(), bean.getHandback());

989					}

990					catch (Throwable ex) {

991						if (logger.isDebugEnabled()) {

992							logger.debug("Unable to unregister NotificationListener", ex);

993						}

994					}

995				}

996			});

997		}

998		this.registeredNotificationListeners.clear();

999	}
