45/**

46 * Helper class for populating a {@link javax.mail.internet.MimeMessage}.

47 *

48 * <p>Mirrors the simple setters of {@link org.springframework.mail.SimpleMailMessage},

49 * directly applying the values to the underlying MimeMessage. Allows for defining

50 * a character encoding for the entire message, automatically applied by all methods

51 * of this helper class.

52 *

53 * <p>Offers support for HTML text content, inline elements such as images, and typical

54 * mail attachments. Also supports personal names that accompany mail addresses. Note that

55 * advanced settings can still be applied directly to the underlying MimeMessage object!

56 *

57 * <p>Typically used in {@link MimeMessagePreparator} implementations or

58 * {@link JavaMailSender} client code: simply instantiating it as a MimeMessage wrapper,

59 * invoking setters on the wrapper, using the underlying MimeMessage for mail sending.

60 * Also used internally by {@link JavaMailSenderImpl}.

61 *

62 * <p>Sample code for an HTML mail with an inline image and a PDF attachment:

63 *

64 * <pre class="code">

65 * mailSender.send(new MimeMessagePreparator() {

66 *   public void prepare(MimeMessage mimeMessage) throws MessagingException {

67 *     MimeMessageHelper message = new MimeMessageHelper(mimeMessage, true, "UTF-8");

68 *     message.setFrom("me@mail.com");

69 *     message.setTo("you@mail.com");

70 *     message.setSubject("my subject");

71 *     message.setText("my text &lt;img src='cid:myLogo'&gt;", true);

72 *     message.addInline("myLogo", new ClassPathResource("img/mylogo.gif"));

73 *     message.addAttachment("myDocument.pdf", new ClassPathResource("doc/myDocument.pdf"));

74 *   }

75 * });</pre>

76 *

77 * Consider using {@link MimeMailMessage} (which implements the common

78 * {@link org.springframework.mail.MailMessage} interface, just like

79 * {@link org.springframework.mail.SimpleMailMessage}) on top of this helper,

80 * in order to let message population code interact with a simple message

81 * or a MIME message through a common interface.

82 *

83 * <p><b>Warning regarding multipart mails:</b> Simple MIME messages that

84 * just contain HTML text but no inline elements or attachments will work on

85 * more or less any email client that is capable of HTML rendering. However,

86 * inline elements and attachments are still a major compatibility issue

87 * between email clients: It's virtually impossible to get inline elements

88 * and attachments working across Microsoft Outlook, Lotus Notes and Mac Mail.

89 * Consider choosing a specific multipart mode for your needs: The javadoc

90 * on the MULTIPART_MODE constants contains more detailed information.

91 *

92 * @author Juergen Hoeller

93 * @since 19.01.2004

94 * @see #setText(String, boolean)

95 * @see #setText(String, String)

96 * @see #addInline(String, org.springframework.core.io.Resource)

97 * @see #addAttachment(String, org.springframework.core.io.InputStreamSource)

98 * @see #MULTIPART_MODE_MIXED_RELATED

99 * @see #MULTIPART_MODE_RELATED

100 * @see #getMimeMessage()

101 * @see JavaMailSender

102 */

105	/**

106	 * Constant indicating a non-multipart message.

107	 */

108	public static final int MULTIPART_MODE_NO = 0;

109

110	/**

111	 * Constant indicating a multipart message with a single root multipart

112	 * element of type "mixed". Texts, inline elements and attachements

113	 * will all get added to that root element.

114	 * <p>This was Spring 1.0's default behavior. It is known to work properly

115	 * on Outlook. However, other mail clients tend to misinterpret inline

116	 * elements as attachments and/or show attachments inline as well.

117	 */

118	public static final int MULTIPART_MODE_MIXED = 1;

119

120	/**

121	 * Constant indicating a multipart message with a single root multipart

122	 * element of type "related". Texts, inline elements and attachements

123	 * will all get added to that root element.

124	 * <p>This was the default behavior from Spring 1.1 up to 1.2 final.

125	 * This is the "Microsoft multipart mode", as natively sent by Outlook.

126	 * It is known to work properly on Outlook, Outlook Express, Yahoo Mail, and

127	 * to a large degree also on Mac Mail (with an additional attachment listed

128	 * for an inline element, despite the inline element also shown inline).

129	 * Does not work properly on Lotus Notes (attachments won't be shown there).

130	 */

131	public static final int MULTIPART_MODE_RELATED = 2;

132

133	/**

134	 * Constant indicating a multipart message with a root multipart element

135	 * "mixed" plus a nested multipart element of type "related". Texts and

136	 * inline elements will get added to the nested "related" element,

137	 * while attachments will get added to the "mixed" root element.

138	 * <p>This is the default since Spring 1.2.1. This is arguably the most correct

139	 * MIME structure, according to the MIME spec: It is known to work properly

140	 * on Outlook, Outlook Express, Yahoo Mail, and Lotus Notes. Does not work

141	 * properly on Mac Mail. If you target Mac Mail or experience issues with

142	 * specific mails on Outlook, consider using MULTIPART_MODE_RELATED instead.

143	 */

144	public static final int MULTIPART_MODE_MIXED_RELATED = 3;

308	/**

309	 * Determine the MimeMultipart objects to use, which will be used

310	 * to store attachments on the one hand and text(s) and inline elements

311	 * on the other hand.

312	 * <p>Texts and inline elements can either be stored in the root element

313	 * itself (MULTIPART_MODE_MIXED, MULTIPART_MODE_RELATED) or in a nested element

314	 * rather than the root element directly (MULTIPART_MODE_MIXED_RELATED).

315	 * <p>By default, the root MimeMultipart element will be of type "mixed"

316	 * (MULTIPART_MODE_MIXED) or "related" (MULTIPART_MODE_RELATED).

317	 * The main multipart element will either be added as nested element of

318	 * type "related" (MULTIPART_MODE_MIXED_RELATED) or be identical to the root

319	 * element itself (MULTIPART_MODE_MIXED, MULTIPART_MODE_RELATED).

320	 * @param mimeMessage the MimeMessage object to add the root MimeMultipart

321	 * object to

322	 * @param multipartMode the multipart mode, as passed into the constructor

323	 * (MIXED, RELATED, MIXED_RELATED, or NO)

324	 * @throws MessagingException if multipart creation failed

325	 * @see #setMimeMultiparts

326	 * @see #MULTIPART_MODE_NO

327	 * @see #MULTIPART_MODE_MIXED

328	 * @see #MULTIPART_MODE_RELATED

329	 * @see #MULTIPART_MODE_MIXED_RELATED

330	 */

331	protected void createMimeMultiparts(MimeMessage mimeMessage, int multipartMode) throws MessagingException {

332		switch (multipartMode) {

333			case MULTIPART_MODE_NO:

334				setMimeMultiparts(null, null);

335				break;

336			case MULTIPART_MODE_MIXED:

337				MimeMultipart mixedMultipart = new MimeMultipart(MULTIPART_SUBTYPE_MIXED);

338				mimeMessage.setContent(mixedMultipart);

339				setMimeMultiparts(mixedMultipart, mixedMultipart);

340				break;

341			case MULTIPART_MODE_RELATED:

342				MimeMultipart relatedMultipart = new MimeMultipart(MULTIPART_SUBTYPE_RELATED);

343				mimeMessage.setContent(relatedMultipart);

344				setMimeMultiparts(relatedMultipart, relatedMultipart);

345				break;

346			case MULTIPART_MODE_MIXED_RELATED:

347				MimeMultipart rootMixedMultipart = new MimeMultipart(MULTIPART_SUBTYPE_MIXED);

348				mimeMessage.setContent(rootMixedMultipart);

349				MimeMultipart nestedRelatedMultipart = new MimeMultipart(MULTIPART_SUBTYPE_RELATED);

350				MimeBodyPart relatedBodyPart = new MimeBodyPart();

351				relatedBodyPart.setContent(nestedRelatedMultipart);

352				rootMixedMultipart.addBodyPart(relatedBodyPart);

353				setMimeMultiparts(rootMixedMultipart, nestedRelatedMultipart);

354				break;

355			default:

356				throw new IllegalArgumentException("Only multipart modes MIXED_RELATED, RELATED and NO supported");

357		}

358	}

384	 * Can be used to manually add attachments.

385	 * <p>This will be the direct content of the MimeMessage,

386	 * in case of a multipart mail.

392	public final MimeMultipart getRootMimeMultipart() throws IllegalStateException {

393		if (this.rootMimeMultipart == null) {

394			throw new IllegalStateException("Not in multipart mode - " +

395					"create an appropriate MimeMessageHelper via a constructor that takes a 'multipart' flag " +

396					"if you need to set alternative texts or add inline elements or attachments.");

397		}

398		return this.rootMimeMultipart;

399	}

507	/**

508	 * Validate the given mail address.

509	 * Called by all of MimeMessageHelper's address setters and adders.

510	 * <p>Default implementation invokes {@code InternetAddress.validate()},

511	 * provided that address validation is activated for the helper instance.

512	 * <p>Note that this method will just work on JavaMail >= 1.3. You can override

513	 * it for validation on older JavaMail versions or for custom validation.

514	 * @param address the address to validate

515	 * @throws AddressException if validation failed

516	 * @see #isValidateAddresses()

517	 * @see javax.mail.internet.InternetAddress#validate()

518	 */

519	protected void validateAddress(InternetAddress address) throws AddressException {

520		if (isValidateAddresses()) {

521			address.validate();

522		}

523	}

525	/**

526	 * Validate all given mail addresses.

527	 * Default implementation simply delegates to validateAddress for each address.

528	 * @param addresses the addresses to validate

529	 * @throws AddressException if validation failed

530	 * @see #validateAddress(InternetAddress)

531	 */

532	protected void validateAddresses(InternetAddress[] addresses) throws AddressException {

533		for (InternetAddress address : addresses) {

534			validateAddress(address);

535		}

536	}

592	public void setTo(String[] to) throws MessagingException {

593		Assert.notNull(to, "To address array must not be null");

594		InternetAddress[] addresses = new InternetAddress[to.length];

595		for (int i = 0; i < to.length; i++) {

709	private InternetAddress parseAddress(String address) throws MessagingException {

710		InternetAddress[] parsed = InternetAddress.parse(address);

711		if (parsed.length != 1) {

712			throw new AddressException("Illegal address", address);

713		}

714		InternetAddress raw = parsed[0];

715		try {

716			return (getEncoding() != null ?

717					new InternetAddress(raw.getAddress(), raw.getPersonal(), getEncoding()) : raw);

718		}

719		catch (UnsupportedEncodingException ex) {

720			throw new MessagingException("Failed to parse embedded personal name to correct encoding", ex);

721		}

722	}

775	 * Set the given text directly as content in non-multipart mode

776	 * or as default body part in multipart mode.

777	 * The "html" flag determines the content type to apply.

778	 * <p><b>NOTE:</b> Invoke {@link #addInline} <i>after</i> {@code setText};

779	 * else, mail readers might not be able to resolve inline references correctly.

780	 * @param text the text for the message

781	 * @param html whether to apply content type "text/html" for an

782	 * HTML mail, using default content type ("text/plain") else

783	 * @throws MessagingException in case of errors

784	 */

785	public void setText(String text, boolean html) throws MessagingException {

786		Assert.notNull(text, "Text must not be null");

787		MimePart partToUse;

788		if (isMultipart()) {

789			partToUse = getMainPart();

790		}

791		else {

792			partToUse = this.mimeMessage;

793		}

794		if (html) {

795			setHtmlTextToMimePart(partToUse, text);

796		}

797		else {

798			setPlainTextToMimePart(partToUse, text);

799		}

800	}

802	/**

803	 * Set the given plain text and HTML text as alternatives, offering

804	 * both options to the email client. Requires multipart mode.

805	 * <p><b>NOTE:</b> Invoke {@link #addInline} <i>after</i> {@code setText};

806	 * else, mail readers might not be able to resolve inline references correctly.

807	 * @param plainText the plain text for the message

808	 * @param htmlText the HTML text for the message

809	 * @throws MessagingException in case of errors

810	 */

811	public void setText(String plainText, String htmlText) throws MessagingException {

812		Assert.notNull(plainText, "Plain text must not be null");

813		Assert.notNull(htmlText, "HTML text must not be null");

814

815		MimeMultipart messageBody = new MimeMultipart(MULTIPART_SUBTYPE_ALTERNATIVE);

816		getMainPart().setContent(messageBody, CONTENT_TYPE_ALTERNATIVE);

817

818		// Create the plain text part of the message.

819		MimeBodyPart plainTextPart = new MimeBodyPart();

820		setPlainTextToMimePart(plainTextPart, plainText);

821		messageBody.addBodyPart(plainTextPart);

822

823		// Create the HTML text part of the message.

824		MimeBodyPart htmlTextPart = new MimeBodyPart();

825		setHtmlTextToMimePart(htmlTextPart, htmlText);

826		messageBody.addBodyPart(htmlTextPart);

827	}

828

829	private MimeBodyPart getMainPart() throws MessagingException {

830		MimeMultipart mimeMultipart = getMimeMultipart();

831		MimeBodyPart bodyPart = null;

832		for (int i = 0; i < mimeMultipart.getCount(); i++) {

833			BodyPart bp = mimeMultipart.getBodyPart(i);

834			if (bp.getFileName() == null) {

835				bodyPart = (MimeBodyPart) bp;

836			}

837		}

838		if (bodyPart == null) {

839			MimeBodyPart mimeBodyPart = new MimeBodyPart();

840			mimeMultipart.addBodyPart(mimeBodyPart);

841			bodyPart = mimeBodyPart;

842		}

843		return bodyPart;

844	}

850		else {

866	 * Add an inline element to the MimeMessage, taking the content from a

867	 * {@code javax.activation.DataSource}.

868	 * <p>Note that the InputStream returned by the DataSource implementation

869	 * needs to be a <i>fresh one on each call</i>, as JavaMail will invoke

870	 * {@code getInputStream()} multiple times.

871	 * <p><b>NOTE:</b> Invoke {@code addInline} <i>after</i> {@link #setText};

872	 * else, mail readers might not be able to resolve inline references correctly.

873	 * @param contentId the content ID to use. Will end up as "Content-ID" header

874	 * in the body part, surrounded by angle brackets: e.g. "myId" -> "&lt;myId&gt;".

875	 * Can be referenced in HTML source via src="cid:myId" expressions.

876	 * @param dataSource the {@code javax.activation.DataSource} to take

877	 * the content from, determining the InputStream and the content type

878	 * @throws MessagingException in case of errors

879	 * @see #addInline(String, java.io.File)

880	 * @see #addInline(String, org.springframework.core.io.Resource)

881	 */

882	public void addInline(String contentId, DataSource dataSource) throws MessagingException {

883		Assert.notNull(contentId, "Content ID must not be null");

884		Assert.notNull(dataSource, "DataSource must not be null");

885		MimeBodyPart mimeBodyPart = new MimeBodyPart();

886		mimeBodyPart.setDisposition(MimeBodyPart.INLINE);

887		// We're using setHeader here to remain compatible with JavaMail 1.2,

888		// rather than JavaMail 1.3's setContentID.

889		mimeBodyPart.setHeader(HEADER_CONTENT_ID, "<" + contentId + ">");

890		mimeBodyPart.setDataHandler(new DataHandler(dataSource));

891		getMimeMultipart().addBodyPart(mimeBodyPart);

892	}

1083	/**

1084	 * Create an Activation Framework DataSource for the given InputStreamSource.

1085	 * @param inputStreamSource the InputStreamSource (typically a Spring Resource)

1086	 * @param contentType the content type

1087	 * @param name the name of the DataSource

1088	 * @return the Activation Framework DataSource

1089	 */

1090	protected DataSource createDataSource(

1091		final InputStreamSource inputStreamSource, final String contentType, final String name) {

1092

1093		return new DataSource() {

1094			@Override

1095			public InputStream getInputStream() throws IOException {

1096				return inputStreamSource.getInputStream();

1097			}

1098			@Override

1099			public OutputStream getOutputStream() {

1100				throw new UnsupportedOperationException("Read-only javax.activation.DataSource");

1101			}

1102			@Override

1103			public String getContentType() {

1104				return contentType;

1105			}

1106			@Override

1107			public String getName() {

1108				return name;

1109			}

1110		};
