51/**

52 * Represents HTTP request and response headers, mapping string header names to a list of string values.

53 *

54 * <p>In addition to the normal methods defined by {@link Map}, this class offers the following

55 * convenience methods:

56 * <ul>

57 * <li>{@link #getFirst(String)} returns the first value associated with a given header name</li>

58 * <li>{@link #add(String, String)} adds a header value to the list of values for a header name</li>

59 * <li>{@link #set(String, String)} sets the header value to a single string value</li>

60 * </ul>

61 *

62 * <p>Inspired by {@code com.sun.net.httpserver.Headers}.

65 * @author Sebastien Deleuze

406	/**

407	 * Constructs a new, empty instance of the {@code HttpHeaders} object.

408	 */

409	public HttpHeaders() {

410		this(new LinkedCaseInsensitiveMap<>(8, Locale.ENGLISH), false);

411	}

412

413	/**

414	 * Private constructor that can create read-only {@code HttpHeader} instances.

415	 */

416	private HttpHeaders(Map<String, List<String>> headers, boolean readOnly) {

417		Assert.notNull(headers, "'headers' must not be null");

418		if (readOnly) {

419			Map<String, List<String>> map =

420					new LinkedCaseInsensitiveMap<>(headers.size(), Locale.ENGLISH);

421			for (Entry<String, List<String>> entry : headers.entrySet()) {

422				List<String> values = Collections.unmodifiableList(entry.getValue());

423				map.put(entry.getKey(), values);

424			}

425			this.headers = Collections.unmodifiableMap(map);

426		}

427		else {

428			this.headers = headers;

429		}

430		this.readOnly = readOnly;

431	}

451	/**

452	 * Set the acceptable language ranges, as specified by the

453	 * {@literal Accept-Language} header.

454	 * @since 5.0

455	 */

456	public void setAcceptLanguage(List<Locale.LanguageRange> languages) {

457		Assert.notNull(languages, "'languages' must not be null");

458		DecimalFormat decimal = new DecimalFormat("0.0", DECIMAL_FORMAT_SYMBOLS);

459		List<String> values = languages.stream()

460				.map(range ->

461						range.getWeight() == Locale.LanguageRange.MAX_WEIGHT ?

462								range.getRange() :

463								range.getRange() + ";q=" + decimal.format(range.getWeight()))

464				.collect(Collectors.toList());

465		set(ACCEPT_LANGUAGE, toCommaDelimitedString(values));

466	}

468	/**

469	 * Return the language ranges from the {@literal "Accept-Language"} header.

470	 * <p>If you only need sorted, preferred locales only use

471	 * {@link #getAcceptLanguageAsLocales()} or if you need to filter based on

472	 * a list of supported locales you can pass the returned list to

473	 * {@link Locale#filter(List, Collection)}.

474	 * @throws IllegalArgumentException if the value cannot be converted to a language range

475	 * @since 5.0

476	 */

477	public List<Locale.LanguageRange> getAcceptLanguage() {

478		String value = getFirst(ACCEPT_LANGUAGE);

479		return (StringUtils.hasText(value) ? Locale.LanguageRange.parse(value) : Collections.emptyList());

480	}

481

482	/**

483	 * Variant of {@link #setAcceptLanguage(List)} using {@link Locale}'s.

484	 * @since 5.0

485	 */

486	public void setAcceptLanguageAsLocales(List<Locale> locales) {

487		setAcceptLanguage(locales.stream()

488				.map(locale -> new Locale.LanguageRange(locale.toLanguageTag()))

489				.collect(Collectors.toList()));

490	}

491

492	/**

493	 * A variant of {@link #getAcceptLanguage()} that converts each

494	 * {@link java.util.Locale.LanguageRange} to a {@link Locale}.

495	 * @return the locales or an empty list

496	 * @throws IllegalArgumentException if the value cannot be converted to a locale

497	 * @since 5.0

498	 */

499	public List<Locale> getAcceptLanguageAsLocales() {

500		List<Locale.LanguageRange> ranges = getAcceptLanguage();

501		if (ranges.isEmpty()) {

502			return Collections.emptyList();

503		}

504		return ranges.stream()

505				.map(range -> Locale.forLanguageTag(range.getRange()))

506				.filter(locale -> StringUtils.hasText(locale.getDisplayName()))

507				.collect(Collectors.toList());

508	}

511	 * Set the (new) value of the {@code Access-Control-Allow-Credentials} response header.

637	/**

638	 * Set the list of acceptable {@linkplain Charset charsets},

639	 * as specified by the {@code Accept-Charset} header.

640	 */

641	public void setAcceptCharset(List<Charset> acceptableCharsets) {

642		StringBuilder builder = new StringBuilder();

643		for (Iterator<Charset> iterator = acceptableCharsets.iterator(); iterator.hasNext();) {

644			Charset charset = iterator.next();

645			builder.append(charset.name().toLowerCase(Locale.ENGLISH));

646			if (iterator.hasNext()) {

647				builder.append(", ");

648			}

649		}

650		set(ACCEPT_CHARSET, builder.toString());

651	}

652

653	/**

654	 * Return the list of acceptable {@linkplain Charset charsets},

655	 * as specified by the {@code Accept-Charset} header.

656	 */

657	public List<Charset> getAcceptCharset() {

658		String value = getFirst(ACCEPT_CHARSET);

659		if (value != null) {

660			String[] tokens = StringUtils.tokenizeToStringArray(value, ",");

661			List<Charset> result = new ArrayList<>(tokens.length);

662			for (String token : tokens) {

663				int paramIdx = token.indexOf(';');

664				String charsetName;

665				if (paramIdx == -1) {

666					charsetName = token;

667				}

668				else {

669					charsetName = token.substring(0, paramIdx);

670				}

671				if (!charsetName.equals("*")) {

672					result.add(Charset.forName(charsetName));

673				}

674			}

675			return result;

676		}

677		else {

678			return Collections.emptyList();

679		}

680	}

955	public void setHost(@Nullable InetSocketAddress host) {

956		if (host != null) {

957			String value = host.getHostString();

958			int port = host.getPort();

959			if (port != 0) {

960				value = value + ":" + port;

969	/**

970	 * Return the value of the required {@code Host} header.

971	 * <p>If the header value does not contain a port, the returned

972	 * {@linkplain InetSocketAddress#getPort() port} will be {@code 0}.

973	 * @since 5.0

974	 */

975	@Nullable

976	public InetSocketAddress getHost() {

977		String value = getFirst(HOST);

978		if (value == null) {

979			return null;

980		}

981

982		String host = null;

983		int port = 0;

984		int separator = (value.startsWith("[") ? value.indexOf(':', value.indexOf(']')) : value.lastIndexOf(':'));

985		if (separator != -1) {

986			host = value.substring(0, separator);

987			String portString = value.substring(separator + 1);

988			try {

989				port = Integer.parseInt(portString);

990			}

991			catch (NumberFormatException ex) {

992				// ignore

993			}

994		}

995

996		if (host == null) {

997			host = value;

998		}

999		return InetSocketAddress.createUnresolved(host, port);

1000	}

1274	/**

1275	 * Parse the first header value for the given header name as a date,

1276	 * return {@code null} if there is no value or also in case of an invalid value

1277	 * (if {@code rejectInvalid=false}), or raise {@link IllegalArgumentException}

1278	 * if the value cannot be parsed as a date.

1279	 * @param headerName the header name

1280	 * @param rejectInvalid whether to reject invalid values with an

1281	 * {@link IllegalArgumentException} ({@code true}) or rather return {@code null}

1282	 * in that case ({@code false})

1283	 * @return the parsed date header, or {@code null} if none (or invalid)

1284 	 */

1285	@Nullable

1286	private ZonedDateTime getFirstZonedDateTime(String headerName, boolean rejectInvalid) {

1287		String headerValue = getFirst(headerName);

1288		if (headerValue == null) {

1289			// No header value sent at all

1290			return null;

1291		}

1292		if (headerValue.length() >= 3) {

1293			// Short "0" or "-1" like values are never valid HTTP date headers...

1294			// Let's only bother with DateTimeFormatter parsing for long enough values.

1295

1296			// See https://stackoverflow.com/questions/12626699/if-modified-since-http-header-passed-by-ie9-includes-length

1297			int parametersIndex = headerValue.indexOf(';');

1298			if (parametersIndex != -1) {

1299				headerValue = headerValue.substring(0, parametersIndex);

1300			}

1301

1302			for (DateTimeFormatter dateFormatter : DATE_FORMATTERS) {

1303				try {

1304					return ZonedDateTime.parse(headerValue, dateFormatter);

1305				}

1306				catch (DateTimeParseException ex) {

1307					// ignore

1308				}

1309			}

1310

1311		}

1312		if (rejectInvalid) {

1313			throw new IllegalArgumentException("Cannot parse date value \"" + headerValue +

1314					"\" for \"" + headerName + "\" header");

1315		}

1316		return null;

1317	}

1319	/**

1320	 * Return all values of a given header name,

1321	 * even if this header is set multiple times.

1322	 * @param headerName the header name

1323	 * @return all associated values

1324	 * @since 4.3

1325	 */

1326	public List<String> getValuesAsList(String headerName) {

1327		List<String> values = get(headerName);

1328		if (values != null) {

1329			List<String> result = new ArrayList<>();

1330			for (String value : values) {

1331				if (value != null) {

1332					String[] tokens = StringUtils.tokenizeToStringArray(value, ",");

1333					for (String token : tokens) {

1334						result.add(token);

1335					}

1336				}

1337			}

1338			return result;

1339		}

1340		return Collections.emptyList();

1341	}

1343	/**

1344	 * Retrieve a combined result from the field values of the ETag header.

1345	 * @param headerName the header name

1346	 * @return the combined result

1347	 * @since 4.3

1348	 */

1349	protected List<String> getETagValuesAsList(String headerName) {

1350		List<String> values = get(headerName);

1351		if (values != null) {

1352			List<String> result = new ArrayList<>();

1353			for (String value : values) {

1354				if (value != null) {

1355					Matcher matcher = ETAG_HEADER_VALUE_PATTERN.matcher(value);

1356					while (matcher.find()) {

1357						if ("*".equals(matcher.group())) {

1358							result.add(matcher.group());

1359						}

1360						else {

1361							result.add(matcher.group(1));

1362						}

1363					}

1364					if (result.isEmpty()) {

1365						throw new IllegalArgumentException(

1366								"Could not parse header '" + headerName + "' with value '" + value + "'");

1367					}

1368				}

1369			}

1370			return result;

1371		}

1372		return Collections.emptyList();

1373	}

1387	/**

1388	 * Turn the given list of header values into a comma-delimited result.

1389	 * @param headerValues the list of header values

1390	 * @return a combined result with comma delimitation

1391	 */

1392	protected String toCommaDelimitedString(List<String> headerValues) {

1393		StringBuilder builder = new StringBuilder();

1394		for (Iterator<String> it = headerValues.iterator(); it.hasNext(); ) {

1395			String val = it.next();

1396			builder.append(val);

1397			if (it.hasNext()) {

1398				builder.append(", ");

1399			}

1400		}

1401		return builder.toString();

1402	}
